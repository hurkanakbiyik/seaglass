28  /**
29   * <p>Represents a built in, or custom, state in Sea Glass.</p>
30   *
31   * <p>Synth provides several built in states, which are:</p>
32   *
33   * <ul>
34   *   <li>Enabled</li>
35   *   <li>Mouse Over</li>
36   *   <li>Pressed</li>
37   *   <li>Disabled</li>
38   *   <li>Focused</li>
39   *   <li>Selected</li>
40   *   <li>Default</li>
41   * </ul>
42   *
43   * <p>However, there are many more states that could be described in a
44   * LookAndFeel, and it would be nice to style components differently based on
45   * these different states. For example, a progress bar could be "indeterminate".
46   * It would be very convenient to allow this to be defined as a "state".</p>
47   *
48   * <p>This class, State, is intended to be used for such situations. Simply
49   * implement the abstract #isInState method. It returns true if the given
50   * JComponent is "in this state", false otherwise. This method will be called
51   * <em>many</em> times in <em>performance sensitive loops</em>. It must execute
52   * very quickly.</p>
53   *
54   * <p>For example, the following might be an implementation of a custom
55   * "Indeterminate" state for JProgressBars:</p>
56   *
57   * <pre>
58   * &lt;code&gt;
59   *     public final class IndeterminateState extends State&lt;JProgressBar&gt; {
60   *         public IndeterminateState() {
61   *             super(&quot;Indeterminate&quot;);
62   *         }
63   *
64   *         &#064;Override
65   *         protected boolean isInState(JProgressBar c) {
66   *             return c.isIndeterminate();
67   *         }
68   *     }
69   * &lt;/code&gt;
70   * </pre>
71   *
72   * <p>Based on Nimbus's State class, which has too much package local stuff.</p>
73   *
74   */
75  public abstract class State<T extends JComponent> {
76      static final Map<String, StandardState> standardStates = new HashMap<String, StandardState>(7);
77  
78      /** Enabled state. */
79      public static final State               Enabled   = new StandardState(SynthConstants.ENABLED);
80  
81      /** MouseOver state. */
82      public static final State               MouseOver = new StandardState(SynthConstants.MOUSE_OVER);
83  
84      /** Pressed state. */
85      public static final State               Pressed  = new StandardState(SynthConstants.PRESSED);
86  
87      /** Disabled state. */
88      public static final State               Disabled = new StandardState(SynthConstants.DISABLED);
89  
90      /** Focused state. */
91      public static final State               Focused  = new StandardState(SynthConstants.FOCUSED);
92  
93      /** Selected state. */
94      public static final State               Selected = new StandardState(SynthConstants.SELECTED);
95  
96      /** Default state. */
97      public static final State               Default = new StandardState(SynthConstants.DEFAULT);
98  
99      private String name;
100 
101     /**
102      * <p>Create a new custom State. Specify the name for the state. The name
103      * should be unique within the states set for any one particular component.
104      * The name of the state should coincide with the name used in UIDefaults.
105      * </p>
106      *
107      * <p>For example, the following would be correct:</p>
108      *
109      * <pre>
110      * &lt;code&gt;
111      *     defaults.put(&quot;Button.States&quot;, &quot;Enabled, Foo, Disabled&quot;);
112      *     defaults.put(&quot;Button.Foo&quot;, new FooState(&quot;Foo&quot;));
113      * &lt;/code&gt;
114      * </pre>
115      *
116      * @param name a simple user friendly name for the state, such as
117      *             "Indeterminate" or "EmbeddedPanel" or "Blurred". It is
118      *             customary to use camel case, with the first letter
119      *             capitalized.
120      */
121     protected State(String name) {
122         this.name = name;
123     }

125     /**
126      * @see java.lang.Object#toString()
127      */
128     public String toString() {
129         return name;
130     }

132     /**
133      * <p>This is the main entry point, called by NimbusStyle.</p>
134      *
135      * <p>There are both custom states and standard states. Standard states
136      * correlate to the states defined in SynthConstants. When a UI delegate
137      * constructs a SynthContext, it specifies the state that the component is
138      * in according to the states defined in SynthConstants. Our NimbusStyle
139      * will then take this state, and query each State instance in the style
140      * asking whether isInState(c, s).</p>
141      *
142      * <p>Now, only the standard states care about the "s" param. So we have
143      * this odd arrangement:</p>
144      *
145      * <ul>
146      *   <li>NimbusStyle calls State.isInState(c, s)</li>
147      *   <li>State.isInState(c, s) simply delegates to State.isInState(c)</li>
148      *   <li><em>EXCEPT</em>, StandardState overrides State.isInState(c, s) and
149      *     returns directly from that method after checking its state, and does
150      *     not call isInState(c) (since it is not needed for standard
151      *     states).</li>
152      * </ul>
153      *
154      * @param  c DOCUMENT ME!
155      * @param  s DOCUMENT ME!
156      *
157      * @return DOCUMENT ME!
158      */
159     public boolean isInState(T c, int s) {
160         return isInState(c);
161     }
162 
163     /**
164      * <p>Gets whether the specified JComponent is in the custom state
165      * represented by this class. <em>This is an extremely performance sensitive
166      * loop.</em> Please take proper precautions to ensure that it executes
167      * quickly.</p>
168      *
169      * <p>Nimbus uses this method to help determine what state a JComponent is
170      * in. For example, a custom State could exist for JProgressBar such that it
171      * would return <code>true</code> when the progress bar is indeterminate.
172      * Such an implementation of this method would simply be:</p>
173      *
174      * <pre>
175      * &lt;code&gt; return c.isIndeterminate();&lt;/code&gt;
176      * </pre>
177      *
178      * @param  c the JComponent to test. This will never be null.
179      *
180      * @return true if <code>c</code> is in the custom state represented by this
181      *         <code>State</code> instance
182      */
183     public abstract boolean isInState(T c);

259         /**
260          * DOCUMENT ME!
261          *
262          * @param  state DOCUMENT ME!
263          *
264          * @return DOCUMENT ME!
265          */
266         private static String toString(int state) {
267             StringBuffer buffer = new StringBuffer();
268 
269             if ((state & SynthConstants.DEFAULT) == SynthConstants.DEFAULT) {
270                 buffer.append("Default");
271             }
272 
273             if ((state & SynthConstants.DISABLED) == SynthConstants.DISABLED) {
274 
275                 if (buffer.length() > 0)
276                     buffer.append("+");
277 
278                 buffer.append("Disabled");
279             }
280 
281             if ((state & SynthConstants.ENABLED) == SynthConstants.ENABLED) {
282 
283                 if (buffer.length() > 0)
284                     buffer.append("+");
285 
286                 buffer.append("Enabled");
287             }
288 
289             if ((state & SynthConstants.FOCUSED) == SynthConstants.FOCUSED) {
290 
291                 if (buffer.length() > 0)
292                     buffer.append("+");
293 
294                 buffer.append("Focused");
295             }
296 
297             if ((state & SynthConstants.MOUSE_OVER) == SynthConstants.MOUSE_OVER) {
298 
299                 if (buffer.length() > 0)
300                     buffer.append("+");
301 
302                 buffer.append("MouseOver");
303             }
304 
305             if ((state & SynthConstants.PRESSED) == SynthConstants.PRESSED) {
306 
307                 if (buffer.length() > 0)
308                     buffer.append("+");
309 
310                 buffer.append("Pressed");
311             }
312 
313             if ((state & SynthConstants.SELECTED) == SynthConstants.SELECTED) {
314 
315                 if (buffer.length() > 0)
316                     buffer.append("+");
317 
318                 buffer.append("Selected");
319             }
320 
321             return buffer.toString();
322         }
