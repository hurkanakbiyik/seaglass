98      /**
99       * Paint the effect based around a solid shape in the graphics supplied.
100      *
101      * @param g the graphics to paint into.
102      * @param s the shape to base the effect around.
103      */
104     public void fill(Graphics2D g, Shape s) {
105         Rectangle bounds = s.getBounds();
106         int       width  = bounds.width;
107         int       height = bounds.height;
108 
109         BufferedImage bimage = Effect.createBufferedImage(width, height, true);
110         Graphics2D    gbi    = bimage.createGraphics();
111 
112         gbi.setColor(Color.BLACK);
113         gbi.fill(s);
114 
115         g.drawImage(applyEffect(bimage, null, width, height), 0, 0, null);
116     }

135     /**
136      * Create a buffered image of the given width and height, compatible with
137      * the alpha requirement.
138      *
139      * @param  width    the width of the new buffered image.
140      * @param  height   the height of the new buffered image.
141      * @param  hasAlpha {@code true} if the new buffered image needs to support
142      *                  alpha transparency, {@code false} otherwise.
143      *
144      * @return the newly created buffered image.
145      */
146     protected static BufferedImage createBufferedImage(int width, int height, boolean hasAlpha) {
147         BufferedImage       bimage = null;
148         GraphicsEnvironment ge     = GraphicsEnvironment.getLocalGraphicsEnvironment();
149 
150         try {
151 
152             // Determine the type of transparency of the new buffered image
153             int transparency = Transparency.OPAQUE;
154 
155             if (hasAlpha) {
156                 transparency = Transparency.TRANSLUCENT;
157             }
158 
159             // Create the buffered image
160             GraphicsDevice        gs = ge.getDefaultScreenDevice();
161             GraphicsConfiguration gc = gs.getDefaultConfiguration();
162 
163             bimage = gc.createCompatibleImage(width, height, transparency);
164         } catch (HeadlessException e) {
165             // The system does not have a screen
166         }
167 
168         if (bimage == null) {
169 
170             // Create a buffered image using the default color model
171             int type = BufferedImage.TYPE_INT_RGB;
172 
173             if (hasAlpha) {
174                 type = BufferedImage.TYPE_INT_ARGB;
175             }
176 
177             bimage = new BufferedImage(width, height, type);
178         }
179 
180         return bimage;
181     }

183     /**
184      * This method returns a buffered image with the contents of an image
185      *
186      * @param  image the image to copy into a buffered image.
187      *
188      * @return the buffered image containing the image.
189      */
190     protected static BufferedImage toBufferedImage(Image image) {
191         if (image instanceof BufferedImage) {
192             return (BufferedImage) image;
193         }
194 
195         // This code ensures that all the pixels in the image are loaded
196         image = new ImageIcon(image).getImage();
197 
198         // Determine if the image has transparent pixels; for this method's
199         // implementation, see e661 Determining If an Image Has Transparent
200         // Pixels
201         boolean hasAlpha = hasAlpha(image);
202 
203         // Create a buffered image with a format that's compatible with the
204         // screen
205         BufferedImage       bimage = null;
206         GraphicsEnvironment ge     = GraphicsEnvironment.getLocalGraphicsEnvironment();
207 
208         try {
209 
210             // Determine the type of transparency of the new buffered image
211             int transparency = Transparency.OPAQUE;
212 
213             if (hasAlpha) {
214                 transparency = Transparency.BITMASK;
215             }
216 
217             // Create the buffered image
218             GraphicsDevice        gs = ge.getDefaultScreenDevice();
219             GraphicsConfiguration gc = gs.getDefaultConfiguration();
220 
221             bimage = gc.createCompatibleImage(image.getWidth(null), image.getHeight(null), transparency);
222         } catch (HeadlessException e) {
223             // The system does not have a screen
224         }
225 
226         if (bimage == null) {
227 
228             // Create a buffered image using the default color model
229             int type = BufferedImage.TYPE_INT_RGB;
230 
231             if (hasAlpha) {
232                 type = BufferedImage.TYPE_INT_ARGB;
233             }
234 
235             bimage = new BufferedImage(image.getWidth(null), image.getHeight(null), type);
236         }
237 
238         // Copy image to buffered image
239         Graphics g = bimage.createGraphics();
240 
241         // Paint the image onto the buffered image
242         g.drawImage(image, 0, 0, null);
243         g.dispose();
244 
245         return bimage;
246     }

248     /**
249      * This method returns true if the specified image has transparent pixels
250      *
251      * @param  image an image.
252      *
253      * @return {@code true} if the image has transparent pixels, {@code false}
254      *         otherwise.
255      */
256     private static boolean hasAlpha(Image image) {
257         // If buffered image, the color model is readily available
258         if (image instanceof BufferedImage) {
259             BufferedImage bimage = (BufferedImage) image;
260 
261             return bimage.getColorModel().hasAlpha();
262         }
263 
264         // Use a pixel grabber to retrieve the image's color model;
265         // grabbing a single pixel is usually sufficient
266         PixelGrabber pg = new PixelGrabber(image, 0, 0, 1, 1, false);
267 
268         try {
269             pg.grabPixels();
270         } catch (InterruptedException e) {
271         }
272 
273         // Get the image's color model
274         ColorModel cm = pg.getColorModel();
275 
276         return cm.hasAlpha();
277     }
