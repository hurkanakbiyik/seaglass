213     /**
214      * Invokes supers implementation of <code>installUI</code> to install the
215      * necessary state onto the passed in <code>JRootPane</code> to render the
216      * metal look and feel implementation of <code>RootPaneUI</code>. If the
217      * <code>windowDecorationStyle</code> property of the <code>JRootPane</code>
218      * is other than <code>JRootPane.NONE</code>, this will add a custom <code>
219      * Component</code> to render the widgets to <code>JRootPane</code>, as well
220      * as installing a custom <code>Border</code> and <code>LayoutManager</code>
221      * on the <code>JRootPane</code>.
222      *
223      * @param c the JRootPane to install state onto.
224      */
225     public void installUI(JComponent c) {
226         super.installUI(c);
227 
228         root = (JRootPane) c;
229 
230         updateTextured();
231 
232         int       style  = root.getWindowDecorationStyle();
233         Container parent = root.getParent();
234 
235         if (parent != null && (parent instanceof JFrame || parent instanceof JDialog) && style != JRootPane.NONE) {
236             installClientDecorations(root);
237         }
238     }

240     /**
241      * Set the textured properties.
242      */
243     private void updateTextured() {
244         // Need the content pane to not be opaque.
245         paintTextured = (root.getClientProperty(UNIFIED_TOOLBAR_LOOK) == Boolean.TRUE);
246 
247         if (paintTextured && PlatformUtils.isMac()) {
248             if (root.isValid()) {
249                 throw new IllegalArgumentException("This method only works if the given JRootPane has not yet been realized.");
250             }
251 
252             root.putClientProperty("apple.awt.brushMetalLook", Boolean.TRUE);
253 
254             // Need the content pane to not be opaque.
255             LookAndFeel.installProperty((JComponent) root.getContentPane(), "opaque", Boolean.FALSE);
256         } else {
257             root.putClientProperty("apple.awt.brushMetalLook", null);
258             // FIXME Restore original content pane opacity.
259         }
260     }

334     /**
335      * Installs the necessary Listeners on the parent <code>Window</code>, if
336      * there is one.
337      *
338      * <p>This takes the parent so that cleanup can be done from <code>
339      * removeNotify</code>, at which point the parent hasn't been reset yet.</p>
340      *
341      * @param root   the JRootPane.
342      * @param parent The parent of the JRootPane
343      */
344     private void installWindowListeners(JRootPane root, Component parent) {
345         if (parent instanceof Window) {
346             window = (Window) parent;
347         } else {
348             window = SwingUtilities.getWindowAncestor(parent);
349         }
350 
351         if (window != null) {
352             if (mouseInputListener == null) {
353                 mouseInputListener = createWindowMouseInputListener(root);
354             }
355 
356             window.addMouseListener(mouseInputListener);
357             window.addMouseMotionListener(mouseInputListener);
358             if (windowListener == null) {
359                 windowListener = createFocusListener();
360                 window.addWindowListener(windowListener);
361             }
362         }
363     }

424     /**
425      * Installs the necessary state onto the JRootPane to render client
426      * decorations. This is ONLY invoked if the <code>JRootPane</code> has a
427      * decoration style other than <code>JRootPane.NONE</code>.
428      *
429      * @param root the JRootPane.
430      */
431     private void installClientDecorations(JRootPane root) {
432         installBorder(root);
433         if (root.getParent() instanceof JFrame || root.getParent() instanceof JDialog) {
434             // Indicate that this frame should not make all the content
435             // draggable. By default, when you set the opacity to 0 (like we do
436             // below) this property automatically gets set to true. Also note
437             // that this client property must be set *before* changing the
438             // opacity (not sure why).
439             root.putClientProperty("apple.awt.draggableWindowBackground", Boolean.FALSE);
440             WindowUtils.makeWindowNonOpaque((Window) root.getParent());
441         }
442 
443         JComponent titlePane = createTitlePane(root);
444 
445         setTitlePane(root, titlePane);
446         installWindowListeners(root, root.getParent());
447         installLayout(root);
448         if (window != null) {
449             root.revalidate();
450             root.repaint();
451         }
452     }
453 
454     /**
455      * Uninstalls any state that <code>installClientDecorations</code> has
456      * installed.
457      *
458      * <p>NOTE: This may be called if you haven't installed client decorations
459      * yet (ie before <code>installClientDecorations</code> has been invoked).
460      * </p>
461      *
462      * @param root the JRootPane.
463      */
464     private void uninstallClientDecorations(JRootPane root) {
465         uninstallBorder(root);
466         uninstallWindowListeners(root);
467         setTitlePane(root, null);
468         uninstallLayout(root);
469         // We have to revalidate/repaint root if the style is JRootPane.NONE
470         // only. When we needs to call revalidate/repaint with other styles
471         // the installClientDecorations is always called after this method
472         // imediatly and it will cause the revalidate/repaint at the proper
473         // time.
474         int style = root.getWindowDecorationStyle();
475 
476         if (style == JRootPane.NONE) {
477             root.repaint();
478             root.revalidate();
479         }
480         // Reset the cursor, as we may have changed it to a resize cursor
481         if (window != null) {
482             window.setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
483         }
484 
485         window = null;
486     }

571     /**
572      * @see javax.swing.plaf.ComponentUI#update(java.awt.Graphics, javax.swing.JComponent)
573      */
574     public void update(Graphics g, JComponent c) {
575         SeaGlassContext context = getContext(c);
576 
577         SeaGlassLookAndFeel.update(context, g);
578         if (((JRootPane) c).getWindowDecorationStyle() != JRootPane.NONE) {
579             context.getPainter().paintRootPaneBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
580         } else if (PlatformUtils.isMac()) {
581             // We may need to paint the rootpane on a Mac if the window is
582             // decorated.
583             boolean   shouldPaint       = false;
584             Container toplevelContainer = c.getParent();
585 
586             if (toplevelContainer instanceof JFrame) {
587                 shouldPaint = !((JFrame) toplevelContainer).isUndecorated();
588             }
589 
590             if (shouldPaint) {
591                 if (!paintTextured) {
592                     g.setColor(c.getBackground());
593                     g.fillRect(0, 0, c.getWidth(), c.getHeight());
594                 } else if (isWindowFocused.isInState(c)) {
595                     contentActive.paint((Graphics2D) g, c, c.getWidth(), c.getHeight());
596                 } else {
597                     contentInactive.paint((Graphics2D) g, c, c.getWidth(), c.getHeight());
598                 }
599             }
600         }
601 
602         paint(context, g);
603         context.dispose();
604     }

633     /**
634      * Invoked when a property changes. <code>AqvavitRootPaneUI</code> is
635      * primarily interested in events originating from the <code>
636      * JRootPane</code> it has been installed on identifying the property <code>
637      * windowDecorationStyle</code>. If the <code>windowDecorationStyle</code>
638      * has changed to a value other than <code>JRootPane.NONE</code>, this will
639      * add a <code>Component</code> to the <code>JRootPane</code> to render the
640      * window decorations, as well as installing a <code>Border</code> on the
641      * <code>JRootPane</code>. On the other hand, if the <code>
642      * windowDecorationStyle</code> has changed to <code>JRootPane.NONE</code>,
643      * this will remove the <code>Component</code> that has been added to the
644      * <code>JRootPane</code> as well resetting the Border to what it was before
645      * <code>installUI</code> was invoked.
646      *
647      * @param e A PropertyChangeEvent object describing the event source and the
648      *          property that has changed.
649      */
650     public void propertyChange(PropertyChangeEvent e) {
651         super.propertyChange(e);
652 
653         String propertyName = e.getPropertyName();
654 
655         if (propertyName == null) {
656             return;
657         }
658 
659         if (propertyName.equals("windowDecorationStyle")) {
660             JRootPane root  = (JRootPane) e.getSource();
661             int       style = root.getWindowDecorationStyle();
662 
663             // This is potentially more than needs to be done,
664             // but it rarely happens and makes the install/uninstall process
665             // simpler. AqvavitTitlePane also assumes it will be recreated if
666             // the decoration style changes.
667             uninstallClientDecorations(root);
668             Container parent = root.getParent();
669 
670             if (parent != null && (parent instanceof JFrame || parent instanceof JDialog) && style != JRootPane.NONE) {
671                 installClientDecorations(root);
672             }
673         } else if (propertyName.equals("ancestor")) {
674             uninstallWindowListeners(root);
675             installWindowListeners(root, root.getParent());
676         } else if (propertyName.equals(UNIFIED_TOOLBAR_LOOK)) {
677             updateTextured();
678         }
679     }

681     /**
682      * A custom layout manager that is responsible for the layout of
683      * layeredPane, glassPane, menuBar and titlePane, if one has been installed.
684      */
685     // NOTE: Ideally this would extends JRootPane.RootLayout, but that
686     // would force this to be non-static.
687     private class SeaGlassRootLayout implements LayoutManager2 {
688 
689         /**
690          * Returns the amount of space the layout would like to have.
691          *
692          * @param  parent the Container for which this layout manager is being
693          *                used
694          *
695          * @return a Dimension object containing the layout's preferred size
696          */
697         public Dimension preferredLayoutSize(Container parent) {
