87      private void updateStyle(JToolBar c) {
88          SeaGlassContext context = getContext(c, Region.TOOL_BAR_CONTENT, null, ENABLED);
89          contentStyle = SeaGlassLookAndFeel.updateStyle(context, this);
90          context.getComponent().setOpaque(false);
91          context.dispose();
92  
93          context = getContext(c, Region.TOOL_BAR_DRAG_WINDOW, null, ENABLED);
94          context.getComponent().setOpaque(false);
95          dragWindowStyle = SeaGlassLookAndFeel.updateStyle(context, this);
96          context.dispose();
97  
98          context = getContext(c, ENABLED);
99          context.getComponent().setOpaque(false);
100         SynthStyle oldStyle = style;
101 
102         style = SeaGlassLookAndFeel.updateStyle(context, this);
103         if (oldStyle != style) {
104             handleIcon = style.getIcon(context, "ToolBar.handleIcon");
105             if (oldStyle != null) {
106                 uninstallKeyboardActions();
107                 installKeyboardActions();
108             }
109         }
110         context.dispose();
111     }

209     protected void paint(SeaGlassContext context, Graphics g) {
210         if (handleIcon != null && toolBar.isFloatable()) {
211             int startX = toolBar.getComponentOrientation().isLeftToRight() ? 0 : toolBar.getWidth()
212                     - SeaGlassIcon.getIconWidth(handleIcon, context);
213             SeaGlassIcon.paintIcon(handleIcon, context, g, startX, 0, SeaGlassIcon.getIconWidth(handleIcon, context), SeaGlassIcon.getIconHeight(
214                 handleIcon, context));
215         }
216 
217         SeaGlassContext subcontext = getContext(toolBar, Region.TOOL_BAR_CONTENT, contentStyle);
218         // paintContent(subcontext, g, contentRect);
219         subcontext.dispose();
220     }

258         public Dimension minimumLayoutSize(Container parent) {
259             JToolBar tb = (JToolBar) parent;
260             Insets insets = tb.getInsets();
261             Dimension dim = new Dimension();
262             SeaGlassContext context = getContext(tb);
263 
264             if (tb.getOrientation() == JToolBar.HORIZONTAL) {
265                 dim.width = tb.isFloatable() ? SeaGlassIcon.getIconWidth(handleIcon, context) : 0;
266                 Dimension compDim;
267                 for (int i = 0; i < tb.getComponentCount(); i++) {
268                     Component component = tb.getComponent(i);
269                     if (component.isVisible()) {
270                         compDim = component.getMinimumSize();
271                         dim.width += compDim.width;
272                         dim.height = Math.max(dim.height, compDim.height);
273                     }
274                 }
275             } else {
276                 dim.height = tb.isFloatable() ? SeaGlassIcon.getIconHeight(handleIcon, context) : 0;
277                 Dimension compDim;
278                 for (int i = 0; i < tb.getComponentCount(); i++) {
279                     Component component = tb.getComponent(i);
280                     if (component.isVisible()) {
281                         compDim = component.getMinimumSize();
282                         dim.width = Math.max(dim.width, compDim.width);
283                         dim.height += compDim.height;
284                     }
285                 }
286             }
287             dim.width += insets.left + insets.right;
288             dim.height += insets.top + insets.bottom;
289 
290             context.dispose();
291             return dim;
292         }

330         public void layoutContainer(Container parent) {
331             JToolBar tb = (JToolBar) parent;
332             Insets insets = tb.getInsets();
333             boolean ltr = tb.getComponentOrientation().isLeftToRight();
334             SeaGlassContext context = getContext(tb);
335 
336             Component c;
337             Dimension d;
338 
339             // JToolBar by default uses a somewhat modified BoxLayout as
340             // its layout manager. For compatibility reasons, we want to
341             // support Box "glue" as a way to move things around on the
342             // toolbar. "glue" is represented in BoxLayout as a Box.Filler
343             // with a minimum and preferred size of (0,0).
344             // So what we do here is find the number of such glue fillers
345             // and figure out how much space should be allocated to them.
346             int glueCount = 0;
347             for (int i = 0; i < tb.getComponentCount(); i++) {
348                 if (isGlue(tb.getComponent(i))) glueCount++;
349             }
350 
351             if (tb.getOrientation() == JToolBar.HORIZONTAL) {
352                 int handleWidth = tb.isFloatable() ? SeaGlassIcon.getIconWidth(handleIcon, context) : 0;
353 
354                 // Note: contentRect does not take insets into account
355                 // since it is used for determining the bounds that are
356                 // passed to paintToolBarContentBackground().
357                 contentRect.x = ltr ? handleWidth : 0;
358                 contentRect.y = 0;
359                 contentRect.width = tb.getWidth() - handleWidth;
360                 contentRect.height = tb.getHeight();
361 
362                 // However, we do take the insets into account here for
363                 // the purposes of laying out the toolbar child components.
364                 int x = ltr ? handleWidth + insets.left : tb.getWidth() - handleWidth - insets.right;
365                 int baseY = insets.top;
366                 int baseH = tb.getHeight() - insets.top - insets.bottom;
367 
368                 // we need to get the minimum width for laying things out
369                 // so that we can calculate how much empty space needs to
370                 // be distributed among the "glue", if any
371                 int extraSpacePerGlue = 0;
372                 if (glueCount > 0) {
373                     int minWidth = minimumLayoutSize(parent).width;
374                     extraSpacePerGlue = (tb.getWidth() - minWidth) / glueCount;
375                     if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
376                 }
377 
378                 for (int i = 0; i < tb.getComponentCount(); i++) {
379                     c = tb.getComponent(i);
380                     if (c.isVisible()) {
381                         d = c.getPreferredSize();
382                         int y, h;
383                         if (d.height >= baseH || c instanceof JSeparator) {
384                             // Fill available height
385                             y = baseY;
386                             h = baseH;
387                         } else {
388                             // Center component vertically in the available
389                             // space
390                             y = baseY + (baseH / 2) - (d.height / 2);
391                             h = d.height;
392                         }
393                         // if the component is a "glue" component then add to
394                         // its
395                         // width the extraSpacePerGlue it is due
396                         if (isGlue(c)) d.width += extraSpacePerGlue;
397                         c.setBounds(ltr ? x : x - d.width, y, d.width, h);
398                         x = ltr ? x + d.width : x - d.width;
399                     }
400                 }
401             } else {
402                 int handleHeight = tb.isFloatable() ? SeaGlassIcon.getIconHeight(handleIcon, context) : 0;
403 
404                 // See notes above regarding the use of insets
405                 contentRect.x = 0;
406                 contentRect.y = handleHeight;
407                 contentRect.width = tb.getWidth();
408                 contentRect.height = tb.getHeight() - handleHeight;
409 
410                 int baseX = insets.left;
411                 int baseW = tb.getWidth() - insets.left - insets.right;
412                 int y = handleHeight + insets.top;
413 
414                 // we need to get the minimum height for laying things out
415                 // so that we can calculate how much empty space needs to
416                 // be distributed among the "glue", if any
417                 int extraSpacePerGlue = 0;
418                 if (glueCount > 0) {
419                     int minHeight = minimumLayoutSize(parent).height;
420                     extraSpacePerGlue = (tb.getHeight() - minHeight) / glueCount;
421                     if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
422                 }
423 
424                 for (int i = 0; i < tb.getComponentCount(); i++) {
425                     c = tb.getComponent(i);
426                     if (c.isVisible()) {
427                         d = c.getPreferredSize();
428                         int x, w;
429                         if (d.width >= baseW || c instanceof JSeparator) {
430                             // Fill available width
431                             x = baseX;
432                             w = baseW;
433                         } else {
434                             // Center component horizontally in the available
435                             // space
436                             x = baseX + (baseW / 2) - (d.width / 2);
437                             w = d.width;
438                         }
439                         // if the component is a "glue" component then add to
440                         // its
441                         // height the extraSpacePerGlue it is due
442                         if (isGlue(c)) d.height += extraSpacePerGlue;
443                         c.setBounds(x, y, w, d.height);
444                         y += d.height;
445                     }
446                 }
447             }
448             context.dispose();
449         }

451         private boolean isGlue(Component c) {
452             if (c.isVisible() && c instanceof Box.Filler) {
453                 Box.Filler f = (Box.Filler) c;
454                 Dimension min = f.getMinimumSize();
455                 Dimension pref = f.getPreferredSize();
456                 return min.width == 0 && min.height == 0 && pref.width == 0 && pref.height == 0;
457             }
458             return false;
459         }
