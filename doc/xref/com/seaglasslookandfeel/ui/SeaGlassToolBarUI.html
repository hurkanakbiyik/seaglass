89      private void updateStyle(JToolBar c) {
90          SeaGlassContext context = getContext(c, Region.TOOL_BAR_CONTENT, null, ENABLED);
91          contentStyle = SeaGlassLookAndFeel.updateStyle(context, this);
92          context.getComponent().setOpaque(false);
93          context.dispose();
94  
95          context = getContext(c, Region.TOOL_BAR_DRAG_WINDOW, null, ENABLED);
96          context.getComponent().setOpaque(false);
97          dragWindowStyle = SeaGlassLookAndFeel.updateStyle(context, this);
98          context.dispose();
99  
100         context = getContext(c, ENABLED);
101         context.getComponent().setOpaque(false);
102         SynthStyle oldStyle = style;
103 
104         style = SeaGlassLookAndFeel.updateStyle(context, this);
105         if (oldStyle != style) {
106             handleIcon = style.getIcon(context, "ToolBar.handleIcon");
107             if (oldStyle != null) {
108                 uninstallKeyboardActions();
109                 installKeyboardActions();
110             }
111         }
112         context.dispose();
113     }

211     protected void paint(SeaGlassContext context, Graphics g) {
212         if (handleIcon != null && toolBar.isFloatable()) {
213             int startX = toolBar.getComponentOrientation().isLeftToRight() ? 0 : toolBar.getWidth()
214                     - SynthIcon.getIconWidth(handleIcon, context);
215             SynthIcon.paintIcon(handleIcon, context, g, startX, 0, SynthIcon.getIconWidth(handleIcon, context), SynthIcon.getIconHeight(
216                 handleIcon, context));
217         }
218 
219         SeaGlassContext subcontext = getContext(toolBar, Region.TOOL_BAR_CONTENT, contentStyle);
220         // paintContent(subcontext, g, contentRect);
221         subcontext.dispose();
222     }

260         public Dimension minimumLayoutSize(Container parent) {
261             JToolBar tb = (JToolBar) parent;
262             Insets insets = tb.getInsets();
263             Dimension dim = new Dimension();
264             SeaGlassContext context = getContext(tb);
265 
266             if (tb.getOrientation() == JToolBar.HORIZONTAL) {
267                 dim.width = tb.isFloatable() ? SynthIcon.getIconWidth(handleIcon, context) : 0;
268                 Dimension compDim;
269                 for (int i = 0; i < tb.getComponentCount(); i++) {
270                     Component component = tb.getComponent(i);
271                     if (component.isVisible()) {
272                         compDim = component.getMinimumSize();
273                         dim.width += compDim.width;
274                         dim.height = Math.max(dim.height, compDim.height);
275                     }
276                 }
277             } else {
278                 dim.height = tb.isFloatable() ? SynthIcon.getIconHeight(handleIcon, context) : 0;
279                 Dimension compDim;
280                 for (int i = 0; i < tb.getComponentCount(); i++) {
281                     Component component = tb.getComponent(i);
282                     if (component.isVisible()) {
283                         compDim = component.getMinimumSize();
284                         dim.width = Math.max(dim.width, compDim.width);
285                         dim.height += compDim.height;
286                     }
287                 }
288             }
289             dim.width += insets.left + insets.right;
290             dim.height += insets.top + insets.bottom;
291 
292             context.dispose();
293             return dim;
294         }

332         public void layoutContainer(Container parent) {
333             JToolBar tb = (JToolBar) parent;
334             Insets insets = tb.getInsets();
335             boolean ltr = tb.getComponentOrientation().isLeftToRight();
336             SeaGlassContext context = getContext(tb);
337 
338             Component c;
339             Dimension d;
340 
341             // JToolBar by default uses a somewhat modified BoxLayout as
342             // its layout manager. For compatibility reasons, we want to
343             // support Box "glue" as a way to move things around on the
344             // toolbar. "glue" is represented in BoxLayout as a Box.Filler
345             // with a minimum and preferred size of (0,0).
346             // So what we do here is find the number of such glue fillers
347             // and figure out how much space should be allocated to them.
348             int glueCount = 0;
349             for (int i = 0; i < tb.getComponentCount(); i++) {
350                 if (isGlue(tb.getComponent(i))) glueCount++;
351             }
352 
353             if (tb.getOrientation() == JToolBar.HORIZONTAL) {
354                 int handleWidth = tb.isFloatable() ? SynthIcon.getIconWidth(handleIcon, context) : 0;
355 
356                 // Note: contentRect does not take insets into account
357                 // since it is used for determining the bounds that are
358                 // passed to paintToolBarContentBackground().
359                 contentRect.x = ltr ? handleWidth : 0;
360                 contentRect.y = 0;
361                 contentRect.width = tb.getWidth() - handleWidth;
362                 contentRect.height = tb.getHeight();
363 
364                 // However, we do take the insets into account here for
365                 // the purposes of laying out the toolbar child components.
366                 int x = ltr ? handleWidth + insets.left : tb.getWidth() - handleWidth - insets.right;
367                 int baseY = insets.top;
368                 int baseH = tb.getHeight() - insets.top - insets.bottom;
369 
370                 // we need to get the minimum width for laying things out
371                 // so that we can calculate how much empty space needs to
372                 // be distributed among the "glue", if any
373                 int extraSpacePerGlue = 0;
374                 if (glueCount > 0) {
375                     int minWidth = minimumLayoutSize(parent).width;
376                     extraSpacePerGlue = (tb.getWidth() - minWidth) / glueCount;
377                     if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
378                 }
379 
380                 for (int i = 0; i < tb.getComponentCount(); i++) {
381                     c = tb.getComponent(i);
382                     if (c.isVisible()) {
383                         d = c.getPreferredSize();
384                         int y, h;
385                         if (d.height >= baseH || c instanceof JSeparator) {
386                             // Fill available height
387                             y = baseY;
388                             h = baseH;
389                         } else {
390                             // Center component vertically in the available
391                             // space
392                             y = baseY + (baseH / 2) - (d.height / 2);
393                             h = d.height;
394                         }
395                         // if the component is a "glue" component then add to
396                         // its
397                         // width the extraSpacePerGlue it is due
398                         if (isGlue(c)) d.width += extraSpacePerGlue;
399                         c.setBounds(ltr ? x : x - d.width, y, d.width, h);
400                         x = ltr ? x + d.width : x - d.width;
401                     }
402                 }
403             } else {
404                 int handleHeight = tb.isFloatable() ? SynthIcon.getIconHeight(handleIcon, context) : 0;
405 
406                 // See notes above regarding the use of insets
407                 contentRect.x = 0;
408                 contentRect.y = handleHeight;
409                 contentRect.width = tb.getWidth();
410                 contentRect.height = tb.getHeight() - handleHeight;
411 
412                 int baseX = insets.left;
413                 int baseW = tb.getWidth() - insets.left - insets.right;
414                 int y = handleHeight + insets.top;
415 
416                 // we need to get the minimum height for laying things out
417                 // so that we can calculate how much empty space needs to
418                 // be distributed among the "glue", if any
419                 int extraSpacePerGlue = 0;
420                 if (glueCount > 0) {
421                     int minHeight = minimumLayoutSize(parent).height;
422                     extraSpacePerGlue = (tb.getHeight() - minHeight) / glueCount;
423                     if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
424                 }
425 
426                 for (int i = 0; i < tb.getComponentCount(); i++) {
427                     c = tb.getComponent(i);
428                     if (c.isVisible()) {
429                         d = c.getPreferredSize();
430                         int x, w;
431                         if (d.width >= baseW || c instanceof JSeparator) {
432                             // Fill available width
433                             x = baseX;
434                             w = baseW;
435                         } else {
436                             // Center component horizontally in the available
437                             // space
438                             x = baseX + (baseW / 2) - (d.width / 2);
439                             w = d.width;
440                         }
441                         // if the component is a "glue" component then add to
442                         // its
443                         // height the extraSpacePerGlue it is due
444                         if (isGlue(c)) d.height += extraSpacePerGlue;
445                         c.setBounds(x, y, w, d.height);
446                         y += d.height;
447                     }
448                 }
449             }
450             context.dispose();
451         }

453         private boolean isGlue(Component c) {
454             if (c.isVisible() && c instanceof Box.Filler) {
455                 Box.Filler f = (Box.Filler) c;
456                 Dimension min = f.getMinimumSize();
457                 Dimension pref = f.getPreferredSize();
458                 return min.width == 0 && min.height == 0 && pref.width == 0 && pref.height == 0;
459             }
460             return false;
461         }
