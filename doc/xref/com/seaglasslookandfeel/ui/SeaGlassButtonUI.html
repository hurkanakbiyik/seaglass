94      /**
95       * Update the style of the button.
96       *
97       * @param b the button.
98       */
99      public void updateStyle(AbstractButton b) {
100         SeaGlassContext context  = getContext(b, SynthConstants.ENABLED);
101         SynthStyle      oldStyle = style;
102 
103         style = SeaGlassLookAndFeel.updateStyle(context, this);
104         if (style != oldStyle) {
105             if (b.getMargin() == null || (b.getMargin() instanceof UIResource)) {
106                 Insets margin = (Insets) style.get(context, getPropertyPrefix() + "margin");
107 
108                 if (margin == null) {
109                     // Some places assume margins are non-null.
110                     margin = SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS;
111                 }
112 
113                 b.setMargin(margin);
114             }
115 
116             Object value = style.get(context, getPropertyPrefix() + "iconTextGap");
117 
118             if (value != null) {
119                 LookAndFeel.installProperty(b, "iconTextGap", value);
120             }
121 
122             value = style.get(context, getPropertyPrefix() + "contentAreaFilled");
123             LookAndFeel.installProperty(b, "contentAreaFilled", value != null ? value : Boolean.TRUE);
124 
125             value = b.getClientProperty(APPLE_PREFIX + "buttonType");
126             if (value != null) {
127                 if ("segmented".equals(value)) {
128                     b.setMargin(SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS);
129                 }
130             }
131 
132             if (oldStyle != null) {
133                 uninstallKeyboardActions(b);
134                 installKeyboardActions(b);
135             }
136         }
137 
138         context.dispose();
139     }

192     /**
193      * Returns the current state of the passed in <code>AbstractButton</code>.
194      *
195      * @param  c the button component.
196      *
197      * @return the button's state.
198      */
199     private int getComponentState(JComponent c) {
200         int state = ENABLED;
201 
202         if (!c.isEnabled()) {
203             state = DISABLED;
204         }
205 
206         if (SeaGlassLookAndFeel.selectedUI == this) {
207             return SeaGlassLookAndFeel.selectedUIState | SynthConstants.ENABLED;
208         }
209 
210         AbstractButton button = (AbstractButton) c;
211         ButtonModel    model  = button.getModel();
212 
213         if (model.isPressed()) {
214             if (model.isArmed()) {
215                 state = PRESSED;
216             } else {
217                 state = MOUSE_OVER;
218             }
219         }
220 
221         if (model.isRollover()) {
222             state |= MOUSE_OVER;
223         }
224 
225         if (model.isSelected()) {
226             state |= SELECTED;
227         }
228 
229         if (c.isFocusOwner() && button.isFocusPainted()) {
230             state |= FOCUSED;
231         }
232 
233         if ((c instanceof JButton) && ((JButton) c).isDefaultButton()) {
234             state |= DEFAULT;
235         }
236 
237         return state;
238     }

240     /**
241      * @see javax.swing.plaf.basic.BasicButtonUI#getBaseline(javax.swing.JComponent,
242      *      int, int)
243      */
244     public int getBaseline(JComponent c, int width, int height) {
245         if (c == null) {
246             throw new NullPointerException("Component must be non-null");
247         }
248 
249         if (width < 0 || height < 0) {
250             throw new IllegalArgumentException("Width and height must be >= 0");
251         }
252 
253         AbstractButton b    = (AbstractButton) c;
254         String         text = b.getText();
255 
256         if (text == null || "".equals(text)) {
257             return -1;
258         }
259 
260         Insets    i        = b.getInsets();
261         Rectangle viewRect = new Rectangle();
262         Rectangle textRect = new Rectangle();
263         Rectangle iconRect = new Rectangle();
264 
265         viewRect.x      = i.left;
266         viewRect.y      = i.top;
267         viewRect.width  = width - (i.right + viewRect.x);
268         viewRect.height = height - (i.bottom + viewRect.y);
269 
270         // layout the text and icon
271         SeaGlassContext context = getContext(b);
272         FontMetrics     fm = context.getComponent().getFontMetrics(context.getStyle().getFont(context));
273 
274         context.getStyle().getGraphicsUtils(context).layoutText(context, fm, b.getText(), b.getIcon(), b.getHorizontalAlignment(),
275                                                                 b.getVerticalAlignment(), b.getHorizontalTextPosition(),
276                                                                 b.getVerticalTextPosition(), viewRect, iconRect, textRect,
277                                                                 b.getIconTextGap());
278         View view     = (View) b.getClientProperty(BasicHTML.propertyKey);
279         int  baseline;
280 
281         if (view != null) {
282             baseline = BasicHTML.getHTMLBaseline(view, textRect.width, textRect.height);
283             if (baseline >= 0) {
284                 baseline += textRect.y;
285             }
286         } else {
287             baseline = textRect.y + fm.getAscent();
288         }
289 
290         context.dispose();
291         return baseline;
292     }

374     /**
375      * Returns the Icon to use in painting the button.
376      *
377      * @param  b the button.
378      *
379      * @return the icon.
380      */
381     protected Icon getIcon(AbstractButton b) {
382         Icon        icon  = b.getIcon();
383         ButtonModel model = b.getModel();
384 
385         if (!model.isEnabled()) {
386             icon = getSynthDisabledIcon(b, icon);
387         } else if (model.isPressed() && model.isArmed()) {
388             icon = getPressedIcon(b, getSelectedIcon(b, icon));
389         } else if (b.isRolloverEnabled() && model.isRollover()) {
390             icon = getRolloverIcon(b, getSelectedIcon(b, icon));
391         } else if (model.isSelected()) {
392             icon = getSelectedIcon(b, icon);
393         } else {
394             icon = getEnabledIcon(b, icon);
395         }
396 
397         if (icon == null) {
398             return getDefaultIcon(b);
399         }
400 
401         return icon;
402     }

404     /**
405      * This method will return the icon that should be used for a button. We
406      * only want to use the synth icon defined by the style if the specific icon
407      * has not been defined for the button state and the backup icon is a
408      * UIResource (we set it, not the developer).
409      *
410      * @param  b            button
411      * @param  specificIcon icon returned from the button for the specific state
412      * @param  defaultIcon  fallback icon
413      * @param  state        the synth state of the button
414      *
415      * @return the icon.
416      */
417     private Icon getIcon(AbstractButton b, Icon specificIcon, Icon defaultIcon, int state) {
418         Icon icon = specificIcon;
419 
420         if (icon == null) {
421             if (defaultIcon instanceof UIResource) {
422                 icon = getSynthIcon(b, state);
423                 if (icon == null) {
424                     icon = defaultIcon;
425                 }
426             } else {
427                 icon = defaultIcon;
428             }
429         }
430 
431         return icon;
432     }
