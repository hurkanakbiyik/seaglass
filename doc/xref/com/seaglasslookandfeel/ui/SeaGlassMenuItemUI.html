83      static Dimension getPreferredMenuItemSize(SeaGlassContext context, SeaGlassContext accContext, boolean useCheckAndArrow, JComponent c,
84          Icon checkIcon, Icon arrowIcon, int defaultTextIconGap, String acceleratorDelimiter) {
85          JMenuItem b = (JMenuItem) c;
86          Icon icon = (Icon) b.getIcon();
87          String text = b.getText();
88          KeyStroke accelerator = b.getAccelerator();
89          String acceleratorText = "";
90  
91          if (accelerator != null) {
92              int modifiers = accelerator.getModifiers();
93              if (modifiers > 0) {
94                  acceleratorText = KeyEvent.getKeyModifiersText(modifiers);
95                  acceleratorText += acceleratorDelimiter;
96              }
97              int keyCode = accelerator.getKeyCode();
98              if (keyCode != 0) {
99                  acceleratorText += KeyEvent.getKeyText(keyCode);
100             } else {
101                 acceleratorText += accelerator.getKeyChar();
102             }
103         }
104 
105         Font font = context.getStyle().getFont(context);
106         FontMetrics fm = b.getFontMetrics(font);
107         FontMetrics fmAccel = b.getFontMetrics(accContext.getStyle().getFont(accContext));
108 
109         resetRects();
110 
111         layoutMenuItem(context, fm, accContext, text, fmAccel, acceleratorText, icon, checkIcon, arrowIcon, b.getVerticalAlignment(), b
112             .getHorizontalAlignment(), b.getVerticalTextPosition(), b.getHorizontalTextPosition(), viewRect, iconRect, textRect,
113             acceleratorRect, checkIconRect, arrowIconRect, text == null ? 0 : defaultTextIconGap, defaultTextIconGap, useCheckAndArrow);
114         // find the union of the icon and text rects
115         r.setBounds(textRect);
116         r = SwingUtilities.computeUnion(iconRect.x, iconRect.y, iconRect.width, iconRect.height, r);
117 
118         // To make the accelerator texts appear in a column,
119         // find the widest MenuItem text and the widest accelerator text.
120 
121         // Get the parent, which stores the information.
122         Container parent = b.getParent();
123 
124         if (parent instanceof JPopupMenu) {
125             SeaGlassPopupMenuUI popupUI = (SeaGlassPopupMenuUI) SeaGlassLookAndFeel.getUIOfType(((JPopupMenu) parent).getUI(),
126                 SeaGlassPopupMenuUI.class);
127 
128             if (popupUI != null) {
129                 r.width = popupUI.adjustTextWidth(r.width);
130 
131                 popupUI.adjustAcceleratorWidth(acceleratorRect.width);
132 
133                 r.width += popupUI.getMaxAcceleratorWidth();
134             }
135         } else if (parent != null && !(b instanceof JMenu && ((JMenu) b).isTopLevelMenu())) {
136             r.width += acceleratorRect.width;
137         }
138 
139         if (useCheckAndArrow) {
140             // Add in the checkIcon
141             r.width += checkIconRect.width;
142             r.width += defaultTextIconGap;
143 
144             // Add in the arrowIcon
145             r.width += defaultTextIconGap;
146             r.width += arrowIconRect.width;
147         }
148 
149         r.width += 2 * defaultTextIconGap;
150 
151         Insets insets = b.getInsets();
152         if (insets != null) {
153             r.width += insets.left + insets.right;
154             r.height += insets.top + insets.bottom;
155         }
156 
157         // if the width is even, bump it up one. This is critical
158         // for the focus dash line to draw properly
159         if (r.width % 2 == 0) {
160             r.width++;
161         }
162 
163         // if the height is even, bump it up one. This is critical
164         // for the text to center properly
165         if (r.height % 2 == 0) {
166             r.height++;
167         }
168         return r.getSize();
169     }

171     static void paint(SeaGlassContext context, SeaGlassContext accContext, Graphics g, Icon checkIcon, Icon arrowIcon,
172         boolean useCheckAndArrow, String acceleratorDelimiter, int defaultTextIconGap) {
173         JComponent c = context.getComponent();
174         JMenuItem b = (JMenuItem) c;
175         ButtonModel model = b.getModel();
176         Insets i = b.getInsets();
177 
178         resetRects();
179 
180         viewRect.setBounds(0, 0, b.getWidth(), b.getHeight());
181 
182         viewRect.x += i.left;
183         viewRect.y += i.top;
184         viewRect.width -= (i.right + viewRect.x);
185         viewRect.height -= (i.bottom + viewRect.y);
186 
187         SynthStyle style = context.getStyle();
188         Font f = style.getFont(context);
189         g.setFont(f);
190         FontMetrics fm = SwingUtilities2.getFontMetrics(c, g, f);
191         FontMetrics accFM = SwingUtilities2.getFontMetrics(c, g, accContext.getStyle().getFont(accContext));
192 
193         // get Accelerator text
194         KeyStroke accelerator = b.getAccelerator();
195         String acceleratorText = "";
196         if (accelerator != null) {
197             int modifiers = accelerator.getModifiers();
198             if (modifiers > 0) {
199                 acceleratorText = KeyEvent.getKeyModifiersText(modifiers);
200                 acceleratorText += acceleratorDelimiter;
201             }
202 
203             int keyCode = accelerator.getKeyCode();
204             if (keyCode != 0) {
205                 acceleratorText += KeyEvent.getKeyText(keyCode);
206             } else {
207                 acceleratorText += accelerator.getKeyChar();
208             }
209         }
210 
211         // layoutl the text and icon
212         String text = layoutMenuItem(context, fm, accContext, b.getText(), accFM, acceleratorText, b.getIcon(), checkIcon, arrowIcon, b
213             .getVerticalAlignment(), b.getHorizontalAlignment(), b.getVerticalTextPosition(), b.getHorizontalTextPosition(), viewRect,
214             iconRect, textRect, acceleratorRect, checkIconRect, arrowIconRect, b.getText() == null ? 0 : defaultTextIconGap,
215             defaultTextIconGap, useCheckAndArrow);
216 
217         // Paint the Check
218         if (checkIcon != null && useCheckAndArrow) {
219             SynthIcon.paintIcon(checkIcon, context, g, checkIconRect.x, checkIconRect.y, checkIconRect.width, checkIconRect.height);
220         }
221 
222         // Paint the Icon
223         if (b.getIcon() != null) {
224             Icon icon;
225             if (!model.isEnabled()) {
226                 icon = (Icon) b.getDisabledIcon();
227             } else if (model.isPressed() && model.isArmed()) {
228                 icon = (Icon) b.getPressedIcon();
229                 if (icon == null) {
230                     // Use default icon
231                     icon = (Icon) b.getIcon();
232                 }
233             } else {
234                 icon = (Icon) b.getIcon();
235             }
236 
237             if (icon != null) {
238                 SynthIcon.paintIcon(icon, context, g, iconRect.x, iconRect.y, iconRect.width, iconRect.height);
239             }
240         }
241 
242         // Draw the Text
243         if (text != null) {
244             View v = (View) c.getClientProperty(BasicHTML.propertyKey);
245             if (v != null) {
246                 v.paint(g, textRect);
247             } else {
248                 g.setColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
249                 g.setFont(style.getFont(context));
250                 style.getGraphicsUtils(context).paintText(context, g, text, textRect.x, textRect.y, b.getDisplayedMnemonicIndex());
251             }
252         }
253 
254         // Draw the Accelerator Text
255         if (acceleratorText != null && !acceleratorText.equals("")) {
256             // Get the maxAccWidth from the parent to calculate the offset.
257             int accOffset = 0;
258             Container parent = b.getParent();
259             if (parent != null && parent instanceof JPopupMenu) {
260                 SeaGlassPopupMenuUI popupUI = (SeaGlassPopupMenuUI) ((JPopupMenu) parent).getUI();
261                 if (popupUI != null) {
262                     // Calculate the offset, with which the accelerator texts
263                     // will be drawn with.
264                     int max = popupUI.getMaxAcceleratorWidth();
265                     if (max > 0) {
266                         accOffset = max - acceleratorRect.width;
267                     }
268                 }
269             }
270 
271             SynthStyle accStyle = accContext.getStyle();
272 
273             g.setColor(accStyle.getColor(accContext, ColorType.TEXT_FOREGROUND));
274             g.setFont(accStyle.getFont(accContext));
275             accStyle.getGraphicsUtils(accContext).paintText(accContext, g, acceleratorText, acceleratorRect.x - accOffset,
276                 acceleratorRect.y, -1);
277         }
278 
279         // Paint the Arrow
280         if (arrowIcon != null && useCheckAndArrow) {
281             SynthIcon.paintIcon(arrowIcon, context, g, arrowIconRect.x, arrowIconRect.y, arrowIconRect.width, arrowIconRect.height);
282         }
283     }

285     /**
286      * Compute and return the location of the icons origin, the location of
287      * origin of the text baseline, and a possibly clipped version of the
288      * compound labels string. Locations are computed relative to the viewRect
289      * rectangle.
290      */

292     private static String layoutMenuItem(SeaGlassContext context, FontMetrics fm, SeaGlassContext accContext, String text,
293         FontMetrics fmAccel, String acceleratorText, Icon icon, Icon checkIcon, Icon arrowIcon, int verticalAlignment,
294         int horizontalAlignment, int verticalTextPosition, int horizontalTextPosition, Rectangle viewRect, Rectangle iconRect,
295         Rectangle textRect, Rectangle acceleratorRect, Rectangle checkIconRect, Rectangle arrowIconRect, int textIconGap, int menuItemGap,
296         boolean useCheckAndArrow) {
297         // If parent is JPopupMenu, get and store it's UI
298         SeaGlassPopupMenuUI popupUI = null;
299         JComponent b = context.getComponent();
300         Container parent = b.getParent();
301         if (parent instanceof JPopupMenu) {
302             popupUI = (SeaGlassPopupMenuUI) SeaGlassLookAndFeel.getUIOfType(((JPopupMenu) parent).getUI(), SeaGlassPopupMenuUI.class);
303         }
304 
305         context.getStyle().getGraphicsUtils(context).layoutText(context, fm, text, icon, horizontalAlignment, verticalAlignment,
306             horizontalTextPosition, verticalTextPosition, viewRect, iconRect, textRect, textIconGap);
307 
308         /*
309          * Initialize the acceelratorText bounds rectangle textRect. If a null
310          * or and empty String was specified we substitute "" here and use
311          * 0,0,0,0 for acceleratorTextRect.
312          */
313         if ((acceleratorText == null) || acceleratorText.equals("")) {
314             acceleratorRect.width = acceleratorRect.height = 0;
315             acceleratorText = "";
316         } else {
317             SynthStyle style = accContext.getStyle();
318             acceleratorRect.width = style.getGraphicsUtils(accContext).computeStringWidth(accContext, fmAccel.getFont(), fmAccel,
319                 acceleratorText);
320             acceleratorRect.height = fmAccel.getHeight();
321         }
322 
323         /*
324          * Initialize the checkIcon bounds rectangle's width & height.
325          */
326 
327         if (useCheckAndArrow) {
328             if (checkIcon != null) {
329                 checkIconRect.width = SynthIcon.getIconWidth(checkIcon, context);
330                 checkIconRect.height = SynthIcon.getIconHeight(checkIcon, context);
331             } else {
332                 checkIconRect.width = checkIconRect.height = 0;
333             }
334 
335             /*
336              * Initialize the arrowIcon bounds rectangle width & height.
337              */
338 
339             if (arrowIcon != null) {
340                 arrowIconRect.width = SynthIcon.getIconWidth(arrowIcon, context);
341                 arrowIconRect.height = SynthIcon.getIconHeight(arrowIcon, context);
342             } else {
343                 arrowIconRect.width = arrowIconRect.height = 0;
344             }
345         }
346 
347         Rectangle labelRect = iconRect.union(textRect);
348         if (SeaGlassLookAndFeel.isLeftToRight(context.getComponent())) {
349             textRect.x += menuItemGap;
350             iconRect.x += menuItemGap;
351 
352             // Position the Accelerator text rect
353             acceleratorRect.x = viewRect.x + viewRect.width - arrowIconRect.width - menuItemGap - acceleratorRect.width;
354 
355             // Position the Check and Arrow Icons
356             if (useCheckAndArrow) {
357                 checkIconRect.x = viewRect.x + menuItemGap;
358                 textRect.x += menuItemGap + checkIconRect.width;
359                 iconRect.x += menuItemGap + checkIconRect.width;
360                 arrowIconRect.x = viewRect.x + viewRect.width - menuItemGap - arrowIconRect.width;
361             }
362             /* Align icons and text horizontally */
363             if (popupUI != null) {
364                 int thisTextOffset = popupUI.adjustTextOffset(textRect.x - viewRect.x);
365                 textRect.x = thisTextOffset + viewRect.x;
366 
367                 if (icon != null) {
368                     if (horizontalTextPosition == SwingConstants.TRAILING || horizontalTextPosition == SwingConstants.RIGHT) {
369                         int thisIconOffset = popupUI.adjustIconOffset(iconRect.x - viewRect.x);
370                         iconRect.x = thisIconOffset + viewRect.x;
371                     } else if (horizontalTextPosition == SwingConstants.LEADING || horizontalTextPosition == SwingConstants.LEFT) {
372                         iconRect.x = textRect.x + textRect.width + menuItemGap;
373                     } else {
374                         int maxIconValue = popupUI.adjustIconOffset(0);
375                         iconRect.x = Math.max(textRect.x + textRect.width / 2 - iconRect.width / 2, maxIconValue + viewRect.x);
376                     }
377                 }
378             }
379         } else {
380             textRect.x -= menuItemGap;
381             iconRect.x -= menuItemGap;
382 
383             // Position the Accelerator text rect
384             acceleratorRect.x = viewRect.x + arrowIconRect.width + menuItemGap;
385 
386             // Position the Check and Arrow Icons
387             if (useCheckAndArrow) {
388                 checkIconRect.x = viewRect.x + viewRect.width - menuItemGap - checkIconRect.width;
389                 textRect.x -= menuItemGap + checkIconRect.width;
390                 iconRect.x -= menuItemGap + checkIconRect.width;
391                 arrowIconRect.x = viewRect.x + menuItemGap;
392             }
393             /* Align icons and text horizontally */
394             if (popupUI != null) {
395                 int thisTextOffset = viewRect.x + viewRect.width - textRect.x - textRect.width;
396                 thisTextOffset = popupUI.adjustTextOffset(thisTextOffset);
397                 textRect.x = viewRect.x + viewRect.width - thisTextOffset - textRect.width;
398                 if (icon != null) {
399                     if (horizontalTextPosition == SwingConstants.TRAILING || horizontalTextPosition == SwingConstants.LEFT) {
400                         int thisIconOffset = viewRect.x + viewRect.width - iconRect.x - iconRect.width;
401                         thisIconOffset = popupUI.adjustIconOffset(thisIconOffset);
402                         iconRect.x = viewRect.x + viewRect.width - thisIconOffset - iconRect.width;
403 
404                     } else if (horizontalTextPosition == SwingConstants.LEADING || horizontalTextPosition == SwingConstants.RIGHT) {
405                         iconRect.x = textRect.x - menuItemGap - iconRect.width;
406                     } else {
407                         int maxIconValue = popupUI.adjustIconOffset(0);
408                         iconRect.x = textRect.x + textRect.width / 2 - iconRect.width / 2;
409                         if (iconRect.x + iconRect.width > viewRect.x + viewRect.width - maxIconValue) {
410                             iconRect.x = iconRect.x = viewRect.x + viewRect.width - maxIconValue - iconRect.width;
411                         }
412                     }
413                 }
414             }
415         }
416 
417         // Align the accelertor text and the check and arrow icons vertically
418         // with the center of the label rect.
419         acceleratorRect.y = labelRect.y + (labelRect.height / 2) - (acceleratorRect.height / 2);
420         if (useCheckAndArrow) {
421             arrowIconRect.y = labelRect.y + (labelRect.height / 2) - (arrowIconRect.height / 2);
422             checkIconRect.y = labelRect.y + (labelRect.height / 2) - (checkIconRect.height / 2);
423         }
424 
425         return text;
426     }

457     private void updateStyle(JMenuItem mi) {
458         SeaGlassContext context = getContext(mi, ENABLED);
459         SynthStyle oldStyle = style;
460 
461         style = SeaGlassLookAndFeel.updateStyle(context, this);
462         if (oldStyle != style) {
463             String prefix = getPropertyPrefix();
464 
465             Object value = style.get(context, prefix + ".textIconGap");
466             if (value != null) {
467                 LookAndFeel.installProperty(mi, "iconTextGap", value);
468             }
469             defaultTextIconGap = mi.getIconTextGap();
470 
471             if (menuItem.getMargin() == null || (menuItem.getMargin() instanceof UIResource)) {
472                 Insets insets = (Insets) style.get(context, prefix + ".margin");
473 
474                 if (insets == null) {
475                     // Some places assume margins are non-null.
476                     insets = SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS;
477                 }
478                 menuItem.setMargin(insets);
479             }
480             acceleratorDelimiter = style.getString(context, prefix + ".acceleratorDelimiter", "+");
481 
482             arrowIcon = style.getIcon(context, prefix + ".arrowIcon");
483 
484             checkIcon = style.getIcon(context, prefix + ".checkIcon");
485             if (oldStyle != null) {
486                 uninstallKeyboardActions();
487                 installKeyboardActions();
488             }
489         }
490         context.dispose();
491 
492         SeaGlassContext accContext = getContext(mi, Region.MENU_ITEM_ACCELERATOR, ENABLED);
493 
494         accStyle = SeaGlassLookAndFeel.updateStyle(accContext, this);
495         accContext.dispose();
496     }

533     private int getComponentState(JComponent c) {
534         int state;
535 
536         if (!c.isEnabled()) {
537             state = DISABLED;
538         } else if (menuItem.isArmed()) {
539             state = MOUSE_OVER;
540         } else {
541             state = SeaGlassLookAndFeel.getComponentState(c);
542         }
543         if (menuItem.isSelected()) {
544             state |= SELECTED;
545         }
546         return state;
