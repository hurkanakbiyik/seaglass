82      static Dimension getPreferredMenuItemSize(SeaGlassContext context, SeaGlassContext accContext, boolean useCheckAndArrow, JComponent c,
83          Icon checkIcon, Icon arrowIcon, int defaultTextIconGap, String acceleratorDelimiter) {
84          JMenuItem b = (JMenuItem) c;
85          Icon icon = (Icon) b.getIcon();
86          String text = b.getText();
87          KeyStroke accelerator = b.getAccelerator();
88          String acceleratorText = "";
89  
90          if (accelerator != null) {
91              int modifiers = accelerator.getModifiers();
92              if (modifiers > 0) {
93                  acceleratorText = KeyEvent.getKeyModifiersText(modifiers);
94                  acceleratorText += acceleratorDelimiter;
95              }
96              int keyCode = accelerator.getKeyCode();
97              if (keyCode != 0) {
98                  acceleratorText += KeyEvent.getKeyText(keyCode);
99              } else {
100                 acceleratorText += accelerator.getKeyChar();
101             }
102         }
103 
104         Font font = context.getStyle().getFont(context);
105         FontMetrics fm = b.getFontMetrics(font);
106         FontMetrics fmAccel = b.getFontMetrics(accContext.getStyle().getFont(accContext));
107 
108         resetRects();
109 
110         layoutMenuItem(context, fm, accContext, text, fmAccel, acceleratorText, icon, checkIcon, arrowIcon, b.getVerticalAlignment(), b
111             .getHorizontalAlignment(), b.getVerticalTextPosition(), b.getHorizontalTextPosition(), viewRect, iconRect, textRect,
112             acceleratorRect, checkIconRect, arrowIconRect, text == null ? 0 : defaultTextIconGap, defaultTextIconGap, useCheckAndArrow);
113         // find the union of the icon and text rects
114         r.setBounds(textRect);
115         r = SwingUtilities.computeUnion(iconRect.x, iconRect.y, iconRect.width, iconRect.height, r);
116         // To make the accelerator texts appear in a column,
117         // find the widest MenuItem text and the widest accelerator text.
118 
119         // Get the parent, which stores the information.
120         Container parent = b.getParent();
121 
122         if (parent instanceof JPopupMenu) {
123             SeaGlassPopupMenuUI popupUI = (SeaGlassPopupMenuUI) SeaGlassLookAndFeel.getUIOfType(((JPopupMenu) parent).getUI(),
124                 SeaGlassPopupMenuUI.class);
125 
126             if (popupUI != null) {
127                 r.width = popupUI.adjustTextWidth(r.width);
128 
129                 popupUI.adjustAcceleratorWidth(acceleratorRect.width);
130 
131                 r.width += popupUI.getMaxAcceleratorWidth();
132             }
133         } else if (parent != null && !(b instanceof JMenu && ((JMenu) b).isTopLevelMenu())) {
134             r.width += acceleratorRect.width;
135         }
136 
137         if (useCheckAndArrow) {
138             // Add in the checkIcon
139             r.width += checkIconRect.width;
140             r.width += defaultTextIconGap;
141 
142             // Add in the arrowIcon
143             r.width += defaultTextIconGap;
144             r.width += arrowIconRect.width;
145         }
146 
147         r.width += 2 * defaultTextIconGap;
148 
149         Insets insets = b.getInsets();
150         if (insets != null) {
151             r.width += insets.left + insets.right;
152             r.height += insets.top + insets.bottom;
153         }
154 
155         // if the width is even, bump it up one. This is critical
156         // for the focus dash line to draw properly
157         if (r.width % 2 == 0) {
158             r.width++;
159         }
160 
161         // if the height is even, bump it up one. This is critical
162         // for the text to center properly
163         if (r.height % 2 == 0) {
164             r.height++;
165         }
166         return r.getSize();
167     }

169     static void paint(SeaGlassContext context, SeaGlassContext accContext, Graphics g, Icon checkIcon, Icon arrowIcon,
170         boolean useCheckAndArrow, String acceleratorDelimiter, int defaultTextIconGap) {
171         JComponent c = context.getComponent();
172         JMenuItem b = (JMenuItem) c;
173         ButtonModel model = b.getModel();
174         Insets i = b.getInsets();
175 
176         resetRects();
177 
178         viewRect.setBounds(0, 0, b.getWidth(), b.getHeight());
179 
180         viewRect.x += i.left;
181         viewRect.y += i.top;
182         viewRect.width -= (i.right + viewRect.x);
183         viewRect.height -= (i.bottom + viewRect.y);
184 
185         SynthStyle style = context.getStyle();
186         Font f = style.getFont(context);
187         g.setFont(f);
188         FontMetrics fm = SwingUtilities2.getFontMetrics(c, g, f);
189         FontMetrics accFM = SwingUtilities2.getFontMetrics(c, g, accContext.getStyle().getFont(accContext));
190 
191         // get Accelerator text
192         KeyStroke accelerator = b.getAccelerator();
193         String acceleratorText = "";
194         if (accelerator != null) {
195             int modifiers = accelerator.getModifiers();
196             if (modifiers > 0) {
197                 acceleratorText = KeyEvent.getKeyModifiersText(modifiers);
198                 acceleratorText += acceleratorDelimiter;
199             }
200 
201             int keyCode = accelerator.getKeyCode();
202             if (keyCode != 0) {
203                 acceleratorText += KeyEvent.getKeyText(keyCode);
204             } else {
205                 acceleratorText += accelerator.getKeyChar();
206             }
207         }
208 
209         // layoutl the text and icon
210         String text = layoutMenuItem(context, fm, accContext, b.getText(), accFM, acceleratorText, b.getIcon(), checkIcon, arrowIcon, b
211             .getVerticalAlignment(), b.getHorizontalAlignment(), b.getVerticalTextPosition(), b.getHorizontalTextPosition(), viewRect,
212             iconRect, textRect, acceleratorRect, checkIconRect, arrowIconRect, b.getText() == null ? 0 : defaultTextIconGap,
213             defaultTextIconGap, useCheckAndArrow);
214 
215         // Paint the Check
216         if (checkIcon != null && useCheckAndArrow) {
217             SeaGlassIcon.paintIcon(checkIcon, context, g, checkIconRect.x, checkIconRect.y, checkIconRect.width, checkIconRect.height);
218         }
219 
220         // Paint the Icon
221         if (b.getIcon() != null) {
222             Icon icon;
223             if (!model.isEnabled()) {
224                 icon = (Icon) b.getDisabledIcon();
225             } else if (model.isPressed() && model.isArmed()) {
226                 icon = (Icon) b.getPressedIcon();
227                 if (icon == null) {
228                     // Use default icon
229                     icon = (Icon) b.getIcon();
230                 }
231             } else {
232                 icon = (Icon) b.getIcon();
233             }
234 
235             if (icon != null) {
236                 SeaGlassIcon.paintIcon(icon, context, g, iconRect.x, iconRect.y, iconRect.width, iconRect.height);
237             }
238         }
239 
240         // Draw the Text
241         if (text != null) {
242             View v = (View) c.getClientProperty(BasicHTML.propertyKey);
243             if (v != null) {
244                 v.paint(g, textRect);
245             } else {
246                 g.setColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
247                 g.setFont(style.getFont(context));
248                 style.getGraphicsUtils(context).paintText(context, g, text, textRect.x, textRect.y, b.getDisplayedMnemonicIndex());
249             }
250         }
251 
252         // Draw the Accelerator Text
253         if (acceleratorText != null && !acceleratorText.equals("")) {
254             // Get the maxAccWidth from the parent to calculate the offset.
255             int accOffset = 0;
256             Container parent = b.getParent();
257             if (parent != null && parent instanceof JPopupMenu) {
258                 SeaGlassPopupMenuUI popupUI = (SeaGlassPopupMenuUI) ((JPopupMenu) parent).getUI();
259                 if (popupUI != null) {
260                     // Calculate the offset, with which the accelerator texts
261                     // will be drawn with.
262                     int max = popupUI.getMaxAcceleratorWidth();
263                     if (max > 0) {
264                         accOffset = max - acceleratorRect.width;
265                     }
266                 }
267             }
268 
269             SynthStyle accStyle = accContext.getStyle();
270 
271             g.setColor(accStyle.getColor(accContext, ColorType.TEXT_FOREGROUND));
272             g.setFont(accStyle.getFont(accContext));
273             accStyle.getGraphicsUtils(accContext).paintText(accContext, g, acceleratorText, acceleratorRect.x - accOffset,
274                 acceleratorRect.y, -1);
275         }
276 
277         // Paint the Arrow
278         if (arrowIcon != null && useCheckAndArrow) {
279             SeaGlassIcon.paintIcon(arrowIcon, context, g, arrowIconRect.x, arrowIconRect.y, arrowIconRect.width, arrowIconRect.height);
280         }
281     }

283     /**
284      * Compute and return the location of the icons origin, the location of
285      * origin of the text baseline, and a possibly clipped version of the
286      * compound labels string. Locations are computed relative to the viewRect
287      * rectangle.
288      */

290     private static String layoutMenuItem(SeaGlassContext context, FontMetrics fm, SeaGlassContext accContext, String text,
291         FontMetrics fmAccel, String acceleratorText, Icon icon, Icon checkIcon, Icon arrowIcon, int verticalAlignment,
292         int horizontalAlignment, int verticalTextPosition, int horizontalTextPosition, Rectangle viewRect, Rectangle iconRect,
293         Rectangle textRect, Rectangle acceleratorRect, Rectangle checkIconRect, Rectangle arrowIconRect, int textIconGap, int menuItemGap,
294         boolean useCheckAndArrow) {
295         // If parent is JPopupMenu, get and store it's UI
296         SeaGlassPopupMenuUI popupUI = null;
297         JComponent b = context.getComponent();
298         Container parent = b.getParent();
299         if (parent instanceof JPopupMenu) {
300             popupUI = (SeaGlassPopupMenuUI) SeaGlassLookAndFeel.getUIOfType(((JPopupMenu) parent).getUI(), SeaGlassPopupMenuUI.class);
301         }
302 
303         context.getStyle().getGraphicsUtils(context).layoutText(context, fm, text, icon, horizontalAlignment, verticalAlignment,
304             horizontalTextPosition, verticalTextPosition, viewRect, iconRect, textRect, textIconGap);
305 
306         /*
307          * Initialize the acceelratorText bounds rectangle textRect. If a null
308          * or and empty String was specified we substitute "" here and use
309          * 0,0,0,0 for acceleratorTextRect.
310          */
311         if ((acceleratorText == null) || acceleratorText.equals("")) {
312             acceleratorRect.width = acceleratorRect.height = 0;
313             acceleratorText = "";
314         } else {
315             SynthStyle style = accContext.getStyle();
316             acceleratorRect.width = style.getGraphicsUtils(accContext).computeStringWidth(accContext, fmAccel.getFont(), fmAccel,
317                 acceleratorText);
318             acceleratorRect.height = fmAccel.getHeight();
319         }
320 
321         /*
322          * Initialize the checkIcon bounds rectangle's width & height.
323          */
324 
325         if (useCheckAndArrow) {
326             if (checkIcon != null) {
327                 checkIconRect.width = SeaGlassIcon.getIconWidth(checkIcon, context);
328                 checkIconRect.height = SeaGlassIcon.getIconHeight(checkIcon, context);
329             } else {
330                 checkIconRect.width = checkIconRect.height = 0;
331             }
332 
333             /*
334              * Initialize the arrowIcon bounds rectangle width & height.
335              */
336 
337             if (arrowIcon != null) {
338                 arrowIconRect.width = SeaGlassIcon.getIconWidth(arrowIcon, context);
339                 arrowIconRect.height = SeaGlassIcon.getIconHeight(arrowIcon, context);
340             } else {
341                 arrowIconRect.width = arrowIconRect.height = 0;
342             }
343         }
344 
345         Rectangle labelRect = iconRect.union(textRect);
346         if (SeaGlassLookAndFeel.isLeftToRight(context.getComponent())) {
347             textRect.x += menuItemGap;
348             iconRect.x += menuItemGap;
349 
350             // Position the Accelerator text rect
351             acceleratorRect.x = viewRect.x + viewRect.width - arrowIconRect.width - menuItemGap - acceleratorRect.width;
352 
353             // Position the Check and Arrow Icons
354             if (useCheckAndArrow) {
355                 checkIconRect.x = viewRect.x + menuItemGap;
356                 textRect.x += menuItemGap + checkIconRect.width;
357                 iconRect.x += menuItemGap + checkIconRect.width;
358                 arrowIconRect.x = viewRect.x + viewRect.width - menuItemGap - arrowIconRect.width;
359             }
360             /* Align icons and text horizontally */
361             if (popupUI != null) {
362                 int thisTextOffset = popupUI.adjustTextOffset(textRect.x - viewRect.x);
363                 textRect.x = thisTextOffset + viewRect.x;
364 
365                 if (icon != null) {
366                     if (horizontalTextPosition == SwingConstants.TRAILING || horizontalTextPosition == SwingConstants.RIGHT) {
367                         int thisIconOffset = popupUI.adjustIconOffset(iconRect.x - viewRect.x);
368                         iconRect.x = thisIconOffset + viewRect.x;
369                     } else if (horizontalTextPosition == SwingConstants.LEADING || horizontalTextPosition == SwingConstants.LEFT) {
370                         iconRect.x = textRect.x + textRect.width + menuItemGap;
371                     } else {
372                         int maxIconValue = popupUI.adjustIconOffset(0);
373                         iconRect.x = Math.max(textRect.x + textRect.width / 2 - iconRect.width / 2, maxIconValue + viewRect.x);
374                     }
375                 }
376             }
377         } else {
378             textRect.x -= menuItemGap;
379             iconRect.x -= menuItemGap;
380 
381             // Position the Accelerator text rect
382             acceleratorRect.x = viewRect.x + arrowIconRect.width + menuItemGap;
383 
384             // Position the Check and Arrow Icons
385             if (useCheckAndArrow) {
386                 checkIconRect.x = viewRect.x + viewRect.width - menuItemGap - checkIconRect.width;
387                 textRect.x -= menuItemGap + checkIconRect.width;
388                 iconRect.x -= menuItemGap + checkIconRect.width;
389                 arrowIconRect.x = viewRect.x + menuItemGap;
390             }
391             /* Align icons and text horizontally */
392             if (popupUI != null) {
393                 int thisTextOffset = viewRect.x + viewRect.width - textRect.x - textRect.width;
394                 thisTextOffset = popupUI.adjustTextOffset(thisTextOffset);
395                 textRect.x = viewRect.x + viewRect.width - thisTextOffset - textRect.width;
396                 if (icon != null) {
397                     if (horizontalTextPosition == SwingConstants.TRAILING || horizontalTextPosition == SwingConstants.LEFT) {
398                         int thisIconOffset = viewRect.x + viewRect.width - iconRect.x - iconRect.width;
399                         thisIconOffset = popupUI.adjustIconOffset(thisIconOffset);
400                         iconRect.x = viewRect.x + viewRect.width - thisIconOffset - iconRect.width;
401 
402                     } else if (horizontalTextPosition == SwingConstants.LEADING || horizontalTextPosition == SwingConstants.RIGHT) {
403                         iconRect.x = textRect.x - menuItemGap - iconRect.width;
404                     } else {
405                         int maxIconValue = popupUI.adjustIconOffset(0);
406                         iconRect.x = textRect.x + textRect.width / 2 - iconRect.width / 2;
407                         if (iconRect.x + iconRect.width > viewRect.x + viewRect.width - maxIconValue) {
408                             iconRect.x = iconRect.x = viewRect.x + viewRect.width - maxIconValue - iconRect.width;
409                         }
410                     }
411                 }
412             }
413         }
414 
415         // Align the accelertor text and the check and arrow icons vertically
416         // with the center of the label rect.
417         acceleratorRect.y = labelRect.y + (labelRect.height / 2) - (acceleratorRect.height / 2);
418         if (useCheckAndArrow) {
419             arrowIconRect.y = labelRect.y + (labelRect.height / 2) - (arrowIconRect.height / 2);
420             checkIconRect.y = labelRect.y + (labelRect.height / 2) - (checkIconRect.height / 2);
421         }
422 
423         return text;
424     }

455     private void updateStyle(JMenuItem mi) {
456         SeaGlassContext context = getContext(mi, ENABLED);
457         SynthStyle oldStyle = style;
458 
459         style = SeaGlassLookAndFeel.updateStyle(context, this);
460         if (oldStyle != style) {
461             String prefix = getPropertyPrefix();
462 
463             Object value = style.get(context, prefix + ".textIconGap");
464             if (value != null) {
465                 LookAndFeel.installProperty(mi, "iconTextGap", value);
466             }
467             defaultTextIconGap = mi.getIconTextGap();
468 
469             if (menuItem.getMargin() == null || (menuItem.getMargin() instanceof UIResource)) {
470                 Insets insets = (Insets) style.get(context, prefix + ".margin");
471 
472                 if (insets == null) {
473                     // Some places assume margins are non-null.
474                     insets = SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS;
475                 }
476                 menuItem.setMargin(insets);
477             }
478             acceleratorDelimiter = style.getString(context, prefix + ".acceleratorDelimiter", "+");
479 
480             arrowIcon = style.getIcon(context, prefix + ".arrowIcon");
481 
482             checkIcon = style.getIcon(context, prefix + ".checkIcon");
483             if (oldStyle != null) {
484                 uninstallKeyboardActions();
485                 installKeyboardActions();
486             }
487         }
488         context.dispose();
489 
490         SeaGlassContext accContext = getContext(mi, Region.MENU_ITEM_ACCELERATOR, ENABLED);
491 
492         accStyle = SeaGlassLookAndFeel.updateStyle(accContext, this);
493         accContext.dispose();
494     }

531     private int getComponentState(JComponent c) {
532         int state;
533 
534         if (!c.isEnabled()) {
535             state = DISABLED;
536         } else if (menuItem.isArmed()) {
537             state = MOUSE_OVER;
538         } else {
539             state = SeaGlassLookAndFeel.getComponentState(c);
540         }
541         if (menuItem.isSelected()) {
542             state |= SELECTED;
543         }
544         return state;
