64      /**
65       * When a JSlider is used as a renderer in a JTable, its layout is not being
66       * recomputed even though the size is changing. Even though there is a
67       * ComponentListener installed, it is not being notified. As such, at times
68       * when being asked to paint the layout should first be redone. At the end
69       * of the layout method we set this lastSize variable, which represents the
70       * size of the slider the last time it was layed out.
71       * 
72       * In the paint method we then check to see that this is accurate, that the
73       * slider has not changed sizes since being last layed out. If necessary we
74       * recompute the layout.
75       */
76      private Dimension         lastSize   = null;

140     private void updateStyle(JSlider c) {
141         SeaGlassContext context = getContext(c, ENABLED);
142         SynthStyle oldStyle = style;
143         style = SeaGlassLookAndFeel.updateStyle(context, this);
144 
145         if (style != oldStyle) {
146             thumbWidth = style.getInt(context, "Slider.thumbWidth", 30);
147 
148             thumbHeight = style.getInt(context, "Slider.thumbHeight", 14);
149             
150             trackHeightHack = style.getInt(context, "Slider.trackHeight", 5);
151 
152             // handle scaling for sizeVarients for special case components. The
153             // key "JComponent.sizeVariant" scales for large/small/mini
154             // components are based on Apples LAF
155             String scaleKey = (String) slider.getClientProperty("JComponent.sizeVariant");
156             if (scaleKey != null) {
157                 if ("large".equals(scaleKey)) {
158                     thumbWidth *= 1.15;
159                     thumbHeight *= 1.15;
160                     trackHeightHack = 7;
161                 } else if ("small".equals(scaleKey)) {
162                     thumbWidth *= 0.857;
163                     thumbHeight *= 0.857;
164                     trackHeightHack = 3;
165                 } else if ("mini".equals(scaleKey)) {
166                     thumbWidth *= 0.784;
167                     thumbHeight *= 0.784;
168                     trackHeightHack = 1;
169                 }
170             }
171 
172             trackBorder = style.getInt(context, "Slider.trackBorder", 1);
173 
174             trackHeight = thumbHeight + trackBorder * 2;
175 
176             paintValue = style.getBoolean(context, "Slider.paintValue", true);
177             if (oldStyle != null) {
178                 uninstallKeyboardActions(c);
179                 installKeyboardActions(c);
180             }
181         }
182         context.dispose();
183 
184         context = getContext(c, Region.SLIDER_TRACK, ENABLED);
185         sliderTrackStyle = SeaGlassLookAndFeel.updateStyle(context, this);
186         context.dispose();
187 
188         context = getContext(c, Region.SLIDER_THUMB, ENABLED);
189         sliderThumbStyle = SeaGlassLookAndFeel.updateStyle(context, this);
190         context.dispose();
191     }

220     public int getBaseline(JComponent c, int width, int height) {
221         if (c == null) {
222             throw new NullPointerException("Component must be non-null");
223         }
224         if (width < 0 || height < 0) {
225             throw new IllegalArgumentException("Width and height must be >= 0");
226         }
227         if (slider.getPaintLabels() && labelsHaveSameBaselines()) {
228             // Get the insets for the track.
229             Insets trackInsets = new Insets(0, 0, 0, 0);
230             SeaGlassContext trackContext = getContext(slider, Region.SLIDER_TRACK);
231             style.getInsets(trackContext, trackInsets);
232             trackContext.dispose();
233             if (slider.getOrientation() == JSlider.HORIZONTAL) {
234                 int valueHeight = 0;
235                 if (paintValue) {
236                     SeaGlassContext context = getContext(slider);
237                     valueHeight = context.getStyle().getGraphicsUtils(context).getMaximumCharHeight(context);
238                     context.dispose();
239                 }
240                 int tickHeight = 0;
241                 if (slider.getPaintTicks()) {
242                     tickHeight = getTickLength();
243                 }
244                 int labelHeight = getHeightOfTallestLabel();
245                 int contentHeight = valueHeight + trackHeight + trackInsets.top + trackInsets.bottom + tickHeight + labelHeight + 4;
246                 int centerY = height / 2 - contentHeight / 2;
247                 centerY += valueHeight + 2;
248                 centerY += trackHeight + trackInsets.top + trackInsets.bottom;
249                 centerY += tickHeight + 2;
250                 Component label = (Component) slider.getLabelTable().elements().nextElement();
251                 Dimension pref = label.getPreferredSize();
252                 return centerY + label.getBaseline(pref.width, pref.height);
253             } else { // VERTICAL
254                 Integer value = slider.getInverted() ? getLowestValue() : getHighestValue();
255                 if (value != null) {
256                     int valueY = insetCache.top;
257                     int valueHeight = 0;
258                     if (paintValue) {
259                         SeaGlassContext context = getContext(slider);
260                         valueHeight = context.getStyle().getGraphicsUtils(context).getMaximumCharHeight(context);
261                         context.dispose();
262                     }
263                     int contentHeight = height - insetCache.top - insetCache.bottom;
264                     int trackY = valueY + valueHeight;
265                     int trackHeight = contentHeight - valueHeight;
266                     int yPosition = yPositionForValue(value.intValue(), trackY, trackHeight);
267                     Component label = (Component) slider.getLabelTable().get(value);
268                     Dimension pref = label.getPreferredSize();
269                     return yPosition - pref.height / 2 + label.getBaseline(pref.width, pref.height);
270                 }
271             }
272         }
273         return -1;
274     }

303     protected void layout() {
304         SeaGlassContext context = getContext(slider);
305         SynthGraphicsUtils synthGraphics = style.getGraphicsUtils(context);
306 
307         // Set the thumb size.
308         Dimension size = getThumbSize();
309         thumbRect.setSize(size.width, size.height);
310 
311         // Get the insets for the track.
312         Insets trackInsets = new Insets(0, 0, 0, 0);
313         SeaGlassContext trackContext = getContext(slider, Region.SLIDER_TRACK);
314         style.getInsets(trackContext, trackInsets);
315         trackContext.dispose();
316 
317         if (slider.getOrientation() == JSlider.HORIZONTAL) {
318             // Calculate the height of all the subcomponents so we can center
319             // them.
320             valueRect.height = 0;
321             if (paintValue) {
322                 valueRect.height = synthGraphics.getMaximumCharHeight(context);
323             }
324 
325             trackRect.height = trackHeight;
326 
327             tickRect.height = 0;
328             if (slider.getPaintTicks()) {
329                 tickRect.height = getTickLength();
330             }
331 
332             labelRect.height = 0;
333             if (slider.getPaintLabels()) {
334                 labelRect.height = getHeightOfTallestLabel();
335             }
336 
337             contentDim.height = valueRect.height + trackRect.height + trackInsets.top + trackInsets.bottom + tickRect.height
338                     + labelRect.height + 4;
339             contentDim.width = slider.getWidth() - insetCache.left - insetCache.right;
340 
341             // Check if any of the labels will paint out of bounds.
342             int pad = 0;
343             if (slider.getPaintLabels()) {
344                 // Calculate the track rectangle. It is necessary for
345                 // xPositionForValue to return correct values.
346                 trackRect.x = insetCache.left;
347                 trackRect.width = contentDim.width;
348 
349                 Dictionary dictionary = slider.getLabelTable();
350                 if (dictionary != null) {
351                     int minValue = slider.getMinimum();
352                     int maxValue = slider.getMaximum();
353 
354                     // Iterate through the keys in the dictionary and find the
355                     // first and last labels indices that fall within the
356                     // slider range.
357                     int firstLblIdx = Integer.MAX_VALUE;
358                     int lastLblIdx = Integer.MIN_VALUE;
359                     for (Enumeration keys = dictionary.keys(); keys.hasMoreElements();) {
360                         int keyInt = ((Integer) keys.nextElement()).intValue();
361                         if (keyInt >= minValue && keyInt < firstLblIdx) {
362                             firstLblIdx = keyInt;
363                         }
364                         if (keyInt <= maxValue && keyInt > lastLblIdx) {
365                             lastLblIdx = keyInt;
366                         }
367                     }
368                     // Calculate the pad necessary for the labels at the first
369                     // and last visible indices.
370                     pad = getPadForLabel(firstLblIdx);
371                     pad = Math.max(pad, getPadForLabel(lastLblIdx));
372                 }
373             }
374             // Calculate the painting rectangles for each of the different
375             // slider areas.
376             valueRect.x = trackRect.x = tickRect.x = labelRect.x = (insetCache.left + pad);
377             valueRect.width = trackRect.width = tickRect.width = labelRect.width = (contentDim.width - (pad * 2));
378 
379             int centerY = slider.getHeight() / 2 - contentDim.height / 2;
380 
381             valueRect.y = centerY;
382             centerY += valueRect.height + 2;
383 
384             trackRect.y = centerY + trackInsets.top;
385             centerY += trackRect.height + trackInsets.top + trackInsets.bottom;
386 
387             tickRect.y = centerY;
388             centerY += tickRect.height + 2;
389 
390             labelRect.y = centerY;
391             centerY += labelRect.height;
392         } else {
393             // Calculate the width of all the subcomponents so we can center
394             // them.
395             trackRect.width = trackHeight;
396 
397             tickRect.width = 0;
398             if (slider.getPaintTicks()) {
399                 tickRect.width = getTickLength();
400             }
401 
402             labelRect.width = 0;
403             if (slider.getPaintLabels()) {
404                 labelRect.width = getWidthOfWidestLabel();
405             }
406 
407             valueRect.y = insetCache.top;
408             valueRect.height = 0;
409             if (paintValue) {
410                 valueRect.height = synthGraphics.getMaximumCharHeight(context);
411             }
412 
413             // Get the max width of the min or max value of the slider.
414             FontMetrics fm = slider.getFontMetrics(slider.getFont());
415             valueRect.width = Math.max(synthGraphics.computeStringWidth(context, slider.getFont(), fm, "" + slider.getMaximum()),
416                 synthGraphics.computeStringWidth(context, slider.getFont(), fm, "" + slider.getMinimum()));
417 
418             int l = valueRect.width / 2;
419             int w1 = trackInsets.left + trackRect.width / 2;
420             int w2 = trackRect.width / 2 + trackInsets.right + tickRect.width + labelRect.width;
421             contentDim.width = Math.max(w1, l) + Math.max(w2, l) + 2 + insetCache.left + insetCache.right;
422             contentDim.height = slider.getHeight() - insetCache.top - insetCache.bottom;
423 
424             // Layout the components.
425             trackRect.y = tickRect.y = labelRect.y = valueRect.y + valueRect.height;
426             trackRect.height = tickRect.height = labelRect.height = contentDim.height - valueRect.height;
427 
428             int startX = slider.getWidth() / 2 - contentDim.width / 2;
429             if (SeaGlassLookAndFeel.isLeftToRight(slider)) {
430                 if (l > w1) {
431                     startX += (l - w1);
432                 }
433                 trackRect.x = startX + trackInsets.left;
434 
435                 startX += trackInsets.left + trackRect.width + trackInsets.right;
436                 tickRect.x = startX;
437                 labelRect.x = startX + tickRect.width + 2;
438             } else {
439                 if (l > w2) {
440                     startX += (l - w2);
441                 }
442                 labelRect.x = startX;
443 
444                 startX += labelRect.width + 2;
445                 tickRect.x = startX;
446                 trackRect.x = startX + tickRect.width + trackInsets.left;
447             }
448         }
449         context.dispose();
450 
451         lastSize = slider.getSize();
452     }

454     /**
455      * Calculates the pad for the label at the specified index.
456      * 
457      * @param index
458      *            index of the label to calculate pad for.
459      * @return padding required to keep label visible.
460      */
461     private int getPadForLabel(int i) {
462         Dictionary dictionary = slider.getLabelTable();
463         int pad = 0;
464 
465         Object o = dictionary.get(i);
466         if (o != null) {
467             Component c = (Component) o;
468             int centerX = xPositionForValue(i);
469             int cHalfWidth = c.getPreferredSize().width / 2;
470             if (centerX - cHalfWidth < insetCache.left) {
471                 pad = Math.max(pad, insetCache.left - (centerX - cHalfWidth));
472             }
473 
474             if (centerX + cHalfWidth > slider.getWidth() - insetCache.right) {
475                 pad = Math.max(pad, (centerX + cHalfWidth) - (slider.getWidth() - insetCache.right));
476             }
477         }
478         return pad;
479     }

481     protected void calculateThumbLocation() {
482         if (slider.getSnapToTicks()) {
483             int sliderValue = slider.getValue();
484             int snappedValue = sliderValue;
485             int majorTickSpacing = slider.getMajorTickSpacing();
486             int minorTickSpacing = slider.getMinorTickSpacing();
487             int tickSpacing = 0;
488 
489             if (minorTickSpacing > 0) {
490                 tickSpacing = minorTickSpacing;
491             } else if (majorTickSpacing > 0) {
492                 tickSpacing = majorTickSpacing;
493             }
494 
495             if (tickSpacing != 0) {
496                 // If it's not on a tick, change the value
497                 if ((sliderValue - slider.getMinimum()) % tickSpacing != 0) {
498                     float temp = (float) (sliderValue - slider.getMinimum()) / (float) tickSpacing;
499                     int whichTick = Math.round(temp);
500                     snappedValue = slider.getMinimum() + (whichTick * tickSpacing);
501                 }
502 
503                 if (snappedValue != sliderValue) {
504                     slider.setValue(snappedValue);
505                 }
506             }
507         }
508 
509         if (slider.getOrientation() == JSlider.HORIZONTAL) {
510             int valuePosition = xPositionForValue(slider.getValue());
511             thumbRect.x = valuePosition - (thumbRect.width / 2);
512             thumbRect.y = trackRect.y + trackBorder;
513         } else {
514             int valuePosition = yPositionForValue(slider.getValue());
515             thumbRect.x = trackRect.x + trackBorder;
516             thumbRect.y = valuePosition - (thumbRect.height / 2);
517         }
518         Point mousePosition = slider.getMousePosition();
519         if (mousePosition != null) {
520             updateThumbState(mousePosition.x, mousePosition.y);
521         }
522     }

524     protected void calculateTickRect() {
525         if (slider.getOrientation() == JSlider.HORIZONTAL) {
526             tickRect.x = trackRect.x;
527             tickRect.y = trackRect.y + trackRect.height + 2 + getTickLength();
528             tickRect.width = trackRect.width;
529             tickRect.height = getTickLength();
530 
531             if (!slider.getPaintTicks()) {
532                 --tickRect.y;
533                 tickRect.height = 0;
534             }
535         } else {
536             if (SeaGlassLookAndFeel.isLeftToRight(slider)) {
537                 tickRect.x = trackRect.x + trackRect.width;
538                 tickRect.width = getTickLength();
539             } else {
540                 tickRect.width = getTickLength();
541                 tickRect.x = trackRect.x - tickRect.width;
542             }
543             tickRect.y = trackRect.y;
544             tickRect.height = trackRect.height;
545 
546             if (!slider.getPaintTicks()) {
547                 --tickRect.x;
548                 tickRect.width = 0;
549             }
550         }
551     }

561     protected int xPositionForValue(int value) {
562         int min = slider.getMinimum();
563         int max = slider.getMaximum();
564         int trackLeft = trackRect.x + thumbRect.width / 2 + trackBorder;
565         int trackRight = trackRect.x + trackRect.width - thumbRect.width / 2 - trackBorder;
566         int trackLength = trackRight - trackLeft;
567         double valueRange = (double) max - (double) min;
568         double pixelsPerValue = (double) trackLength / valueRange;
569         int xPosition;
570 
571         if (!drawInverted()) {
572             xPosition = trackLeft;
573             xPosition += Math.round(pixelsPerValue * ((double) value - min));
574         } else {
575             xPosition = trackRight;
576             xPosition -= Math.round(pixelsPerValue * ((double) value - min));
577         }
578 
579         xPosition = Math.max(trackLeft, xPosition);
580         xPosition = Math.min(trackRight, xPosition);
581 
582         return xPosition;
583     }

585     protected int yPositionForValue(int value, int trackY, int trackHeight) {
586         int min = slider.getMinimum();
587         int max = slider.getMaximum();
588         int trackTop = trackY + thumbRect.height / 2 + trackBorder;
589         int trackBottom = trackY + trackHeight - thumbRect.height / 2 - trackBorder;
590         int trackLength = trackBottom - trackTop;
591         double valueRange = (double) max - (double) min;
592         double pixelsPerValue = (double) trackLength / (double) valueRange;
593         int yPosition;
594 
595         if (!drawInverted()) {
596             yPosition = trackTop;
597             yPosition += Math.round(pixelsPerValue * ((double) max - value));
598         } else {
599             yPosition = trackTop;
600             yPosition += Math.round(pixelsPerValue * ((double) value - min));
601         }
602 
603         yPosition = Math.max(trackTop, yPosition);
604         yPosition = Math.min(trackBottom, yPosition);
605 
606         return yPosition;
607     }

609     /**
610      * Returns a value give a y position. If yPos is past the track at the top
611      * or the bottom it will set the value to the min or max of the slider,
612      * depending if the slider is inverted or not.
613      */
614     public int valueForYPosition(int yPos) {
615         int value;
616         int minValue = slider.getMinimum();
617         int maxValue = slider.getMaximum();
618         int trackTop = trackRect.y + thumbRect.height / 2 + trackBorder;
619         int trackBottom = trackRect.y + trackRect.height - thumbRect.height / 2 - trackBorder;
620         int trackLength = trackBottom - trackTop;
621 
622         if (yPos <= trackTop) {
623             value = drawInverted() ? minValue : maxValue;
624         } else if (yPos >= trackBottom) {
625             value = drawInverted() ? maxValue : minValue;
626         } else {
627             int distanceFromTrackTop = yPos - trackTop;
628             double valueRange = (double) maxValue - (double) minValue;
629             double valuePerPixel = valueRange / (double) trackLength;
630             int valueFromTrackTop = (int) Math.round(distanceFromTrackTop * valuePerPixel);
631             value = drawInverted() ? minValue + valueFromTrackTop : maxValue - valueFromTrackTop;
632         }
633         return value;
634     }

636     /**
637      * Returns a value give an x position. If xPos is past the track at the left
638      * or the right it will set the value to the min or max of the slider,
639      * depending if the slider is inverted or not.
640      */
641     public int valueForXPosition(int xPos) {
642         int value;
643         int minValue = slider.getMinimum();
644         int maxValue = slider.getMaximum();
645         int trackLeft = trackRect.x + thumbRect.width / 2 + trackBorder;
646         int trackRight = trackRect.x + trackRect.width - thumbRect.width / 2 - trackBorder;
647         int trackLength = trackRight - trackLeft;
648 
649         if (xPos <= trackLeft) {
650             value = drawInverted() ? maxValue : minValue;
651         } else if (xPos >= trackRight) {
652             value = drawInverted() ? minValue : maxValue;
653         } else {
654             int distanceFromTrackLeft = xPos - trackLeft;
655             double valueRange = (double) maxValue - (double) minValue;
656             double valuePerPixel = valueRange / (double) trackLength;
657             int valueFromTrackLeft = (int) Math.round(distanceFromTrackLeft * valuePerPixel);
658             value = drawInverted() ? maxValue - valueFromTrackLeft : minValue + valueFromTrackLeft;
659         }
660         return value;

741     public void paint(SeaGlassContext context, Graphics g) {
742         recalculateIfInsetsChanged();
743         recalculateIfOrientationChanged();
744         Rectangle clip = g.getClipBounds();
745 
746         if (lastSize == null || !lastSize.equals(slider.getSize())) {
747             calculateGeometry();
748         }
749 
750         if (paintValue) {
751             FontMetrics fm = SwingUtilities2.getFontMetrics(slider, g);
752             int labelWidth = context.getStyle().getGraphicsUtils(context).computeStringWidth(context, g.getFont(), fm,
753                 "" + slider.getValue());
754             valueRect.x = thumbRect.x + (thumbRect.width - labelWidth) / 2;
755 
756             // For horizontal sliders, make sure value is not painted
757             // outside slider bounds.
758             if (slider.getOrientation() == JSlider.HORIZONTAL) {
759                 if (valueRect.x + labelWidth > contentDim.width) {
760                     valueRect.x = contentDim.width - labelWidth;
761                 }
762                 valueRect.x = Math.max(valueRect.x, 0);
763             }
764 
765             g.setColor(context.getStyle().getColor(context, ColorType.TEXT_FOREGROUND));
766             context.getStyle().getGraphicsUtils(context).paintText(context, g, "" + slider.getValue(), valueRect.x, valueRect.y, -1);
767         }
768 
769         SeaGlassContext subcontext = getContext(slider, Region.SLIDER_TRACK);
770         paintTrack(subcontext, g, trackRect);
771         subcontext.dispose();
772 
773         subcontext = getContext(slider, Region.SLIDER_THUMB);
774         paintThumb(subcontext, g, thumbRect);
775         subcontext.dispose();
776 
777         if (slider.getPaintTicks() && clip.intersects(tickRect)) {
778             paintTicks(g);
779         }
780 
781         if (slider.getPaintLabels() && clip.intersects(labelRect)) {
782             paintLabels(g);
783         }

847         public void mouseDragged(MouseEvent e) {
848             int thumbMiddle = 0;
849 
850             if (!slider.isEnabled()) {
851                 return;
852             }
853 
854             currentMouseX = e.getX();
855             currentMouseY = e.getY();
856 
857             if (!isDragging()) {
858                 return;
859             }
860 
861             slider.setValueIsAdjusting(true);
862 
863             switch (slider.getOrientation()) {
864             case JSlider.VERTICAL:
865                 int halfThumbHeight = thumbRect.height / 2;
866                 int thumbTop = e.getY() - offset;
867                 int trackTop = trackRect.y;
868                 int trackBottom = trackRect.y + trackRect.height - halfThumbHeight - trackBorder;
869                 int vMax = yPositionForValue(slider.getMaximum() - slider.getExtent());
870 
871                 if (drawInverted()) {
872                     trackBottom = vMax;
873                     trackTop = trackTop + halfThumbHeight;
874                 } else {
875                     trackTop = vMax;
876                 }
877                 thumbTop = Math.max(thumbTop, trackTop - halfThumbHeight);
878                 thumbTop = Math.min(thumbTop, trackBottom - halfThumbHeight);
879 
880                 setThumbLocation(thumbRect.x, thumbTop);
881 
882                 thumbMiddle = thumbTop + halfThumbHeight;
883                 slider.setValue(valueForYPosition(thumbMiddle));
884                 break;
885             case JSlider.HORIZONTAL:
886                 int halfThumbWidth = thumbRect.width / 2;
887                 int thumbLeft = e.getX() - offset;
888                 int trackLeft = trackRect.x + halfThumbWidth + trackBorder;
889                 int trackRight = trackRect.x + trackRect.width - halfThumbWidth - trackBorder;
890                 int hMax = xPositionForValue(slider.getMaximum() - slider.getExtent());
891 
892                 if (drawInverted()) {
893                     trackLeft = hMax;
894                 } else {
895                     trackRight = hMax;
896                 }
897                 thumbLeft = Math.max(thumbLeft, trackLeft - halfThumbWidth);
898                 thumbLeft = Math.min(thumbLeft, trackRight - halfThumbWidth);
899 
900                 setThumbLocation(thumbLeft, thumbRect.y);
901 
902                 thumbMiddle = thumbLeft + halfThumbWidth;
903                 slider.setValue(valueForXPosition(thumbMiddle));
904                 break;
905             default:
906                 return;
907             }
908 
909             if (slider.getValueIsAdjusting()) {
910                 setThumbActive(true);
911             }
912         }
