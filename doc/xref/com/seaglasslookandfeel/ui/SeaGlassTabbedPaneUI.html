220     /**
221      * Update the Synth styles when something changes.
222      *
223      * @param c the component.
224      */
225     private void updateStyle(JTabbedPane c) {
226         SeaGlassContext context  = getContext(c, ENABLED);
227         SynthStyle      oldStyle = style;
228 
229         style = SeaGlassLookAndFeel.updateStyle(context, this);
230 
231         tabPlacement = tabPane.getTabPlacement();
232         orientation  = ControlOrientation.getOrientation(tabPlacement == LEFT || tabPlacement == RIGHT ? VERTICAL : HORIZONTAL);
233 
234         closeButtonArmedIndex = -1;
235         Object o = c.getClientProperty("JTabbedPane.closeButton");
236 
237         if (o != null && "left".equals(o)) {
238             tabCloseButtonPlacement = LEFT;
239         } else if (o != null && "right".equals(o)) {
240             tabCloseButtonPlacement = RIGHT;
241         } else {
242             tabCloseButtonPlacement = CENTER;
243         }
244 
245         closeButtonSize   = style.getInt(context, "closeButtonSize", 6);
246         closeButtonInsets = (Insets) style.get(context, "closeButtonInsets");
247         if (closeButtonInsets == null) {
248             closeButtonInsets = new Insets(2, 2, 2, 2);
249         }
250 
251         o = c.getClientProperty("JTabbedPane.closeListener");
252         if (o != null && o instanceof SeaGlassTabCloseListener) {
253             if (tabCloseListener == null) {
254                 tabCloseListener = (SeaGlassTabCloseListener) o;
255             }
256         }
257 
258         // Add properties other than JComponent colors, Borders and
259         // opacity settings here:
260         if (style != oldStyle) {
261 
262             tabRunOverlay        = 0;
263             textIconGap          = style.getInt(context, "TabbedPane.textIconGap", 0);
264             selectedTabPadInsets = (Insets) style.get(context, "TabbedPane.selectedTabPadInsets");
265 
266             if (selectedTabPadInsets == null) {
267                 selectedTabPadInsets = new Insets(0, 0, 0, 0);
268             }
269 
270             if (oldStyle != null) {
271                 uninstallKeyboardActions();
272                 installKeyboardActions();
273             }
274         }
275 
276         context.dispose();
277 
278         if (tabContext != null) {
279             tabContext.dispose();
280         }
281 
282         tabContext    = getContext(c, Region.TABBED_PANE_TAB, ENABLED);
283         this.tabStyle = SeaGlassLookAndFeel.updateStyle(tabContext, this);
284         tabInsets     = tabStyle.getInsets(tabContext, null);
285 
286         if (tabCloseContext != null) {
287             tabCloseContext.dispose();
288         }
289 
290         tabCloseContext    = getContext(c, SeaGlassRegion.TABBED_PANE_TAB_CLOSE_BUTTON, ENABLED);
291         this.tabCloseStyle = SeaGlassLookAndFeel.updateStyle(tabCloseContext, this);
292 
293         if (tabAreaContext != null) {
294             tabAreaContext.dispose();
295         }
296 
297         tabAreaContext    = getContext(c, Region.TABBED_PANE_TAB_AREA, ENABLED);
298         this.tabAreaStyle = SeaGlassLookAndFeel.updateStyle(tabAreaContext, this);
299         tabAreaInsets     = tabAreaStyle.getInsets(tabAreaContext, null);
300 
301         if (tabContentContext != null) {
302             tabContentContext.dispose();
303         }
304 
305         tabContentContext    = getContext(c, Region.TABBED_PANE_CONTENT, ENABLED);
306         this.tabContentStyle = SeaGlassLookAndFeel.updateStyle(tabContentContext, this);
307         contentBorderInsets  = tabContentStyle.getInsets(tabContentContext, null);
308     }

331     /**
332      * Scroll the tab buttons backwards.
333      */
334     protected void scrollBackward() {
335         int selectedIndex = tabPane.getSelectedIndex();
336 
337         if (--selectedIndex < 0) {
338             tabPane.setSelectedIndex(tabPane.getTabCount() == 0 ? -1 : 0);
339         } else {
340             tabPane.setSelectedIndex(selectedIndex);
341         }
342 
343         tabPane.repaint();
344     }
345 
346     /**
347      * Scroll the tab buttons forwards.
348      */
349     protected void scrollForward() {
350         int selectedIndex = tabPane.getSelectedIndex();
351 
352         if (++selectedIndex >= tabPane.getTabCount()) {
353             tabPane.setSelectedIndex(tabPane.getTabCount() - 1);
354         } else {
355             tabPane.setSelectedIndex(selectedIndex);
356         }
357 
358         tabPane.repaint();
359     }

469     /**
470      * Get the state for the specified tab's close button.
471      *
472      * @param  c               the tabbed pane.
473      * @param  tabIndex        the index of the tab.
474      * @param  tabIsMousedOver TODO
475      *
476      * @return the close button state.
477      */
478     public int getCloseButtonState(JComponent c, int tabIndex, boolean tabIsMousedOver) {
479         if (!c.isEnabled()) {
480             return DISABLED;
481         } else if (tabIndex == closeButtonArmedIndex) {
482             return PRESSED;
483         } else if (tabIndex == closeButtonHoverIndex) {
484             return FOCUSED;
485         } else if (tabIsMousedOver) {
486             return MOUSE_OVER;
487         }
488 
489         return ENABLED;
490     }

568     protected int getBaseline(int tab) {
569         if (tabPane.getTabComponentAt(tab) != null || getTextViewForTab(tab) != null) {
570             return super.getBaseline(tab);
571         }
572 
573         String      title   = tabPane.getTitleAt(tab);
574         Font        font    = tabContext.getStyle().getFont(tabContext);
575         FontMetrics metrics = getFontMetrics(font);
576         Icon        icon    = getIconForTab(tab);
577 
578         textRect.setBounds(0, 0, 0, 0);
579         iconRect.setBounds(0, 0, 0, 0);
580         calcRect.setBounds(0, 0, Short.MAX_VALUE, maxTabHeight);

582                                                                       SwingUtilities.CENTER, SwingUtilities.LEADING,
583                                                                       SwingUtilities.TRAILING, calcRect, iconRect, textRect, textIconGap);
584 
585         return textRect.y + metrics.getAscent() + getBaselineOffset();
586     }

630     /**
631      * Paint the tab area, including the tabs.
632      *
633      * @param ss            the SynthContext.
634      * @param g             the Graphics context.
635      * @param tabPlacement  the side the tabs are on.
636      * @param selectedIndex the current selected tab index.
637      * @param tabAreaBounds the bounds of the tab area.
638      */
639     protected void paintTabArea(SeaGlassContext ss, Graphics g, int tabPlacement, int selectedIndex, Rectangle tabAreaBounds) {
640         Rectangle clipRect = g.getClipBounds();
641 
642         ss.setComponentState(SynthConstants.ENABLED);
643 
644         // Paint the tab area.
645         SeaGlassLookAndFeel.updateSubregion(ss, g, tabAreaBounds);
646         ss.getPainter().paintTabbedPaneTabAreaBackground(ss, g, tabAreaBounds.x, tabAreaBounds.y, tabAreaBounds.width,
647                                                          tabAreaBounds.height, tabPlacement);
648         ss.getPainter().paintTabbedPaneTabAreaBorder(ss, g, tabAreaBounds.x, tabAreaBounds.y, tabAreaBounds.width, tabAreaBounds.height,
649                                                      tabPlacement);
650 
651         iconRect.setBounds(0, 0, 0, 0);
652         textRect.setBounds(0, 0, 0, 0);
653 
654         if (runCount == 0) {
655             return;
656         }
657 
658         if (scrollBackwardButton.isVisible()) {
659             paintScrollButtonBackground(ss, g, scrollBackwardButton);
660         }
661 
662         if (scrollForwardButton.isVisible()) {
663             paintScrollButtonBackground(ss, g, scrollForwardButton);
664         }
665 
666         for (int i = leadingTabIndex; i <= trailingTabIndex; i++) {
667             if (rects[i].intersects(clipRect) && selectedIndex != i) {
668                 paintTab(tabContext, g, rects, i, iconRect, textRect);
669             }
670         }
671 
672         if (selectedIndex >= 0) {
673             if (rects[selectedIndex].intersects(clipRect)) {
674                 paintTab(tabContext, g, rects, selectedIndex, iconRect, textRect);
675             }
676         }
677     }

679     /**
680      * @see javax.swing.plaf.basic.BasicTabbedPaneUI#setRolloverTab(int)
681      */
682     protected void setRolloverTab(int index) {
683         int oldRolloverTab = getRolloverTab();
684 
685         super.setRolloverTab(index);
686 
687         Rectangle r = null;
688 
689         if ((oldRolloverTab >= 0) && (oldRolloverTab < tabPane.getTabCount())) {
690             r = getTabBounds(tabPane, oldRolloverTab);
691 
692             if (r != null) {
693                 tabPane.repaint(r);
694             }
695         }
696 
697         if (index >= 0) {
698             r = getTabBounds(tabPane, index);
699 
700             if (r != null) {
701                 tabPane.repaint(r);
702             }
703         }
704     }

1102         }

1565         /**
1566          * Calculate the leading and trailing tab indices that will fit in the
1567          * length, keeping the selected index visible.
1568          *
1569          * @param tabCount      the number of tabs.
1570          * @param selectedIndex the current tab.
1571          * @param tabAreaLength the length of the tab area. This takes the tab
1572          *                      area insets into account.
1573          * @param buttonLength  the length of a scroll button. They are both the
1574          *                      same length.
1575          */
1576         private void determineVisibleTabIndices(int tabCount, int selectedIndex, int tabAreaLength, int buttonLength) {
1577             int desiredMaximumLength = calcDesiredMaximumLength(tabCount);
1578             int leadingTabOffset     = orientation.getPosition(rects[leadingTabIndex]);
1579             int selectedTabEndOffset = orientation.getPosition(rects[selectedIndex].x + rects[selectedIndex].width,
1580                                                                rects[selectedIndex].y + rects[selectedIndex].height);
1581 
1582             if (desiredMaximumLength <= tabAreaLength) {
1583                 // Fits with no scroll buttons.
1584                 leadingTabIndex  = 0;
1585                 trailingTabIndex = tabCount - 1;
1586             } else if (desiredMaximumLength - leadingTabOffset + buttonLength <= tabAreaLength) {
1587                 // Fits from current leading tab index, with scroll backward button. Leave leadingTabIndex alone.
1588                 trailingTabIndex = tabCount - 1;
1589             } else if ((leadingTabIndex == 0 && selectedTabEndOffset - leadingTabOffset + buttonLength <= tabAreaLength)
1590                     || (selectedTabEndOffset - leadingTabOffset + 2 * buttonLength <= tabAreaLength)) {
1591                 // Selected index fits with current leading tab index and one or two scroll buttons. Leave leadingTabIndex alone.
1592                 trailingTabIndex = -1;
1593 
1594                 for (int i = tabCount - 1; i > selectedIndex; i--) {
1595                     int end = orientation.getPosition(rects[i].x + rects[i].width, rects[i].y + rects[i].height);
1596 
1597                     if (end - leadingTabOffset + 2 * buttonLength <= tabAreaLength) {
1598                         trailingTabIndex = i;
1599                         break;
1600                     }
1601                 }
1602 
1603                 if (trailingTabIndex == -1) {
1604                     trailingTabIndex = selectedIndex;
1605                 }
1606             } else {
1607                 // Selected index does not fit with current leading index and two scroll buttons.
1608                 // Make selected index the trailing index and find the leading index that will fit.
1609                 trailingTabIndex = selectedIndex;
1610                 leadingTabIndex  = -1;
1611 
1612                 for (int i = 0; i < selectedIndex; i++) {
1613                     int start = orientation.getPosition(rects[i]);
1614 
1615                     if (selectedTabEndOffset - start + 2 * buttonLength <= tabAreaLength) {
1616                         leadingTabIndex = i;
1617                         break;
1618                     }
1619                 }
1620 
1621                 if (leadingTabIndex == -1) {
1622                     leadingTabIndex = selectedIndex;
1623                 }
1624             }
1625 
1626             tabRuns[0] = leadingTabIndex;
1627         }
