66      public void installDefaults() {
67          // We want to draw rounded corners, so the internal frame must not be
68          // opaque. LookAndFeel.installProperty doesn't seem to work here.
69          frame.setOpaque(false);
70  
71          frame.setBorder(BorderFactory.createEmptyBorder(1, 2, 2, 2));
72  
73          frame.setLayout(internalFrameLayout = createLayoutManager());
74          updateStyle(frame);
75      }

77      protected void installListeners() {
78          super.installListeners();
79          frame.addPropertyChangeListener(this);
80      }
81  
82      protected void uninstallComponents() {
83          if (frame.getComponentPopupMenu() instanceof UIResource) {
84              frame.setComponentPopupMenu(null);
85          }
86          super.uninstallComponents();
87      }
88  
89      protected void uninstallListeners() {
90          frame.removePropertyChangeListener(this);
91          super.uninstallListeners();
92      }

94      private void updateStyle(JComponent c) {
95          SeaGlassContext context = getContext(c, ENABLED);
96          SynthStyle oldStyle = style;
97  
98          style = SeaGlassLookAndFeel.updateStyle(context, this);
99          if (style != oldStyle) {
100             Icon frameIcon = frame.getFrameIcon();
101             if (frameIcon == null || frameIcon instanceof UIResource) {
102                 frame.setFrameIcon(context.getStyle().getIcon(context, "InternalFrame.icon"));
103             }
104             if (oldStyle != null) {
105                 uninstallKeyboardActions();
106                 installKeyboardActions();
107             }
108         }
109         context.dispose();
110     }
111 
112     protected void uninstallDefaults() {
113         SeaGlassContext context = getContext(frame, ENABLED);
114         style.uninstallDefaults(context);
115         context.dispose();
116         style = null;
117         if (frame.getLayout() == internalFrameLayout) {
118             frame.setLayout(null);
119         }

121     }

141     protected ComponentListener createComponentListener() {
142         if (UIManager.getBoolean("InternalFrame.useTaskBar")) {
143             return new ComponentHandler() {
144                 public void componentResized(ComponentEvent e) {
145                     if (frame != null && frame.isMaximum()) {
146                         JDesktopPane desktop = (JDesktopPane) e.getSource();
147                         for (Component comp : desktop.getComponents()) {
148                             if (comp instanceof SeaGlassDesktopPaneUI.TaskBar) {
149                                 frame.setBounds(0, 0, desktop.getWidth(), desktop.getHeight() - comp.getHeight());
150                                 frame.revalidate();
151                                 break;
152                             }
153                         }
154                     }
155 
156                     // Update the new parent bounds for next resize, but don't
157                     // let the super method touch this frame
158                     JInternalFrame f = frame;
159                     frame = null;
160                     super.componentResized(e);
161                     frame = f;
162                 }
163             };
164         } else {
165             return super.createComponentListener();
166         }
167     }

192     public void propertyChange(PropertyChangeEvent evt) {
193         SynthStyle oldStyle = style;
194         JInternalFrame f = (JInternalFrame) evt.getSource();
195         String prop = evt.getPropertyName();
196 
197         if (SeaGlassLookAndFeel.shouldUpdateStyle(evt)) {
198             updateStyle(f);
199         }
200 
201         if (style == oldStyle && (prop == JInternalFrame.IS_MAXIMUM_PROPERTY || prop == JInternalFrame.IS_SELECTED_PROPERTY)) {
202             // Border (and other defaults) may need to change
203             SeaGlassContext context = getContext(f, ENABLED);
204             style.uninstallDefaults(context);
205             ((SeaGlassStyle) style).installDefaults(context, this);
206         }
207     }
