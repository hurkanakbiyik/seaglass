44  public class SeaGlassRadioButtonMenuItemUI extends SeaGlassMenuItemUI {
45      public static ComponentUI createUI(JComponent b) {
46          return new SeaGlassRadioButtonMenuItemUI();
47      }
48  
49      protected String getPropertyPrefix() {
50          return "RadioButtonMenuItem";
51      }
52  
53      public void processMouseEvent(JMenuItem item, MouseEvent e, MenuElement path[], MenuSelectionManager manager) {
54          Point p = e.getPoint();
55          if (p.x >= 0 && p.x < item.getWidth() && p.y >= 0 && p.y < item.getHeight()) {
56              if (e.getID() == MouseEvent.MOUSE_RELEASED) {
57                  manager.clearSelectedPath();
58                  item.doClick(0);
59                  item.setArmed(false);
60              } else
61                  manager.setSelectedPath(path);
62          } else if (item.getModel().isArmed()) {
63              MenuElement newPath[] = new MenuElement[path.length - 1];
64              int i, c;
65              for (i = 0, c = path.length - 1; i < c; i++)
66                  newPath[i] = path[i];
67              manager.setSelectedPath(newPath);
68          }
69      }
70  
71      void paintBackground(SeaGlassContext context, Graphics g, JComponent c) {
72          context.getPainter().paintRadioButtonMenuItemBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
73      }
74  
75      public void paintBorder(SynthContext context, Graphics g, int x, int y, int w, int h) {
76          ((SeaGlassContext) context).getPainter().paintRadioButtonMenuItemBorder(context, g, x, y, w, h);
77      }
