249         /**
250          * @see com.seaglasslookandfeel.ui.SeaGlassTableHeaderUI$DefaultTableCellHeaderRenderer#getTableCellRendererComponent(javax.swing.JTable,
251          *      java.lang.Object, boolean, boolean, int, int)
252          */
253         @Override
254         public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row,
255                 int column) {
256             boolean hasRollover = false; // (column == getRolloverColumn());
257 
258             if (isSelected || hasRollover || hasFocus) {
259                 SeaGlassLookAndFeel.setSelectedUI((SeaGlassLabelUI) SeaGlassLookAndFeel.getUIOfType(getUI(), SeaGlassLabelUI.class),
260                                                   isSelected, hasFocus, table.isEnabled(), hasRollover);
261             } else {
262                 SeaGlassLookAndFeel.resetSelectedUI();
263             }
264 
265             // Stuff a variable into the client property of this renderer
266             // indicating the sort order, so that different rendering can be
267             // done for the header based on sorted state.
268             RowSorter                                   rs       = table == null ? null : table.getRowSorter();
269             java.util.List<? extends RowSorter.SortKey> sortKeys = rs == null ? null : rs.getSortKeys();
270 
271             if (sortKeys != null && sortKeys.size() > 0 && sortKeys.get(0).getColumn() == table.convertColumnIndexToModel(column)) {
272                 switch (sortKeys.get(0).getSortOrder()) {
273 
274                 case ASCENDING:
275                     putClientProperty("Table.sortOrder", "ASCENDING");
276                     break;
277 
278                 case DESCENDING:
279                     putClientProperty("Table.sortOrder", "DESCENDING");
280                     break;
281 
282                 case UNSORTED:
283                     putClientProperty("Table.sortOrder", "UNSORTED");
284                     break;
285 
286                 default:
287                     throw new AssertionError("Cannot happen");
288                 }
289             } else {
290                 putClientProperty("Table.sortOrder", "UNSORTED");
291             }
292 
293             super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
294 
295             return this;
296         }

331         /**
332          * @see javax.swing.table.DefaultTableCellRenderer#getTableCellRendererComponent(javax.swing.JTable,
333          *      java.lang.Object, boolean, boolean, int, int)
334          */
335         public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row,
336                 int column) {
337             Icon sortIcon = null;
338 
339             boolean isPaintingForPrint = false;
340 
341             if (table != null) {
342                 JTableHeader header = table.getTableHeader();
343 
344                 if (header != null) {
345                     Color fgColor = null;
346                     Color bgColor = null;
347 
348                     if (hasFocus) {
349                         fgColor = DefaultLookup.getColor(this, ui, "TableHeader.focusCellForeground");
350                         bgColor = DefaultLookup.getColor(this, ui, "TableHeader.focusCellBackground");
351                     }
352 
353                     if (fgColor == null) {
354                         fgColor = header.getForeground();
355                     }
356 
357                     if (bgColor == null) {
358                         bgColor = header.getBackground();
359                     }
360 
361                     setForeground(fgColor);
362                     setBackground(bgColor);
363 
364                     setFont(header.getFont());
365 
366                     isPaintingForPrint = header.isPaintingForPrint();
367                 }
368 
369                 if (!isPaintingForPrint && table.getRowSorter() != null) {
370                     SortOrder sortOrder = getColumnSortOrder(table, column);
371 
372                     if (sortOrder != null) {
373                         switch (sortOrder) {
374 
375                         case ASCENDING:
376                             sortIcon = DefaultLookup.getIcon(this, ui, "Table.ascendingSortIcon");
377                             break;
378 
379                         case DESCENDING:
380                             sortIcon = DefaultLookup.getIcon(this, ui, "Table.descendingSortIcon");
381                             break;
382 
383                         case UNSORTED:
384                             sortIcon = DefaultLookup.getIcon(this, ui, "Table.naturalSortIcon");
385                             break;
386                         }
387                     }
388                 }
389             }
390 
391             if (value instanceof Icon) {
392                 setText("");
393                 setIcon((Icon) value);
394             } else if (value instanceof JLabel) {
395                 JLabel label = (JLabel) value;
396 
397                 setText(label.getText());
398                 setIcon(label.getIcon());
399                 setHorizontalAlignment(label.getHorizontalAlignment());
400                 setHorizontalTextPosition(label.getHorizontalTextPosition());
401             } else {
402                 setText(value == null ? "" : value.toString());
403                 setIcon(null);
404             }
405 
406             sortArrow = sortIcon;
407 
408             Border border = null;
409 
410             if (hasFocus) {
411                 border = DefaultLookup.getBorder(this, ui, "TableHeader.focusCellBorder");
412             }
413 
414             if (border == null) {
415                 border = DefaultLookup.getBorder(this, ui, "TableHeader.cellBorder");
416             }
417 
418             setBorder(border);
419 
420             return this;
421         }

431         public static SortOrder getColumnSortOrder(JTable table, int column) {
432             SortOrder rv = null;
433 
434             if (table == null || table.getRowSorter() == null) {
435                 return rv;
436             }
437 
438             java.util.List<? extends RowSorter.SortKey> sortKeys = table.getRowSorter().getSortKeys();
439 
440             if (sortKeys.size() > 0 && sortKeys.get(0).getColumn() == table.convertColumnIndexToModel(column)) {
441                 rv = sortKeys.get(0).getSortOrder();
442             }
443 
444             return rv;
445         }

447         /**
448          * @see javax.swing.JComponent#paintComponent(java.awt.Graphics)
449          */
450         @Override
451         public void paintComponent(Graphics g) {
452             if (sortArrow != null) {
453                 // emptyIcon is used so that if the text in the header is right
454                 // aligned, or if the column is too narrow, then the text will
455                 // be sized appropriately to make room for the icon that is
456                 // about to be painted manually here.
457                 emptyIcon.width  = sortArrow.getIconWidth();
458                 emptyIcon.height = sortArrow.getIconHeight();
459                 Point position = computeIconPosition(g);
460 
461                 super.paintComponent(g);
462                 sortArrow.paintIcon(this, g, position.x, position.y);
463             } else {
464                 super.paintComponent(g);
465             }
466         }

468         /**
469          * DOCUMENT ME!
470          *
471          * @param  g DOCUMENT ME!
472          *
473          * @return DOCUMENT ME!
474          */
475         private Point computeIconPosition(Graphics g) {
476             FontMetrics fontMetrics = g.getFontMetrics();
477             Rectangle   viewR       = new Rectangle();
478             Rectangle   textR       = new Rectangle();
479             Rectangle   iconR       = new Rectangle();
480             Insets      i           = getInsets();
481 
482             viewR.x      = i.left;
483             viewR.y      = i.top;
484             viewR.width  = getWidth() - (i.left + i.right);
485             viewR.height = getHeight() - (i.top + i.bottom);
486             SwingUtilities.layoutCompoundLabel(this, fontMetrics, getText(), sortArrow, getVerticalAlignment(), getHorizontalAlignment(),
487                                                getVerticalTextPosition(), getHorizontalTextPosition(), viewR, iconR, textR,
488                                                getIconTextGap());
489             int x = getComponentOrientation().isLeftToRight() ? getWidth() - i.right - sortArrow.getIconWidth() : i.left;
490             int y = iconR.y;
491 
492             return new Point(x, y);
493         }
