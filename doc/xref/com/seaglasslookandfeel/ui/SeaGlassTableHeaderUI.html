175         @Override
176         public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
177 
178             boolean hasRollover = false;// (column == getRolloverColumn());
179             if (isSelected || hasRollover || hasFocus) {
180                 SeaGlassLookAndFeel.setSelectedUI((SeaGlassLabelUI) SeaGlassLookAndFeel.getUIOfType(getUI(), SeaGlassLabelUI.class),
181                     isSelected, hasFocus, table.isEnabled(), hasRollover);
182             } else {
183                 SeaGlassLookAndFeel.resetSelectedUI();
184             }
185 
186             // Stuff a variable into the client property of this renderer
187             // indicating the sort order, so that different rendering can be
188             // done for the header based on sorted state.
189             RowSorter rs = table == null ? null : table.getRowSorter();
190             java.util.List<? extends RowSorter.SortKey> sortKeys = rs == null ? null : rs.getSortKeys();
191             if (sortKeys != null && sortKeys.size() > 0 && sortKeys.get(0).getColumn() == table.convertColumnIndexToModel(column)) {
192                 switch (sortKeys.get(0).getSortOrder()) {
193                 case ASCENDING:
194                     putClientProperty("Table.sortOrder", "ASCENDING");
195                     break;
196                 case DESCENDING:
197                     putClientProperty("Table.sortOrder", "DESCENDING");
198                     break;
199                 case UNSORTED:
200                     putClientProperty("Table.sortOrder", "UNSORTED");
201                     break;
202                 default:
203                     throw new AssertionError("Cannot happen");
204                 }
205             } else {
206                 putClientProperty("Table.sortOrder", "UNSORTED");
207             }
208 
209             super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
210 
211             return this;
212         }

236         public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
237             Icon sortIcon = null;
238 
239             boolean isPaintingForPrint = false;
240 
241             if (table != null) {
242                 JTableHeader header = table.getTableHeader();
243                 if (header != null) {
244                     Color fgColor = null;
245                     Color bgColor = null;
246                     if (hasFocus) {
247                         fgColor = DefaultLookup.getColor(this, ui, "TableHeader.focusCellForeground");
248                         bgColor = DefaultLookup.getColor(this, ui, "TableHeader.focusCellBackground");
249                     }
250                     if (fgColor == null) {
251                         fgColor = header.getForeground();
252                     }
253                     if (bgColor == null) {
254                         bgColor = header.getBackground();
255                     }
256                     setForeground(fgColor);
257                     setBackground(bgColor);
258 
259                     setFont(header.getFont());
260 
261                     isPaintingForPrint = header.isPaintingForPrint();
262                 }
263 
264                 if (!isPaintingForPrint && table.getRowSorter() != null) {
265                     if (!horizontalTextPositionSet) {
266                         // There is a row sorter, and the developer hasn't
267                         // set a text position, change to leading.
268                         setHorizontalTextPosition(JLabel.LEADING);
269                     }
270                     SortOrder sortOrder = getColumnSortOrder(table, column);
271                     if (sortOrder != null) {
272                         switch (sortOrder) {
273                         case ASCENDING:
274                             sortIcon = DefaultLookup.getIcon(this, ui, "Table.ascendingSortIcon");
275                             break;
276                         case DESCENDING:
277                             sortIcon = DefaultLookup.getIcon(this, ui, "Table.descendingSortIcon");
278                             break;
279                         case UNSORTED:
280                             sortIcon = DefaultLookup.getIcon(this, ui, "Table.naturalSortIcon");
281                             break;
282                         }
283                     }
284                 }
285             }
286 
287             setText(value == null ? "" : value.toString());
288             setIcon(sortIcon);
289             sortArrow = sortIcon;
290 
291             Border border = null;
292             if (hasFocus) {
293                 border = DefaultLookup.getBorder(this, ui, "TableHeader.focusCellBorder");
294             }
295             if (border == null) {
296                 border = DefaultLookup.getBorder(this, ui, "TableHeader.cellBorder");
297             }
298             setBorder(border);
299 
300             return this;
301         }

303         public static SortOrder getColumnSortOrder(JTable table, int column) {
304             SortOrder rv = null;
305             if (table == null || table.getRowSorter() == null) {
306                 return rv;
307             }
308             java.util.List<? extends RowSorter.SortKey> sortKeys = table.getRowSorter().getSortKeys();
309             if (sortKeys.size() > 0 && sortKeys.get(0).getColumn() == table.convertColumnIndexToModel(column)) {
310                 rv = sortKeys.get(0).getSortOrder();
311             }
312             return rv;
313         }

315         @Override
316         public void paintComponent(Graphics g) {
317             boolean b = DefaultLookup.getBoolean(this, ui, "TableHeader.rightAlignSortArrow", false);
318             if (b && sortArrow != null) {
319                 // emptyIcon is used so that if the text in the header is right
320                 // aligned, or if the column is too narrow, then the text will
321                 // be sized appropriately to make room for the icon that is
322                 // about
323                 // to be painted manually here.
324                 emptyIcon.width = sortArrow.getIconWidth();
325                 emptyIcon.height = sortArrow.getIconHeight();
326                 setIcon(emptyIcon);
327                 super.paintComponent(g);
328                 Point position = computeIconPosition(g);
329                 sortArrow.paintIcon(this, g, position.x, position.y);
330             } else {
331                 super.paintComponent(g);
332             }
333         }

335         private Point computeIconPosition(Graphics g) {
336             FontMetrics fontMetrics = g.getFontMetrics();
337             Rectangle viewR = new Rectangle();
338             Rectangle textR = new Rectangle();
339             Rectangle iconR = new Rectangle();
340             Insets i = getInsets();
341             viewR.x = i.left;
342             viewR.y = i.top;
343             viewR.width = getWidth() - (i.left + i.right);
344             viewR.height = getHeight() - (i.top + i.bottom);
345             SwingUtilities.layoutCompoundLabel(this, fontMetrics, getText(), sortArrow, getVerticalAlignment(), getHorizontalAlignment(),
346                 getVerticalTextPosition(), getHorizontalTextPosition(), viewR, iconR, textR, getIconTextGap());
347             int x = getWidth() - i.right - sortArrow.getIconWidth();
348             int y = iconR.y;
349             return new Point(x, y);
350         }
