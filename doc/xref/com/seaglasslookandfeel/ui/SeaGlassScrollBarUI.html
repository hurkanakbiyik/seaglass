108     private void updateStyle(JScrollBar c) {
109         SynthStyle oldStyle = style;
110         SeaGlassContext context = getContext(c, ENABLED);
111         style = SeaGlassLookAndFeel.updateStyle(context, this);
112         if (style != oldStyle) {
113             scrollBarWidth = style.getInt(context, "ScrollBar.thumbHeight", 14);
114 
115             minimumThumbSize = (Dimension) style.get(context, "ScrollBar.minimumThumbSize");
116             if (minimumThumbSize == null) {
117                 minimumThumbSize = new Dimension();
118                 validMinimumThumbSize = false;
119             } else {
120                 validMinimumThumbSize = true;
121             }
122             maximumThumbSize = (Dimension) style.get(context, "ScrollBar.maximumThumbSize");
123             if (maximumThumbSize == null) {
124                 maximumThumbSize = new Dimension(4096, 4097);
125             }
126 
127             incrGap = style.getInt(context, "ScrollBar.incrementButtonGap", 0);
128             decrGap = style.getInt(context, "ScrollBar.decrementButtonGap", 0);
129             capSize = style.getInt(context, "ScrollBar.capSize", 0);
130 
131             /*
132              * Handle scaling for sizeVarients for special case components. The
133              * key "JComponent.sizeVariant" scales for large/small/mini
134              * components are based on Apples LAF
135              */
136             String scaleKey = (String) scrollbar.getClientProperty("JComponent.sizeVariant");
137             if (scaleKey != null) {
138                 if ("large".equals(scaleKey)) {
139                     scrollBarWidth *= 1.15;
140                     incrGap *= 1.15;
141                     decrGap *= 1.15;
142                 } else if ("small".equals(scaleKey)) {
143                     scrollBarWidth *= 0.857;
144                     incrGap *= 0.857;
145                     decrGap *= 0.857;
146                 } else if ("mini".equals(scaleKey)) {
147                     scrollBarWidth *= 0.714;
148                     incrGap *= 0.714;
149                     decrGap *= 0.714;
150                 }
151             }
152 
153             if (oldStyle != null) {
154                 uninstallKeyboardActions();
155                 installKeyboardActions();
156             }
157         }
158         context.dispose();
159 
160         context = getContext(c, Region.SCROLL_BAR_TRACK, ENABLED);
161         trackStyle = SeaGlassLookAndFeel.updateStyle(context, this);
162         context.dispose();
163 
164         context = getContext(c, Region.SCROLL_BAR_THUMB, ENABLED);
165         thumbStyle = SeaGlassLookAndFeel.updateStyle(context, this);
166         context.dispose();
167 
168         context = getContext(c, SeaGlassRegion.SCROLL_BAR_CAP, ENABLED);
169         capStyle = SeaGlassLookAndFeel.updateStyle(context, this);
170         context.dispose();
171     }

282     protected void paint(SeaGlassContext context, Graphics g) {
283         SeaGlassContext subcontext = getContext(scrollbar, Region.SCROLL_BAR_TRACK);
284         paintTrack(subcontext, g, getTrackBounds());
285         subcontext.dispose();
286 
287         if (buttonsTogether.isInState(context.getComponent())) {
288             subcontext = getContext(scrollbar, SeaGlassRegion.SCROLL_BAR_CAP);
289             paintCap(subcontext, g, getCapBounds());
290             subcontext.dispose();
291         }
292 
293         subcontext = getContext(scrollbar, Region.SCROLL_BAR_THUMB);
294         paintThumb(subcontext, g, getThumbBounds());
295         subcontext.dispose();
296     }

325         if (scrollbar.getOrientation() == JScrollBar.VERTICAL) {
326             return new Rectangle(0, 0, scrollBarWidth, capSize);
327         } else if (scrollbar.getComponentOrientation().isLeftToRight()) {

334     /**
335      * A vertical scrollbar's preferred width is the maximum of preferred widths
336      * of the (non <code>null</code>) increment/decrement buttons, and the
337      * minimum width of the thumb. The preferred height is the sum of the
338      * preferred heights of the same parts. The basis for the preferred size of
339      * a horizontal scrollbar is similar.
340      * <p>
341      * The <code>preferredSize</code> is only computed once, subsequent calls to
342      * this method just return a cached size.
343      * 
344      * @param c
345      *            the <code>JScrollBar</code> that's delegating this method to
346      *            us
347      * @return the preferred size of a Basic JScrollBar
348      * @see #getMaximumSize
349      * @see #getMinimumSize
350      */
351     public Dimension getPreferredSize(JComponent c) {
352         Insets insets = c.getInsets();
353         return (scrollbar.getOrientation() == JScrollBar.VERTICAL) ? new Dimension(scrollBarWidth + insets.left + insets.right, 48)
354                 : new Dimension(48, scrollBarWidth + insets.top + insets.bottom);
355     }

357     protected Dimension getMinimumThumbSize() {
358         if (!validMinimumThumbSize) {
359             if (scrollbar.getOrientation() == JScrollBar.VERTICAL) {
360                 minimumThumbSize.width = scrollBarWidth;
361                 minimumThumbSize.height = 7;
362             } else {
363                 minimumThumbSize.width = 7;
364                 minimumThumbSize.height = scrollBarWidth;
365             }
366         }
367         return minimumThumbSize;
368     }

370     protected JButton createDecreaseButton(int orientation) {
371         SeaGlassArrowButton synthArrowButton = new SeaGlassArrowButton(orientation) {
372             @Override
373             public boolean contains(int x, int y) {
374                 // if there is an overlap between the track and button
375                 if (decrGap < 0) {
376                     // FIXME Need to take RtL orientation into account.
377                     if (buttonsTogether.isInState(scrollbar)) {
378                         int minX = 0;
379                         int minY = 0;
380                         if (scrollbar.getOrientation() == JScrollBar.VERTICAL) {
381                             // adjust the height by decrGap
382                             // Note: decrGap is negative!
383                             minY -= decrGap;
384                         } else {
385                             // adjust the width by decrGap
386                             // Note: decrGap is negative!
387                             minX -= decrGap;
388                         }
389                         return (x >= minX) && (x < getWidth()) && (y >= minY) && (y < getHeight());
390                     } else {
391                         int width = getWidth();
392                         int height = getHeight();
393                         if (scrollbar.getOrientation() == JScrollBar.VERTICAL) {
394                             // adjust the height by decrGap
395                             // Note: decrGap is negative!
396                             height += decrGap;
397                         } else {
398                             // adjust the width by decrGap
399                             // Note: decrGap is negative!
400                             width += decrGap;
401                         }
402                         return (x >= 0) && (x < width) && (y >= 0) && (y < height);
403                     }
404                 }
405                 return super.contains(x, y);
406             }
407         };
408         synthArrowButton.setName("ScrollBar.button");
409         return synthArrowButton;
410     }

412     protected JButton createIncreaseButton(int orientation) {
413         SeaGlassArrowButton synthArrowButton = new SeaGlassArrowButton(orientation) {
414             @Override
415             public boolean contains(int x, int y) {
416                 // if there is an overlap between the track and button
417                 if (incrGap < 0 && !buttonsTogether.isInState(scrollbar)) {
418                     int width = getWidth();
419                     int height = getHeight();
420                     if (scrollbar.getOrientation() == JScrollBar.VERTICAL) {
421                         // adjust the height and y by incrGap
422                         // Note: incrGap is negative!
423                         height += incrGap;
424                         y += incrGap;
425                     } else {
426                         // adjust the width and x by incrGap
427                         // Note: incrGap is negative!
428                         width += incrGap;
429                         x += incrGap;
430                     }
431                     return (x >= 0) && (x < width) && (y >= 0) && (y < height);
432                 }
433                 return super.contains(x, y);
434             }
435         };
436         synthArrowButton.setName("ScrollBar.button");
437         return synthArrowButton;
438     }

448         int orient = scrollbar.getOrientation();
449         if (scrollbar.getComponentOrientation().isLeftToRight()) {
450             ((SeaGlassArrowButton) incrButton).setDirection(orient == HORIZONTAL ? EAST : SOUTH);
451             ((SeaGlassArrowButton) decrButton).setDirection(orient == HORIZONTAL ? WEST : NORTH);
452         } else {
453             ((SeaGlassArrowButton) incrButton).setDirection(orient == HORIZONTAL ? WEST : SOUTH);
454             ((SeaGlassArrowButton) decrButton).setDirection(orient == HORIZONTAL ? EAST : NORTH);
455         }

496         ScrollbarLayoutValues lv = new ScrollbarLayoutValues();

499         Insets sbInsets = sb.getInsets();
500 
501         layoutScrollbarTogether(sb, lv, sbSize.height, sbSize.width, sbInsets.top, sbInsets.bottom, sbInsets.left, sbInsets.right,
502             decrButton.getPreferredSize().height, getMinimumThumbSize().height, getMaximumThumbSize().height);

504         trackRect.setBounds(lv.itemEdge, lv.trackPosition, lv.itemThickness, lv.trackLength);

506         incrButton.setBounds(lv.itemEdge, lv.incrButtonPosition, lv.itemThickness, lv.incrButtonLength);
507         if (lv.thumbLength > 0) {
508             setThumbBounds(lv.itemEdge, lv.thumbPosition, lv.itemThickness, lv.thumbLength);
509         } else {
510             setThumbBounds(0, 0, 0, 0);
511         }

533     private void layoutHScrollbarTogetherRightToLeft(JScrollBar sb) {
534         ScrollbarLayoutValues lv = new ScrollbarLayoutValues();
535 
536         Dimension sbSize = sb.getSize();
537         Insets sbInsets = sb.getInsets();
538 
539         layoutScrollbarTogether(sb, lv, sbSize.width, sbSize.height, sbInsets.left, sbInsets.right, sbInsets.top, sbInsets.bottom,
540             decrButton.getPreferredSize().width, getMinimumThumbSize().width, getMaximumThumbSize().width);
541 
542         // Flip the positions of the buttons.
543         lv.incrButtonPosition = sbInsets.left;
544         lv.decrButtonPosition = lv.incrButtonPosition + lv.incrButtonLength;
545         // Make the thumb position relative to the old track and flip it,
546         // keeping the position at the left.
547         lv.thumbPosition = lv.trackPosition + lv.trackLength - lv.thumbPosition - lv.thumbLength;
548         // Flip the position of the track.
549         lv.trackPosition = lv.decrButtonPosition + lv.decrButtonLength + incrGap;
550         // Make the thumb position relative to the scrollbar.
551         lv.thumbPosition += lv.trackPosition;
552 
553         trackRect.setBounds(lv.trackPosition, lv.itemEdge, lv.trackLength, lv.itemThickness);
554         decrButton.setBounds(lv.decrButtonPosition, lv.itemEdge, lv.decrButtonLength, lv.itemThickness);
555         incrButton.setBounds(lv.incrButtonPosition, lv.itemEdge, lv.incrButtonLength, lv.itemThickness);
556         if (lv.thumbLength > 0) {
557             setThumbBounds(lv.thumbPosition, lv.itemEdge, lv.thumbLength, lv.itemThickness);
558         } else {
559             setThumbBounds(0, 0, 0, 0);
560         }

563     /**
564      * Holds scrollbar layout values in an orientation-independent way.
565      */
566     private class ScrollbarLayoutValues {
567         int itemEdge;
568         int itemThickness;
569         int trackPosition;
570         int trackLength;
571         int incrButtonPosition;
572         int incrButtonLength;
573         int decrButtonPosition;
574         int decrButtonLength;
575         int thumbPosition;
576         int thumbLength;
577     }

579     private void layoutScrollbarTogether(JScrollBar sb, ScrollbarLayoutValues lv, int sbLength, int sbThickness, int insetLengthStart,
580         int insetLengthEnd, int insetThicknessStart, int insetThicknessEnd, int decrPreferredLength, int minThumbLength, int maxThumbLength) {
581         /*
582          * Width and left edge of the buttons and thumb.
583          */
584         lv.itemThickness = sbThickness - (insetThicknessStart + insetThicknessEnd);
585         lv.itemEdge = insetThicknessStart;
586 
587         /*
588          * Nominal locations of the buttons, assuming their preferred size will
589          * fit.
590          */
591         boolean squareButtons = DefaultLookup.getBoolean(scrollbar, this, "ScrollBar.squareButtons", false);
592 
593         lv.incrButtonLength = lv.itemThickness + 1;
594         lv.incrButtonPosition = sbLength - insetLengthEnd - lv.incrButtonLength;
595 
596         lv.decrButtonLength = squareButtons ? lv.itemThickness : decrPreferredLength;
597         lv.decrButtonPosition = lv.incrButtonPosition - lv.decrButtonLength;
598 
599         /*
600          * The thumb must fit within the height left over after we subtract the
601          * preferredSize of the buttons and the insets and the gaps
602          */
603         int sbInsetsSpace = insetLengthStart + insetLengthEnd;
604         int sbButtonsSpace = lv.decrButtonLength + lv.incrButtonLength;
605 
606         /*
607          * If the buttons don't fit, allocate half of the available space to
608          * each and move the lower one (incrButton) down.
609          */
610         int sbAvailButtonSpace = sbLength - sbInsetsSpace;
611         if (sbAvailButtonSpace < sbButtonsSpace) {
612             lv.incrButtonLength = lv.decrButtonLength = sbAvailButtonSpace / 2;
613             lv.incrButtonPosition = sbLength - insetLengthEnd - lv.incrButtonLength;
614             lv.decrButtonPosition = insetLengthStart;
615         }
616 
617         /*
618          * Update the trackRect field.
619          */
620         lv.trackPosition = insetLengthStart + capSize + decrGap;
621         lv.trackLength = lv.decrButtonPosition - incrGap - lv.trackPosition;
622         int gaps = decrGap + incrGap;
623         float trackLength = sbLength - sbInsetsSpace - sbButtonsSpace - gaps - capSize;
624 
625         /*
626          * Compute the height and origin of the thumb. The case where the thumb
627          * is at the bottom edge is handled specially to avoid numerical
628          * problems in computing thumbY. Enforce the thumbs min/max dimensions.
629          * If the thumb doesn't fit in the track (trackH) we'll hide it later.
630          */
631         float min = sb.getMinimum();
632         float max = sb.getMaximum();
633         float extent = sb.getVisibleAmount();
634         float range = max - min;
635         float value = sb.getValue();
636 
637         /*
638          * Update the thumb bounds.
639          */
640         lv.thumbLength = (range <= 0) ? maxThumbLength : (int) (trackLength * (extent / range));
641         lv.thumbLength = Math.min(Math.max(lv.thumbLength, minThumbLength), maxThumbLength);
642         int maxThumbPosition = lv.trackPosition + lv.trackLength - lv.thumbLength;
643         lv.thumbPosition = maxThumbPosition;
644         if (value < (max - extent)) {
645             float thumbRange = lv.trackLength - lv.thumbLength;
646             lv.thumbPosition = (int) (0.5f + (thumbRange * ((value - min) / (range - extent))));
647             lv.thumbPosition += lv.trackPosition;
648         }
649 
650         /*
651          * If the thumb isn't going to fit, zero its bounds. Otherwise make sure
652          * it fits between the buttons. Note that setting the thumbs bounds will
653          * cause a repaint.
654          */
655         if (lv.thumbLength >= (int) trackLength) {
656             lv.thumbLength = 0;
657         } else {
658             if (lv.thumbPosition > maxThumbPosition) {
659                 lv.thumbPosition = maxThumbPosition;
660             }
661             if (lv.thumbPosition < lv.trackPosition) {
662                 lv.thumbPosition = lv.trackPosition;
663             }
664         }
665     }
