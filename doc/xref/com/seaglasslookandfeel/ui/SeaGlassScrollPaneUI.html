103     protected void installDefaults(JScrollPane scrollpane) {
104         LookAndFeel.installBorder(scrollpane, "ScrollPane.border");
105         LookAndFeel.installColorsAndFont(scrollpane, "ScrollPane.background", "ScrollPane.foreground", "ScrollPane.font");
106 
107         Border vpBorder = scrollpane.getViewportBorder();
108         if ((vpBorder == null) || (vpBorder instanceof UIResource)) {
109             vpBorder = UIManager.getBorder("ScrollPane.viewportBorder");
110             scrollpane.setViewportBorder(vpBorder);
111         }
112 
113         Object obj = UIManager.get("ScrollPane.cornerPainter");
114         if (obj != null && obj instanceof SeaGlassPainter) {
115             cornerPainter = (SeaGlassPainter) obj;
116         }
117 
118         LookAndFeel.installProperty(scrollpane, "opaque", Boolean.TRUE);
119         updateStyle(scrollpane);
120     }

125         style.uninstallDefaults(context);
126         context.dispose();
127 
128         if (scrollpane.getViewportBorder() instanceof UIResource) {
129             scrollpane.setViewportBorder(null);
130         }

133     private void updateStyle(JScrollPane c) {
134         SeaGlassContext context = getContext(c, ENABLED);
135         SynthStyle oldStyle = style;
136 
137         style = SeaGlassLookAndFeel.updateStyle(context, this);
138         if (style != oldStyle) {
139             Border vpBorder = scrollpane.getViewportBorder();
140             if ((vpBorder == null) || (vpBorder instanceof UIResource)) {
141                 scrollpane.setViewportBorder(new ViewportBorder(context));
142             }
143             if (oldStyle != null) {
144                 uninstallKeyboardActions(c);
145                 installKeyboardActions(c);
146             }
147         }
148         context.dispose();
149     }

151     protected void installListeners(JScrollPane c) {
152         vsbChangeListener = createVSBChangeListener();
153         vsbPropertyChangeListener = createVSBPropertyChangeListener();
154         hsbChangeListener = createHSBChangeListener();
155         hsbPropertyChangeListener = createHSBPropertyChangeListener();
156         viewportChangeListener = createViewportChangeListener();
157         spPropertyChangeListener = createPropertyChangeListener();
158 
159         JViewport viewport = scrollpane.getViewport();
160         JScrollBar vsb = scrollpane.getVerticalScrollBar();
161         JScrollBar hsb = scrollpane.getHorizontalScrollBar();
162 
163         if (viewport != null) {
164             viewport.addChangeListener(viewportChangeListener);
165         }
166         if (vsb != null) {
167             vsb.getModel().addChangeListener(vsbChangeListener);
168             vsb.addPropertyChangeListener(vsbPropertyChangeListener);
169         }
170         if (hsb != null) {
171             hsb.getModel().addChangeListener(hsbChangeListener);
172             hsb.addPropertyChangeListener(hsbPropertyChangeListener);
173         }
174 
175         scrollpane.addPropertyChangeListener(spPropertyChangeListener);
176 
177         mouseScrollListener = createMouseWheelListener();
178         scrollpane.addMouseWheelListener(mouseScrollListener);
179 
180         // From SynthScrollPaneUI.
181         c.addPropertyChangeListener(this);
182         if (UIManager.getBoolean("ScrollPane.useChildTextComponentFocus")) {
183             viewportViewFocusHandler = new ViewportViewFocusHandler();
184             c.getViewport().addContainerListener(viewportViewFocusHandler);
185             Component view = c.getViewport().getView();
186             if (view instanceof JTextComponent) {
187                 view.addFocusListener(viewportViewFocusHandler);
188             }
189         }

192     protected void uninstallListeners(JComponent c) {
193         JViewport viewport = scrollpane.getViewport();
194         JScrollBar vsb = scrollpane.getVerticalScrollBar();
195         JScrollBar hsb = scrollpane.getHorizontalScrollBar();
196 
197         if (viewport != null) {
198             viewport.removeChangeListener(viewportChangeListener);
199         }
200         if (vsb != null) {
201             vsb.getModel().removeChangeListener(vsbChangeListener);
202             vsb.removePropertyChangeListener(vsbPropertyChangeListener);
203         }
204         if (hsb != null) {
205             hsb.getModel().removeChangeListener(hsbChangeListener);
206             hsb.removePropertyChangeListener(hsbPropertyChangeListener);
207         }
208 
209         scrollpane.removePropertyChangeListener(spPropertyChangeListener);
210 
211         if (mouseScrollListener != null) {
212             scrollpane.removeMouseWheelListener(mouseScrollListener);
213         }
214 
215         vsbChangeListener = null;
216         hsbChangeListener = null;
217         viewportChangeListener = null;
218         spPropertyChangeListener = null;
219         mouseScrollListener = null;
220         handler = null;
221 
222         // From SynthScrollPaneUI.
223         c.removePropertyChangeListener(this);
224         if (viewportViewFocusHandler != null) {
225             viewport.removeContainerListener(viewportViewFocusHandler);
226             if (viewport.getView() != null) {
227                 viewport.getView().removeFocusListener(viewportViewFocusHandler);
228             }
229             viewportViewFocusHandler = null;
230         }

255     public void update(Graphics g, JComponent c) {
256         SeaGlassContext context = getContext(c);
257 
258         SeaGlassLookAndFeel.update(context, g);
259         context.getPainter().paintScrollPaneBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
260         paintScrollPaneCorner(g, c);
261         paint(context, g);
262         context.dispose();

266      * @param g
267      * @param c
268      */
269     private void paintScrollPaneCorner(Graphics g, JComponent c) {
270         if (scrollpane == null) {
271             return;
272         }
273         if (scrollpane.getHorizontalScrollBar() == null || !scrollpane.getHorizontalScrollBar().isVisible()) {
274             return;
275         }
276         if (scrollpane.getVerticalScrollBar() == null || !scrollpane.getVerticalScrollBar().isVisible()) {
277             return;
278         }
279 
280         int vBarWidth = scrollpane.getVerticalScrollBar().getWidth();
281         int hBarHeight = scrollpane.getHorizontalScrollBar().getHeight();
282 
283         Insets insets = c.getInsets();
284 
285         Graphics2D g2 = (Graphics2D) g.create();
286         if (scrollpane.getComponentOrientation().isLeftToRight()) {
287             g2.translate(c.getWidth() - insets.right - vBarWidth, c.getHeight() - insets.bottom - hBarHeight);
288             g2.setClip(0, 0, vBarWidth, hBarHeight);
289         } else {
290             g2.translate(15 + insets.right, c.getHeight() - insets.bottom - hBarHeight);
291             g2.scale(-1, 1);
292             g2.setClip(0, 0, vBarWidth, hBarHeight);
293         }
294         cornerPainter.paint(g2, c, 15, 15);
295     }

323     protected void syncScrollPaneWithViewport() {
324         JViewport viewport = scrollpane.getViewport();
325         JScrollBar vsb = scrollpane.getVerticalScrollBar();
326         JScrollBar hsb = scrollpane.getHorizontalScrollBar();
327         JViewport rowHead = scrollpane.getRowHeader();
328         JViewport colHead = scrollpane.getColumnHeader();
329         boolean ltr = scrollpane.getComponentOrientation().isLeftToRight();
330 
331         if (viewport != null) {
332             Dimension extentSize = viewport.getExtentSize();
333             Dimension viewSize = viewport.getViewSize();
334             Point viewPosition = viewport.getViewPosition();
335 
336             if (vsb != null) {
337                 int extent = extentSize.height;
338                 int max = viewSize.height;
339                 int value = Math.max(0, Math.min(viewPosition.y, max - extent));
340                 vsb.setValues(value, extent, 0, max);
341             }
342 
343             if (hsb != null) {
344                 int extent = extentSize.width;
345                 int max = viewSize.width;
346                 int value;
347 
348                 if (ltr) {
349                     value = Math.max(0, Math.min(viewPosition.x, max - extent));
350                 } else {
351                     int currentValue = hsb.getValue();
352 
353                     /*
354                      * Use a particular formula to calculate "value" until
355                      * effective x coordinate is calculated.
356                      */
357                     if (setValueCalled && ((max - currentValue) == viewPosition.x)) {
358                         value = Math.max(0, Math.min(max - extent, currentValue));
359                         /*
360                          * After "extent" is set, turn setValueCalled flag off.
361                          */
362                         if (extent != 0) {
363                             setValueCalled = false;
364                         }
365                     } else {
366                         if (extent > max) {
367                             viewPosition.x = max - extent;
368                             viewport.setViewPosition(viewPosition);
369                             value = 0;
370                         } else {
371                             /*
372                              * The following line can't handle a small value of
373                              * viewPosition.x like Integer.MIN_VALUE correctly
374                              * because (max - extent - viewPositoiin.x) causes
375                              * an overflow. As a result, value becomes zero.
376                              * (e.g. setViewPosition(Integer.MAX_VALUE, ...) in
377                              * a user program causes a overflow. Its expected
378                              * value is (max - extent).) However, this seems a
379                              * trivial bug and adding a fix makes this
380                              * often-called method slow, so I'll leave it until
381                              * someone claims.
382                              */
383                             value = Math.max(0, Math.min(max - extent, max - extent - viewPosition.x));
384                         }
385                     }
386                 }
387                 hsb.setValues(value, extent, 0, max);
388             }
389 
390             if (rowHead != null) {
391                 Point p = rowHead.getViewPosition();
392                 p.y = viewport.getViewPosition().y;
393                 p.x = 0;
394                 rowHead.setViewPosition(p);
395             }
396 
397             if (colHead != null) {
398                 Point p = colHead.getViewPosition();
399                 if (ltr) {
400                     p.x = viewport.getViewPosition().x;
401                 } else {
402                     p.x = Math.max(0, viewport.getViewPosition().x);
403                 }
404                 p.y = 0;
405                 colHead.setViewPosition(p);
406             }
407         }
408     }

545             }
546         }
547         sb = (JScrollBar) pce.getNewValue();
548         if (sb != null) {
549             if (cl != null) {
550                 sb.getModel().addChangeListener(cl);
551             }
552             if (pcl != null) {
553                 sb.addPropertyChangeListener(pcl);
554             }

598     class Handler implements ChangeListener, PropertyChangeListener, MouseWheelListener {
599         //
600         // MouseWheelListener
601         //
602         public void mouseWheelMoved(MouseWheelEvent e) {
603             if (scrollpane.isWheelScrollingEnabled() && e.getWheelRotation() != 0) {
604                 boolean isHorizontal = (e.getModifiersEx() & MouseWheelEvent.SHIFT_DOWN_MASK) != 0;
605                 JScrollBar toScroll = scrollpane.getVerticalScrollBar();
606                 int direction = e.getWheelRotation() < 0 ? -1 : 1;
607                 int orientation = SwingConstants.VERTICAL;
608 
609                 // find which scrollbar to scroll, or return if none
610                 if (toScroll == null || !toScroll.isVisible() || isHorizontal) {
611                     toScroll = scrollpane.getHorizontalScrollBar();
612                     if (toScroll == null || !toScroll.isVisible()) {
613                         return;
614                     }
615                     orientation = SwingConstants.HORIZONTAL;
616                 }
617 
618                 if (e.getScrollType() == MouseWheelEvent.WHEEL_UNIT_SCROLL) {
619                     JViewport vp = scrollpane.getViewport();
620                     if (vp == null) {
621                         return;
622                     }
623                     Component comp = vp.getView();
624                     int units = Math.abs(e.getUnitsToScroll());
625 
626                     // When the scrolling speed is set to maximum, it's possible
627                     // for a single wheel click to scroll by more units than
628                     // will fit in the visible area. This makes it
629                     // hard/impossible to get to certain parts of the scrolling
630                     // Component with the wheel. To make for more accurate
631                     // low-speed scrolling, we limit scrolling to the block
632                     // increment if the wheel was only rotated one click.
633                     boolean limitScroll = Math.abs(e.getWheelRotation()) == 1;
634 
635                     // Check if we should use the visibleRect trick
636                     Object fastWheelScroll = toScroll.getClientProperty("JScrollBar.fastWheelScrolling");
637                     if (Boolean.TRUE == fastWheelScroll && comp instanceof Scrollable) {
638                         // 5078454: Under maximum acceleration, we may scroll
639                         // by many 100s of units in ~1 second.
640                         //
641                         // BasicScrollBarUI.scrollByUnits() can bog down the EDT
642                         // with repaints in this situation. However, the
643                         // Scrollable interface allows us to pass in an
644                         // arbitrary visibleRect. This allows us to accurately
645                         // calculate the total scroll amount, and then update
646                         // the GUI once. This technique provides much faster
647                         // accelerated wheel scrolling.
648                         Scrollable scrollComp = (Scrollable) comp;
649                         Rectangle viewRect = vp.getViewRect();
650                         int startingX = viewRect.x;
651                         boolean leftToRight = comp.getComponentOrientation().isLeftToRight();
652                         int scrollMin = toScroll.getMinimum();
653                         int scrollMax = toScroll.getMaximum() - toScroll.getModel().getExtent();
654 
655                         if (limitScroll) {
656                             int blockIncr = scrollComp.getScrollableBlockIncrement(viewRect, orientation, direction);
657                             if (direction < 0) {
658                                 scrollMin = Math.max(scrollMin, toScroll.getValue() - blockIncr);
659                             } else {
660                                 scrollMax = Math.min(scrollMax, toScroll.getValue() + blockIncr);
661                             }
662                         }
663 
664                         for (int i = 0; i < units; i++) {
665                             int unitIncr = scrollComp.getScrollableUnitIncrement(viewRect, orientation, direction);
666                             // Modify the visible rect for the next unit, and
667                             // check to see if we're at the end already.
668                             if (orientation == SwingConstants.VERTICAL) {
669                                 if (direction < 0) {
670                                     viewRect.y -= unitIncr;
671                                     if (viewRect.y <= scrollMin) {
672                                         viewRect.y = scrollMin;
673                                         break;
674                                     }
675                                 } else { // (direction > 0
676                                     viewRect.y += unitIncr;
677                                     if (viewRect.y >= scrollMax) {
678                                         viewRect.y = scrollMax;
679                                         break;
680                                     }
681                                 }
682                             } else {
683                                 // Scroll left
684                                 if ((leftToRight && direction < 0) || (!leftToRight && direction > 0)) {
685                                     viewRect.x -= unitIncr;
686                                     if (leftToRight) {
687                                         if (viewRect.x < scrollMin) {
688                                             viewRect.x = scrollMin;
689                                             break;
690                                         }
691                                     }
692                                 }
693                                 // Scroll right
694                                 else if ((leftToRight && direction > 0) || (!leftToRight && direction < 0)) {
695                                     viewRect.x += unitIncr;
696                                     if (leftToRight) {
697                                         if (viewRect.x > scrollMax) {
698                                             viewRect.x = scrollMax;
699                                             break;
700                                         }
701                                     }
702                                 } else {
703                                     assert false : "Non-sensical ComponentOrientation / scroll direction";
704                                 }
705                             }
706                         }
707                         // Set the final view position on the ScrollBar
708                         if (orientation == SwingConstants.VERTICAL) {
709                             toScroll.setValue(viewRect.y);
710                         } else {
711                             if (leftToRight) {
712                                 toScroll.setValue(viewRect.x);
713                             } else {
714                                 // rightToLeft scrollbars are oriented with
715                                 // minValue on the right and maxValue on the
716                                 // left.
717                                 int newPos = toScroll.getValue() - (viewRect.x - startingX);
718                                 if (newPos < scrollMin) {
719                                     newPos = scrollMin;
720                                 } else if (newPos > scrollMax) {
721                                     newPos = scrollMax;
722                                 }
723                                 toScroll.setValue(newPos);
724                             }
725                         }
726                     } else {
727                         // Viewport's view is not a Scrollable, or fast wheel
728                         // scrolling is not enabled.
729                         scrollByUnits(toScroll, direction, units, limitScroll);
730                     }
731                 } else if (e.getScrollType() == MouseWheelEvent.WHEEL_BLOCK_SCROLL) {
732                     scrollByBlock(toScroll, direction);
733                 }
734                 e.consume();
735             }
736         }

738         /*
739          * Method for scrolling by a block increment. Added for mouse wheel
740          * scrolling support, RFE 4202656.
741          */
742         void scrollByBlock(JScrollBar scrollbar, int direction) {
743             // This method is called from BasicScrollPaneUI to implement wheel
744             // scrolling, and also from scrollByBlock().
745             int oldValue = scrollbar.getValue();
746             int blockIncrement = scrollbar.getBlockIncrement(direction);
747             int delta = blockIncrement * ((direction > 0) ? +1 : -1);
748             int newValue = oldValue + delta;
749 
750             // Check for overflow.
751             if (delta > 0 && newValue < oldValue) {
752                 newValue = scrollbar.getMaximum();
753             } else if (delta < 0 && newValue > oldValue) {
754                 newValue = scrollbar.getMinimum();
755             }
756 
757             scrollbar.setValue(newValue);
758         }

760         /*
761          * Method for scrolling by a unit increment. Added for mouse wheel
762          * scrolling support, RFE 4202656.
763          * 
764          * If limitByBlock is set to true, the scrollbar will scroll at least 1
765          * unit increment, but will not scroll farther than the block increment.
766          * See BasicScrollPaneUI.Handler.mouseWheelMoved().
767          */
768         void scrollByUnits(JScrollBar scrollbar, int direction, int units, boolean limitToBlock) {
769             // This method is called from BasicScrollPaneUI to implement wheel
770             // scrolling, as well as from scrollByUnit().
771             int delta;
772             int limit = -1;
773 
774             if (limitToBlock) {
775                 if (direction < 0) {
776                     limit = scrollbar.getValue() - scrollbar.getBlockIncrement(direction);
777                 } else {
778                     limit = scrollbar.getValue() + scrollbar.getBlockIncrement(direction);
779                 }
780             }
781 
782             for (int i = 0; i < units; i++) {
783                 if (direction > 0) {
784                     delta = scrollbar.getUnitIncrement(direction);
785                 } else {
786                     delta = -scrollbar.getUnitIncrement(direction);
787                 }
788 
789                 int oldValue = scrollbar.getValue();
790                 int newValue = oldValue + delta;
791 
792                 // Check for overflow.
793                 if (delta > 0 && newValue < oldValue) {
794                     newValue = scrollbar.getMaximum();
795                 } else if (delta < 0 && newValue > oldValue) {
796                     newValue = scrollbar.getMinimum();
797                 }
798                 if (oldValue == newValue) {
799                     break;
800                 }
801 
802                 if (limitToBlock && i > 0) {
803                     assert limit != -1;
804                     if ((direction < 0 && newValue < limit) || (direction > 0 && newValue > limit)) {
805                         break;
806                     }
807                 }
808                 scrollbar.setValue(newValue);
809             }
810         }

842         private void hsbStateChanged(JViewport viewport, ChangeEvent e) {
843             BoundedRangeModel model = (BoundedRangeModel) (e.getSource());
844             Point p = viewport.getViewPosition();
845             int value = model.getValue();
846             if (scrollpane.getComponentOrientation().isLeftToRight()) {
847                 p.x = value;
848             } else {
849                 int max = viewport.getViewSize().width;
850                 int extent = viewport.getExtentSize().width;
851                 int oldX = p.x;
852 
853                 /*
854                  * Set new X coordinate based on "value".
855                  */
856                 p.x = max - extent - value;
857 
858                 /*
859                  * If setValue() was called before "extent" was fixed, turn
860                  * setValueCalled flag on.
861                  */
862                 if ((extent == 0) && (value != 0) && (oldX == max)) {
863                     setValueCalled = true;
864                 } else {
865                     /*
866                      * When a pane without a horizontal scroll bar was reduced
867                      * and the bar appeared, the viewport should show the right
868                      * side of the view.
869                      */
870                     if ((extent != 0) && (oldX < 0) && (p.x == 0)) {
871                         p.x += value;
872                     }
873                 }
874             }
875             viewport.setViewPosition(p);
876         }
