124     /**
125      * DOCUMENT ME!
126      *
127      * @param c DOCUMENT ME!
128      */
129     private void updateStyle(JTextComponent c) {
130         SeaGlassContext context  = getContext(c, ENABLED);
131         SynthStyle      oldStyle = style;
132 
133         style = SeaGlassLookAndFeel.updateStyle(context, this);
134 
135         updateSearchStyle(c, context, getPropertyPrefix());
136 
137         if (style != oldStyle) {
138             updateStyle(c, context, getPropertyPrefix());
139 
140             if (oldStyle != null) {
141                 uninstallKeyboardActions();
142                 installKeyboardActions();
143             }
144         }
145 
146         context.dispose();
147 
148         context   = getContext(c, SeaGlassRegion.SEARCH_FIELD_FIND_BUTTON, ENABLED);
149         findStyle = SeaGlassLookAndFeel.updateStyle(context, this);
150         context.dispose();
151 
152         context     = getContext(c, SeaGlassRegion.SEARCH_FIELD_CANCEL_BUTTON, ENABLED);
153         cancelStyle = SeaGlassLookAndFeel.updateStyle(context, this);
154         context.dispose();
155     }

157     /**
158      * Sea Glass code to support the search JTextField.variant.
159      *
160      * @param c       the JTextField component.
161      * @param context the SeaGlassContext.
162      * @param prefix  the control prefix, e.g. "TextField",
163      *                "FormattedTextField", or "PasswordField".
164      */
165     private void updateSearchStyle(JTextComponent c, SeaGlassContext context, String prefix) {
166         searchIconWidth = 0;
167         Object o = style.get(context, prefix + ".searchIconWidth");
168 
169         if (o != null && o instanceof Integer) {
170             searchIconWidth = (Integer) o;
171         }
172 
173         popupIconWidth = 0;
174         o              = style.get(context, prefix + ".popupIconWidth");
175         if (o != null && o instanceof Integer) {
176             popupIconWidth = (Integer) o;
177         }
178 
179         cancelIconWidth = 0;
180         o               = style.get(context, prefix + ".cancelIconWidth");
181         if (o != null && o instanceof Integer) {
182             cancelIconWidth = (Integer) o;
183         }
184 
185         searchLeftInnerMargin = 0;
186         o                     = style.get(context, prefix + ".searchLeftInnerMargin");
187         if (o != null && o instanceof Integer) {
188             searchLeftInnerMargin = (Integer) o;
189         }
190 
191         searchRightInnerMargin = 0;
192         o                      = style.get(context, prefix + ".searchRightInnerMargin");
193         if (o != null && o instanceof Integer) {
194             searchRightInnerMargin = (Integer) o;
195         }
196 
197         placeholderColor = Color.GRAY;
198         o                = style.get(context, "placeholderTextColor");
199         if (o != null && o instanceof Color) {
200             placeholderColor = (Color) o;
201         }
202 
203         Border border = c.getBorder();
204 
205         if (border == null || border instanceof UIResource && !(border instanceof TextFieldBorder)) {
206             c.setBorder(createTextFieldBorder(context));
207         }
208 
209         if (isSearchField.isInState(c)) {
210             o = c.getClientProperty("JTextField.Search.PlaceholderText");
211             if (o != null && o instanceof String) {
212                 placeholderText = (String) o;
213             } else if (placeholderText != null) {
214                 placeholderText = null;
215             }
216 
217             o = c.getClientProperty("JTextField.Search.FindAction");
218             if (o != null && o instanceof ActionListener) {
219                 if (findAction == null) {
220                     findAction = (ActionListener) o;
221                 }
222             }
223 
224             o = c.getClientProperty("JTextField.Search.FindPopup");
225             if (o != null && o instanceof JPopupMenu) {
226                 if (findPopup == null) {
227                     findPopup = (JPopupMenu) o;
228                 }
229             }
230 
231             o = c.getClientProperty("JTextField.Search.CancelAction");
232             if (o != null && o instanceof ActionListener) {
233                 if (cancelAction == null) {
234                     cancelAction = (ActionListener) o;
235                 }
236             }
237 
238             installMouseListeners();
239         } else {
240             placeholderText = null;
241 
242             if (findAction != null) {
243                 findAction = null;
244             }
245 
246             if (findPopup != null) {
247                 findPopup = null;
248             }
249 
250             if (cancelAction != null) {
251                 cancelAction = null;
252             }
253 
254             uninstallMouseListeners();
255         }
256     }

258     /**
259      * Private method to update styles.
260      *
261      * @param c       the JTextField component.
262      * @param context the SeaGlassContext.
263      * @param prefix  the control prefix, e.g. "TextField",
264      *                "FormattedTextField", or "PasswordField".
265      */
266     private void updateStyle(JTextComponent c, SeaGlassContext context, String prefix) {
267         SeaGlassStyle style = (SeaGlassStyle) context.getStyle();
268 
269         Color color = c.getCaretColor();
270 
271         if (color == null || color instanceof UIResource) {
272             c.setCaretColor((Color) style.get(context, prefix + ".caretForeground"));
273         }
274 
275         Color fg = c.getForeground();
276 
277         if (fg == null || fg instanceof UIResource) {
278             fg = style.getColorForState(context, ColorType.TEXT_FOREGROUND);
279             if (fg != null) {
280                 c.setForeground(fg);
281             }
282         }
283 
284         Object ar = style.get(context, prefix + ".caretAspectRatio");
285 
286         if (ar instanceof Number) {
287             c.putClientProperty("caretAspectRatio", ar);
288         }
289 
290         context.setComponentState(SELECTED | FOCUSED);
291 
292         Color s = c.getSelectionColor();
293 
294         if (s == null || s instanceof UIResource) {
295             c.setSelectionColor(style.getColor(context, ColorType.TEXT_BACKGROUND));
296         }
297 
298         Color sfg = c.getSelectedTextColor();
299 
300         if (sfg == null || sfg instanceof UIResource) {
301             c.setSelectedTextColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
302         }
303 
304         context.setComponentState(DISABLED);
305 
306         Color dfg = c.getDisabledTextColor();
307 
308         if (dfg == null || dfg instanceof UIResource) {
309             c.setDisabledTextColor(style.getColor(context, ColorType.TEXT_FOREGROUND));
310         }
311 
312         Insets margin = c.getMargin();
313 
314         if (margin == null || margin instanceof UIResource) {
315             margin = (Insets) style.get(context, prefix + ".margin");
316             if (margin == null) {
317                 // Some places assume margins are non-null.
318                 margin = SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS;
319             }
320 
321             c.setMargin(margin);
322         }
323 
324         Caret caret = c.getCaret();
325 
326         if (caret instanceof UIResource) {
327             Object o = style.get(context, prefix + ".caretBlinkRate");
328 
329             if (o != null && o instanceof Integer) {
330                 Integer rate = (Integer) o;
331 
332                 caret.setBlinkRate(rate.intValue());
333             }
334         }
335     }

494     protected void paintPlaceholderText(SeaGlassContext context, Graphics g, JComponent c) {
495         g.setColor(placeholderColor);
496         g.setFont(c.getFont());
497         Rectangle innerArea    = SwingUtilities.calculateInnerArea(c, null);
498         Rectangle cancelBounds = getCancelButtonBounds();
499         // FIXME Do better baseline calculation than just subtracting 1.
500         context.getStyle().getGraphicsUtils(context).paintText(context, g, getPlaceholderText(g, innerArea.width + cancelBounds.width),
501                                                                innerArea.x,
502                                                                innerArea.y - 1, -1);
503     }

505     /**
506      * Get the placeholder text, clipped if necessary.
507      *
508      * @param  g              fm the font metrics to compute size with.
509      * @param  availTextWidth the available space to display the title in.
510      *
511      * @return the text, clipped to fit the available space.
512      */
513     private String getPlaceholderText(Graphics g, int availTextWidth) {
514         JTextComponent c  = getComponent();
515         FontMetrics    fm = SwingUtilities2.getFontMetrics(c, g);
516 
517         return SwingUtilities2.clipStringIfNecessary(c, fm, placeholderText, availTextWidth);
518     }

541 
542         painter.paintSearchButtonForeground(subcontext, g, bounds.x, bounds.y, bounds.width, bounds.height);
543 
544         subcontext.dispose();
545     }

710     /**
711      * @see javax.swing.plaf.basic.BasicTextUI#getPreferredSize(javax.swing.JComponent)
712      */
713     public Dimension getPreferredSize(JComponent c) {
714         // The following code has been derived from BasicTextUI.
715         Document  doc      = ((JTextComponent) c).getDocument();
716         Insets    i        = c.getInsets();
717         Dimension d        = c.getSize();
718         View      rootView = getRootView((JTextComponent) c);
719 
720         if (doc instanceof AbstractDocument) {
721             ((AbstractDocument) doc).readLock();
722         }
723 
724         try {
725             if ((d.width > (i.left + i.right)) && (d.height > (i.top + i.bottom))) {
726                 rootView.setSize(d.width - i.left - i.right, d.height - i.top - i.bottom);
727             } else if (d.width == 0 && d.height == 0) {
728                 // Probably haven't been layed out yet, force some sort of
729                 // initial sizing.
730                 rootView.setSize(Integer.MAX_VALUE, Integer.MAX_VALUE);
731             }
732 
733             d.width  = (int) Math.min((long) rootView.getPreferredSpan(View.X_AXIS) + (long) i.left + (long) i.right, Integer.MAX_VALUE);
734             d.height = (int) Math.min((long) rootView.getPreferredSpan(View.Y_AXIS) + (long) i.top + (long) i.bottom, Integer.MAX_VALUE);
735         } finally {
736             if (doc instanceof AbstractDocument) {
737                 ((AbstractDocument) doc).readUnlock();
738             }
739         }
740 
741         // Fix: The preferred width is always two pixels too small on a Mac.
742         d.width += 2;
743 
744         // We'd like our heights to be odd by default.
745         if ((d.height & 1) == 0) {
746             d.height--;
747         }
748 
749         return d;
750     }
