165     private void updateStyle(JComboBox comboBox) {
166         SeaGlassStyle oldStyle = style;
167         SeaGlassContext context = getContext(comboBox, ENABLED);
168 
169         style = (SeaGlassStyle) SeaGlassLookAndFeel.updateStyle(context, this);
170         if (style != oldStyle) {
171             popupInsets = (Insets) style.get(context, "ComboBox.popupInsets");
172             useListColors = style.getBoolean(context, "ComboBox.rendererUseListColors", true);
173             buttonWhenNotEditable = style.getBoolean(context, "ComboBox.buttonWhenNotEditable", false);
174             pressedWhenPopupVisible = style.getBoolean(context, "ComboBox.pressedWhenPopupVisible", false);
175             if (oldStyle != null) {
176                 uninstallKeyboardActions();
177                 installKeyboardActions();
178             }
179             forceOpaque = style.getBoolean(context, "ComboBox.forceOpaque", false);
180         }
181         context.dispose();
182     }

228     private int getComponentState(JComponent c) {
229         // currently we have a broken situation where if a developer
230         // takes the border from a JComboBox and sets it on a JTextField
231         // then the codepath will eventually lead back to this method
232         // but pass in a JTextField instead of JComboBox! In case this
233         // happens, we just return the normal synth state for the component
234         // instead of doing anything special
235         if (!(c instanceof JComboBox)) return SeaGlassLookAndFeel.getComponentState(c);
236 
237         JComboBox box = (JComboBox) c;
238         if (shouldActLikeButton()) {
239             int state = ENABLED;
240             if ((!c.isEnabled())) {
241                 state = DISABLED;
242             }
243             if (buttonHandler.isPressed()) {
244                 state |= PRESSED;
245             }
246             if (buttonHandler.isRollover()) {
247                 state |= MOUSE_OVER;
248             }
249             if (box.isFocusOwner()) {
250                 state |= FOCUSED;
251             }
252             return state;
253         } else {
254             // for editable combos the editor component has the focus not the
255             // combo box its self, so we should make the combo paint focused
256             // when its editor has focus
257             int basicState = SeaGlassLookAndFeel.getComponentState(c);
258             if (box.isEditable() && box.getEditor().getEditorComponent().isFocusOwner()) {
259                 basicState |= FOCUSED;
260             }
261             return basicState;
262         }
263     }

333     /**
334      * Paints the currently selected item.
335      */
336     @Override
337     public void paintCurrentValue(Graphics g, Rectangle bounds, boolean hasFocus) {
338         ListCellRenderer renderer = comboBox.getRenderer();
339         Component c;
340 
341         c = renderer.getListCellRendererComponent(listBox, comboBox.getSelectedItem(), -1, false, false);
342 
343         // Fix for 4238829: should lay out the JPanel.
344         boolean shouldValidate = false;
345         if (c instanceof JPanel) {
346             shouldValidate = true;
347         }
348 
349         if (c instanceof UIResource) {
350             c.setName("ComboBox.renderer");
351         }
352 
353         boolean force = forceOpaque && c instanceof JComponent;
354         if (force) {
355             ((JComponent) c).setOpaque(false);
356         }
357 
358         int x = bounds.x, y = bounds.y, w = bounds.width, h = bounds.height;
359         if (padding != null) {
360             x = bounds.x + padding.left;
361             y = bounds.y + padding.top;
362             w = bounds.width - (padding.left + padding.right);
363             h = bounds.height - (padding.top + padding.bottom);
364         }
365 
366         currentValuePane.paintComponent(g, c, comboBox, x, y, w, h, shouldValidate);
367 
368         if (force) {
369             ((JComponent) c).setOpaque(true);
370         }
371 
372     }

418     /**
419      * From BasicComboBoxRenderer v 1.18.
420      * 
421      * Be aware that SynthFileChooserUIImpl relies on the fact that the default
422      * renderer installed on a Synth combo box is a JLabel. If this is changed,
423      * then an assert will fail in SynthFileChooserUIImpl
424      */
425     private class SynthComboBoxRenderer extends JLabel implements ListCellRenderer, UIResource {
426         public SynthComboBoxRenderer() {
427             super();
428             setName("ComboBox.renderer");
429             setText(" ");
430         }
431 
432         public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
433             setName("ComboBox.listRenderer");
434             SeaGlassLookAndFeel.resetSelectedUI();
435             if (isSelected) {
436                 setBackground(list.getSelectionBackground());
437                 setForeground(list.getSelectionForeground());
438                 if (!useListColors) {
439                     SeaGlassLookAndFeel.setSelectedUI((SeaGlassLabelUI) SeaGlassLookAndFeel.getUIOfType(getUI(), SeaGlassLabelUI.class),
440                         isSelected, cellHasFocus, list.isEnabled(), false);
441                 }
442             } else {
443                 setBackground(list.getBackground());
444                 setForeground(list.getForeground());
445             }
446 
447             setFont(list.getFont());
448 
449             if (value instanceof Icon) {
450                 setIcon((Icon) value);
451                 setText("");
452             } else {
453                 String text = (value == null) ? " " : value.toString();
454 
455                 if ("".equals(text)) {
456                     text = " ";
457                 }
458                 setText(text);
459             }
460 
461             // The renderer component should inherit the enabled and
462             // orientation state of its parent combobox. This is
463             // especially needed for GTK comboboxes, where the
464             // ListCellRenderer's state determines the visual state
465             // of the combobox.
466             if (comboBox != null) {
467                 setEnabled(comboBox.isEnabled());
468                 setComponentOrientation(comboBox.getComponentOrientation());
469             }
470 
471             return this;
472         }
473 
474         @Override
475         public void paint(Graphics g) {
476             super.paint(g);
477             SeaGlassLookAndFeel.resetSelectedUI();
478         }

482      * From BasicCombBoxEditor v 1.24.
483      */
484     private static class SynthComboBoxEditor implements ComboBoxEditor, UIResource {
485         protected JTextField editor;
486         private Object       oldValue;
487 
488         public SynthComboBoxEditor() {
489             editor = new JTextField("", 9);
490             editor.setName("ComboBox.textField");
491         }
492 
493         public Component getEditorComponent() {
494             return editor;
495         }

497         /**
498          * Sets the item that should be edited.
499          * 
500          * @param anObject
501          *            the displayed value of the editor
502          */
503         public void setItem(Object anObject) {
504             String text;
505 
506             if (anObject != null) {
507                 text = anObject.toString();
508                 oldValue = anObject;
509             } else {
510                 text = "";
511             }
512             // workaround for 4530952
513             if (!text.equals(editor.getText())) {
514                 editor.setText(text);
515             }
516         }
517 
518         public Object getItem() {
519             Object newValue = editor.getText();
520 
521             if (oldValue != null && !(oldValue instanceof String)) {
522                 // The original value is not a string. Should return the value
523                 // in it's
524                 // original type.
525                 if (newValue.equals(oldValue.toString())) {
526                     return oldValue;
527                 } else {
528                     // Must take the value from the editor and get the value and
529                     // cast it to the new type.
530                     Class cls = oldValue.getClass();
531                     try {
532                         Method method = cls.getMethod("valueOf", new Class[] { String.class });
533                         newValue = method.invoke(oldValue, new Object[] { editor.getText() });
534                     } catch (Exception ex) {
535                         // Fail silently and return the newValue (a String
536                         // object)
537                     }
538                 }
539             }
540             return newValue;
541         }

546         }

557     /**
558      * Handles all the logic for treating the combo as a button when it is not
559      * editable, and when shouldActLikeButton() is true. This class is a special
560      * ButtonModel, and installed on the arrowButton when appropriate. It also
561      * is installed as a mouse listener and mouse motion listener on the combo
562      * box. In this way, the state between the button and combo are in sync.
563      * Whenever one is "over" both are. Whenever one is pressed, both are.
564      */
565     private final class ButtonHandler extends DefaultButtonModel implements MouseListener, PopupMenuListener {
566         /**
567          * Indicates that the mouse is over the combo or the arrow button. This
568          * field only has meaning if buttonWhenNotEnabled is true.
569          */
570         private boolean over;
571         /**
572          * Indicates that the combo or arrow button has been pressed. This field
573          * only has meaning if buttonWhenNotEnabled is true.
574          */
575         private boolean pressed;
576 
577         // ------------------------------------------------------------------
578         // State Methods
579         // ------------------------------------------------------------------
580 
581         /**
582          * <p>
583          * Updates the internal "pressed" state. If shouldActLikeButton() is
584          * true, and if this method call will change the internal state, then
585          * the combo and button will be repainted.
586          * </p>
587          * 
588          * <p>
589          * Note that this method is called either when a press event occurs on
590          * the combo box, or on the arrow button.
591          * </p>
592          */
593         private void updatePressed(boolean p) {
594             this.pressed = p && isEnabled();
595             if (shouldActLikeButton()) {
596                 comboBox.repaint();
597             }
598         }
599 
600         /**
601          * <p>
602          * Updates the internal "over" state. If shouldActLikeButton() is true,
603          * and if this method call will change the internal state, then the
604          * combo and button will be repainted.
605          * </p>
606          * 
607          * <p>
608          * Note that this method is called either when a mouseover/mouseoff
609          * event occurs on the combo box, or on the arrow button.
610          * </p>
611          */
612         private void updateOver(boolean o) {
613             boolean old = isRollover();
614             this.over = o && isEnabled();
615             boolean newo = isRollover();
616             if (shouldActLikeButton() && old != newo) {
617                 comboBox.repaint();
618             }
619         }
620 
621         // ------------------------------------------------------------------
622         // DefaultButtonModel Methods
623         // ------------------------------------------------------------------
624 
625         /**
626          * Indicates if the button is pressed.
627          * 
628          * @return <code>true</code> if the button is pressed
629          * 
630          *         Ensures that isPressed() will return true if the combo is
631          *         pressed, or the arrowButton is pressed, <em>or</em> if the
632          *         combo popup is visible. This is the case because a combo box
633          *         looks pressed when the popup is visible, and so should the
634          *         arrow button.
635          */
636         @Override
637         public boolean isPressed() {
638             boolean b = shouldActLikeButton() ? pressed : super.isPressed();
639             return b || (pressedWhenPopupVisible && comboBox.isPopupVisible());
640         }
641 
642         /**
643          * Indicates partial commitment towards triggering the button.
644          * 
645          * @return <code>true</code> if the button is armed, and ready to be
646          *         triggered
647          * @see javax.swing.ButtonModel#setArmed
648          * 
649          *      Ensures that the armed state is in sync with the pressed state
650          *      if shouldActLikeButton is true. Without this method, the arrow
651          *      button will not look pressed when the popup is open, regardless
652          *      of the result of isPressed() alone.
653          */
654         @Override
655         public boolean isArmed() {
656             boolean b = shouldActLikeButton() || (pressedWhenPopupVisible && comboBox.isPopupVisible());
657             return b ? isPressed() : super.isArmed();
658         }
659 
660         /**
661          * Indicates that the mouse is over the button.
662          * 
663          * @return <code>true</code> if the mouse is over the button
664          * 
665          *         Ensures that isRollover() will return true if the combo is
666          *         rolled over, or the arrowButton is rolled over.
667          */
668         @Override
669         public boolean isRollover() {
670             return shouldActLikeButton() ? over : super.isRollover();
671         }
672 
673         /**
674          * Sets the button to pressed or unpressed.
675          * 
676          * @param b
677          *            whether or not the button should be pressed
678          * @see javax.swing.ButtonModel#isPressed
679          * 
680          *      Forwards pressed states to the internal "pressed" field
681          */
682         @Override
683         public void setPressed(boolean b) {
684             super.setPressed(b);
685             updatePressed(b);
686         }

688         /**
689          * Sets or clears the button's rollover state

692          *            whether or not the button is in the rollover state

694          * 
695          *      Forwards rollover states to the internal "over" field
696          */
697         @Override
698         public void setRollover(boolean b) {
699             super.setRollover(b);
700             updateOver(b);
701         }
