169                     padding.bottom *= 1.15;
170                     padding.top *= 1.15;
171                 } else if (SeaGlassStyle.SMALL_KEY.equals(scaleKey)) {
172                     padding.left *= 0.857;
173                     padding.right *= 0.857;
174                     padding.bottom *= 0.857;

178                     padding.right *= 0.784;

188     private void updateStyle(JComboBox comboBox) {
189         SeaGlassStyle oldStyle = style;
190         SeaGlassContext context = getContext(comboBox, ENABLED);
191 
192         style = (SeaGlassStyle) SeaGlassLookAndFeel.updateStyle(context, this);
193         if (style != oldStyle) {
194             popupInsets = (Insets) style.get(context, "ComboBox.popupInsets");
195             useListColors = style.getBoolean(context, "ComboBox.rendererUseListColors", true);
196             buttonWhenNotEditable = style.getBoolean(context, "ComboBox.buttonWhenNotEditable", false);
197             pressedWhenPopupVisible = style.getBoolean(context, "ComboBox.pressedWhenPopupVisible", false);
198             
199             if (oldStyle != null) {
200                 uninstallKeyboardActions();
201                 installKeyboardActions();
202             }
203             forceOpaque = style.getBoolean(context, "ComboBox.forceOpaque", false);
204         }
205         context.dispose();
206     }

252     private int getComponentState(JComponent c) {
253         // currently we have a broken situation where if a developer
254         // takes the border from a JComboBox and sets it on a JTextField
255         // then the codepath will eventually lead back to this method
256         // but pass in a JTextField instead of JComboBox! In case this
257         // happens, we just return the normal synth state for the component
258         // instead of doing anything special
259         if (!(c instanceof JComboBox)) return SeaGlassLookAndFeel.getComponentState(c);
260 
261         JComboBox box = (JComboBox) c;
262         if (shouldActLikeButton()) {
263             int state = ENABLED;
264             if ((!c.isEnabled())) {
265                 state = DISABLED;
266             }
267             if (buttonHandler.isPressed()) {
268                 state |= PRESSED;
269             }
270             if (buttonHandler.isRollover()) {
271                 state |= MOUSE_OVER;
272             }
273             if (box.isFocusOwner()) {
274                 state |= FOCUSED;
275             }
276             return state;
277         } else {
278             // for editable combos the editor component has the focus not the
279             // combo box its self, so we should make the combo paint focused
280             // when its editor has focus
281             int basicState = SeaGlassLookAndFeel.getComponentState(c);
282             if (box.isEditable() && box.getEditor().getEditorComponent().isFocusOwner()) {
283                 basicState |= FOCUSED;
284             }
285             return basicState;
286         }
287     }

357     /**
358      * Paints the currently selected item.
359      */
360     @Override
361     public void paintCurrentValue(Graphics g, Rectangle bounds, boolean hasFocus) {
362         ListCellRenderer renderer = comboBox.getRenderer();
363         Component c;
364 
365         c = renderer.getListCellRendererComponent(listBox, comboBox.getSelectedItem(), -1, false, false);
366 
367         // Fix for 4238829: should lay out the JPanel.
368         boolean shouldValidate = false;
369         if (c instanceof JPanel) {
370             shouldValidate = true;
371         }
372 
373         if (c instanceof UIResource) {
374             c.setName("ComboBox.renderer");
375         }
376 
377         boolean force = forceOpaque && c instanceof JComponent;
378         if (force) {
379             ((JComponent) c).setOpaque(false);
380         }
381 
382         int x = bounds.x, y = bounds.y, w = bounds.width, h = bounds.height;
383         if (padding != null) {
384             x = bounds.x + padding.left;
385             y = bounds.y + padding.top;
386             w = bounds.width - (padding.left + padding.right);
387             h = bounds.height - (padding.top + padding.bottom);
388         }
389 
390         currentValuePane.paintComponent(g, c, comboBox, x, y, w, h, shouldValidate);
391 
392         if (force) {
393             ((JComponent) c).setOpaque(true);
394         }
395 
396     }

442     /**
443      * From BasicComboBoxRenderer v 1.18.
444      * 
445      * Be aware that SynthFileChooserUIImpl relies on the fact that the default
446      * renderer installed on a Synth combo box is a JLabel. If this is changed,
447      * then an assert will fail in SynthFileChooserUIImpl
448      */
449     private class SynthComboBoxRenderer extends JLabel implements ListCellRenderer, UIResource {
450         public SynthComboBoxRenderer() {
451             super();
452             setName("ComboBox.renderer");
453             setText(" ");
454         }
455 
456         public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
457             setName("ComboBox.listRenderer");
458             SeaGlassLookAndFeel.resetSelectedUI();
459             if (isSelected) {
460                 setBackground(list.getSelectionBackground());
461                 setForeground(list.getSelectionForeground());
462                 if (!useListColors) {
463                     SeaGlassLookAndFeel.setSelectedUI((SeaGlassLabelUI) SeaGlassLookAndFeel.getUIOfType(getUI(), SeaGlassLabelUI.class),
464                         isSelected, cellHasFocus, list.isEnabled(), false);
465                 }
466             } else {
467                 setBackground(list.getBackground());
468                 setForeground(list.getForeground());
469             }
470 
471             setFont(list.getFont());
472 
473             if (value instanceof Icon) {
474                 setIcon((Icon) value);
475                 setText("");
476             } else {
477                 String text = (value == null) ? " " : value.toString();
478 
479                 if ("".equals(text)) {
480                     text = " ";
481                 }
482                 setText(text);
483             }
484 
485             // The renderer component should inherit the enabled and
486             // orientation state of its parent combobox. This is
487             // especially needed for GTK comboboxes, where the
488             // ListCellRenderer's state determines the visual state
489             // of the combobox.
490             if (comboBox != null) {
491                 setEnabled(comboBox.isEnabled());
492                 setComponentOrientation(comboBox.getComponentOrientation());
493             }
494 
495             return this;
496         }
497 
498         @Override
499         public void paint(Graphics g) {
500             super.paint(g);
501             SeaGlassLookAndFeel.resetSelectedUI();
502         }

510         private Object       oldValue;

512         public SynthComboBoxEditor() {
513             editor = new JTextField("", 9);
514             editor.setName("ComboBox.textField");
515         }

521         /**
522          * Sets the item that should be edited.
523          * 
524          * @param anObject
525          *            the displayed value of the editor
526          */
527         public void setItem(Object anObject) {
528             String text;
529 
530             if (anObject != null) {
531                 text = anObject.toString();
532                 oldValue = anObject;
533             } else {
534                 text = "";
535             }
536             // workaround for 4530952
537             if (!text.equals(editor.getText())) {
538                 editor.setText(text);
539             }
540         }
541 
542         public Object getItem() {
543             Object newValue = editor.getText();
544 
545             if (oldValue != null && !(oldValue instanceof String)) {
546                 // The original value is not a string. Should return the value
547                 // in it's
548                 // original type.
549                 if (newValue.equals(oldValue.toString())) {
550                     return oldValue;
551                 } else {
552                     // Must take the value from the editor and get the value and
553                     // cast it to the new type.
554                     Class cls = oldValue.getClass();
555                     try {
556                         Method method = cls.getMethod("valueOf", new Class[] { String.class });
557                         newValue = method.invoke(oldValue, new Object[] { editor.getText() });
558                     } catch (Exception ex) {
559                         // Fail silently and return the newValue (a String
560                         // object)
561                     }
562                 }
563             }
564             return newValue;
565         }

569             editor.requestFocus();
570         }

581     /**
582      * Handles all the logic for treating the combo as a button when it is not
583      * editable, and when shouldActLikeButton() is true. This class is a special
584      * ButtonModel, and installed on the arrowButton when appropriate. It also
585      * is installed as a mouse listener and mouse motion listener on the combo
586      * box. In this way, the state between the button and combo are in sync.
587      * Whenever one is "over" both are. Whenever one is pressed, both are.
588      */
589     private final class ButtonHandler extends DefaultButtonModel implements MouseListener, PopupMenuListener {
590         /**
591          * Indicates that the mouse is over the combo or the arrow button. This
592          * field only has meaning if buttonWhenNotEnabled is true.
593          */
594         private boolean over;
595         /**
596          * Indicates that the combo or arrow button has been pressed. This field
597          * only has meaning if buttonWhenNotEnabled is true.
598          */
599         private boolean pressed;
600 
601         // ------------------------------------------------------------------
602         // State Methods
603         // ------------------------------------------------------------------
604 
605         /**
606          * <p>
607          * Updates the internal "pressed" state. If shouldActLikeButton() is
608          * true, and if this method call will change the internal state, then
609          * the combo and button will be repainted.
610          * </p>
611          * 
612          * <p>
613          * Note that this method is called either when a press event occurs on
614          * the combo box, or on the arrow button.
615          * </p>
616          */
617         private void updatePressed(boolean p) {
618             this.pressed = p && isEnabled();
619             if (shouldActLikeButton()) {
620                 comboBox.repaint();
621             }
622         }
623 
624         /**
625          * <p>
626          * Updates the internal "over" state. If shouldActLikeButton() is true,
627          * and if this method call will change the internal state, then the
628          * combo and button will be repainted.
629          * </p>
630          * 
631          * <p>
632          * Note that this method is called either when a mouseover/mouseoff
633          * event occurs on the combo box, or on the arrow button.
634          * </p>
635          */
636         private void updateOver(boolean o) {
637             boolean old = isRollover();
638             this.over = o && isEnabled();
639             boolean newo = isRollover();
640             if (shouldActLikeButton() && old != newo) {
641                 comboBox.repaint();
642             }
643         }
644 
645         // ------------------------------------------------------------------
646         // DefaultButtonModel Methods
647         // ------------------------------------------------------------------
648 
649         /**
650          * Indicates if the button is pressed.
651          * 
652          * @return <code>true</code> if the button is pressed
653          * 
654          *         Ensures that isPressed() will return true if the combo is
655          *         pressed, or the arrowButton is pressed, <em>or</em> if the
656          *         combo popup is visible. This is the case because a combo box
657          *         looks pressed when the popup is visible, and so should the
658          *         arrow button.
659          */
660         @Override
661         public boolean isPressed() {
662             boolean b = shouldActLikeButton() ? pressed : super.isPressed();
663             return b || (pressedWhenPopupVisible && comboBox.isPopupVisible());
664         }
665 
666         /**
667          * Indicates partial commitment towards triggering the button.
668          * 
669          * @return <code>true</code> if the button is armed, and ready to be
670          *         triggered
671          * @see javax.swing.ButtonModel#setArmed
672          * 
673          *      Ensures that the armed state is in sync with the pressed state
674          *      if shouldActLikeButton is true. Without this method, the arrow
675          *      button will not look pressed when the popup is open, regardless
676          *      of the result of isPressed() alone.
677          */
678         @Override
679         public boolean isArmed() {
680             boolean b = shouldActLikeButton() || (pressedWhenPopupVisible && comboBox.isPopupVisible());
681             return b ? isPressed() : super.isArmed();
682         }
683 
684         /**
685          * Indicates that the mouse is over the button.
686          * 
687          * @return <code>true</code> if the mouse is over the button
688          * 
689          *         Ensures that isRollover() will return true if the combo is
690          *         rolled over, or the arrowButton is rolled over.
691          */
692         @Override
693         public boolean isRollover() {
694             return shouldActLikeButton() ? over : super.isRollover();
695         }
696 
697         /**
698          * Sets the button to pressed or unpressed.
699          * 
700          * @param b
701          *            whether or not the button should be pressed
702          * @see javax.swing.ButtonModel#isPressed
703          * 
704          *      Forwards pressed states to the internal "pressed" field
705          */
706         @Override
707         public void setPressed(boolean b) {
708             super.setPressed(b);
709             updatePressed(b);
710         }
711 
712         /**
713          * Sets or clears the button's rollover state

787             this.comboBox = comboBox;
788             editor = comboBox.getEditor();
789             if (editor != null) {

802             }
803         }
