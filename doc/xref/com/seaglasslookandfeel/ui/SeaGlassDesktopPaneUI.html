74          super.installListeners();
75          desktop.addPropertyChangeListener(this);

84      protected void installDefaults() {
85          updateStyle(desktop);
86  
87          if (UIManager.getBoolean("InternalFrame.useTaskBar")) {
88              taskBar = new TaskBar();
89  
90              for (Component comp : desktop.getComponents()) {
91                  JInternalFrame.JDesktopIcon desktopIcon;
92  
93                  if (comp instanceof JInternalFrame.JDesktopIcon) {
94                      desktopIcon = (JInternalFrame.JDesktopIcon) comp;
95                  } else if (comp instanceof JInternalFrame) {
96                      desktopIcon = ((JInternalFrame) comp).getDesktopIcon();
97                  } else {
98                      continue;
99                  }
100                 // Move desktopIcon from desktop to taskBar
101                 if (desktopIcon.getParent() == desktop) {
102                     desktop.remove(desktopIcon);
103                 }
104                 if (desktopIcon.getParent() != taskBar) {
105                     taskBar.add(desktopIcon);
106                     desktopIcon.getInternalFrame().addComponentListener(taskBar);
107                 }
108             }
109             taskBar.setBackground(desktop.getBackground());
110             desktop.add(taskBar, new Integer(JLayeredPane.PALETTE_LAYER.intValue() + 1));
111             if (desktop.isShowing()) {
112                 taskBar.adjustSize();
113             }
114         }

139

141         context.dispose();
142         style = null;
143 
144         if (taskBar != null) {

146                 JInternalFrame.JDesktopIcon desktopIcon = (JInternalFrame.JDesktopIcon) comp;
147                 taskBar.remove(desktopIcon);

183     static class TaskBar extends JPanel implements ComponentListener, ContainerListener {
184         TaskBar() {
185             setOpaque(true);
186             setLayout(new FlowLayout(FlowLayout.LEFT, 0, 0) {
187                 public void layoutContainer(Container target) {
188                     // First shrink buttons to fit
189                     Component[] comps = target.getComponents();
190                     int n = comps.length;
191                     if (n > 0) {
192                         // Start with the largest preferred width
193                         int prefWidth = 0;
194                         for (Component c : comps) {
195                             c.setPreferredSize(null);
196                             Dimension prefSize = c.getPreferredSize();
197                             if (prefSize.width > prefWidth) {
198                                 prefWidth = prefSize.width;
199                             }
200                         }
201                         // Shrink equally to fit if needed
202                         Insets insets = target.getInsets();
203                         int tw = target.getWidth() - insets.left - insets.right;
204                         int w = Math.min(prefWidth, Math.max(10, tw / n));
205                         for (Component c : comps) {
206                             Dimension prefSize = c.getPreferredSize();
207                             c.setPreferredSize(new Dimension(w, prefSize.height));
208                         }
209                     }
210                     super.layoutContainer(target);
211                 }
212             });
213 
214             // PENDING: This should be handled by the painter
215             setBorder(new BevelBorder(BevelBorder.RAISED) {
216                 protected void paintRaisedBevel(Component c, Graphics g, int x, int y, int w, int h) {
217                     Color oldColor = g.getColor();
218                     g.translate(x, y);
219                     g.setColor(getHighlightOuterColor(c));
220                     g.drawLine(0, 0, 0, h - 2);
221                     g.drawLine(1, 0, w - 2, 0);
222                     g.setColor(getShadowOuterColor(c));
223                     g.drawLine(0, h - 1, w - 1, h - 1);
224                     g.drawLine(w - 1, 0, w - 1, h - 2);
225                     g.translate(-x, -y);
226                     g.setColor(oldColor);
227                 }
228             });
229         }

231         void adjustSize() {
232             JDesktopPane desktop = (JDesktopPane) getParent();
233             if (desktop != null) {
234                 int height = getPreferredSize().height;
235                 Insets insets = getInsets();
236                 if (height == insets.top + insets.bottom) {
237                     if (getHeight() <= height) {
238                         // Initial size, because we have no buttons yet
239                         height += 21;
240                     } else {
241                         // We already have a good height
242                         height = getHeight();
243                     }
244                 }
245                 setBounds(0, desktop.getHeight() - height, desktop.getWidth(), height);
246                 revalidate();
247                 repaint();
248             }
249         }

277         public void componentAdded(ContainerEvent e) {
278             if (e.getChild() instanceof JInternalFrame) {
279                 JInternalFrame f = (JInternalFrame) e.getChild();
280                 JInternalFrame.JDesktopIcon desktopIcon = f.getDesktopIcon();
281                 for (Component comp : getComponents()) {
282                     if (comp == desktopIcon) {
283                         // We have it already
284                         return;
285                     }
286                 }
287                 add(desktopIcon);
288                 f.addComponentListener(this);
289                 if (getComponentCount() == 1) {
290                     adjustSize();
291                 }
292             }
293         }
294 
295         public void componentRemoved(ContainerEvent e) {
296             if (e.getChild() instanceof JInternalFrame) {
297                 JInternalFrame f = (JInternalFrame) e.getChild();
298                 if (!f.isIcon()) {
299                     // Frame was removed without using setClosed(true)
300                     remove(f.getDesktopIcon());
301                     f.removeComponentListener(this);
302                     revalidate();
303                     repaint();
304                 }
305             }
306         }

311         public void maximizeFrame(JInternalFrame f) {
312             if (f.isIcon()) {
313                 try {
314                     f.setIcon(false);
315                 } catch (PropertyVetoException e2) {
316                 }
317             } else {
318                 f.setNormalBounds(f.getBounds());
319                 Component desktop = f.getParent();
320                 setBoundsForFrame(f, 0, 0, desktop.getWidth(), desktop.getHeight() - taskBar.getHeight());
321             }
322 
323             try {
324                 f.setSelected(true);
325             } catch (PropertyVetoException e2) {
326             }
327         }
328 
329         public void iconifyFrame(JInternalFrame f) {
330             Container c = f.getParent();
331             boolean findNext = f.isSelected();
332 
333             if (c == null) {
334                 return;
335             }
336 
337             if (!f.isMaximum()) {
338                 f.setNormalBounds(f.getBounds());
339             }
340             c.remove(f);
341             c.repaint(f.getX(), f.getY(), f.getWidth(), f.getHeight());
342             try {
343                 f.setSelected(false);
344             } catch (PropertyVetoException e2) {
345             }
346 
347             // Get topmost of the remaining frames
348             if (findNext) {
349                 for (Component comp : c.getComponents()) {
350                     if (comp instanceof JInternalFrame) {
351                         try {
352                             ((JInternalFrame) comp).setSelected(true);
353                         } catch (PropertyVetoException e2) {
354                         }
355                         ((JInternalFrame) comp).moveToFront();
356                         return;
357                     }
358                 }
359             }
360         }
361 
362         public void deiconifyFrame(JInternalFrame f) {
363             JInternalFrame.JDesktopIcon desktopIcon = f.getDesktopIcon();
364             Container c = desktopIcon.getParent();
365             if (c != null) {
366                 c = c.getParent();
367                 if (c != null) {
368                     c.add(f);
369                     if (f.isMaximum()) {
370                         int w = c.getWidth();
371                         int h = c.getHeight() - taskBar.getHeight();
372                         if (f.getWidth() != w || f.getHeight() != h) {
373                             setBoundsForFrame(f, 0, 0, w, h);
374                         }
375                     }
376                     if (f.isSelected()) {
377                         f.moveToFront();
378                     } else {
379                         try {
380                             f.setSelected(true);
381                         } catch (PropertyVetoException e2) {
382                         }
383                     }
384                 }
385             }
386         }
