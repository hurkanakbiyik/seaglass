432     @Override
433     public UIDefaults getDefaults() {
434         if (uiDefaults == null) {
435             uiDefaults =  new UIWrapper(super.getDefaults());
436 
437             // Install Keybindings for the operating system.
438             if (PlatformUtils.isWindows()) {
439                 WindowsKeybindings.installKeybindings(uiDefaults);
440             } else if (PlatformUtils.isMac()) {
441                 MacKeybindings.installKeybindings(uiDefaults);
442             } else {
443                 GTKKeybindings.installKeybindings(uiDefaults);
444             }
445 
446             // Set the default font.
447             defineDefaultFont(uiDefaults);
448 
449             // Override some of the Synth UI delegates with copied and modified
450             // versions.
451             useOurUIs();
452 
453             defineBaseColors(uiDefaults);
454             defineDefaultBorders(uiDefaults);
455             defineArrowButtons(uiDefaults);
456             defineButtons(uiDefaults);
457             defineComboBoxes(uiDefaults);
458             defineDesktopPanes(uiDefaults);
459             defineInternalFrames(uiDefaults);
460             defineInternalFrameMenuButtons(uiDefaults);
461             defineInternalFrameCloseButtons(uiDefaults);
462             defineInternalFrameIconifyButtons(uiDefaults);
463             defineInternalFrameMaximizeButton(uiDefaults);
464             defineLists(uiDefaults);
465             defineMenus(uiDefaults);
466             definePanels(uiDefaults);
467             definePopups(uiDefaults);
468             defineProgressBars(uiDefaults);
469             defineRootPanes(uiDefaults);
470             defineSeparators(uiDefaults);
471             defineSpinners(uiDefaults);
472             defineScrollBars(uiDefaults);
473             defineScrollPane(uiDefaults);
474             defineSliders(uiDefaults);
475             defineSplitPanes(uiDefaults);
476             defineTabbedPanes(uiDefaults);
477             defineTables(uiDefaults);
478             defineTextControls(uiDefaults);
479             defineToolBars(uiDefaults);
480             defineTrees(uiDefaults);
481             defineToolTips(uiDefaults);
482             defineOptionPane(uiDefaults);
483             defineFileChooser(uiDefaults);
484 
485             if (!PlatformUtils.isMac()) {
486                 uiDefaults.put("MenuBar[Enabled].backgroundPainter", null);
487                 uiDefaults.put("MenuBar[Enabled].borderPainter", null);
488 
489                 // If we're not on a Mac, draw our own title bar.
490                 JFrame.setDefaultLookAndFeelDecorated(true);
491                 JDialog.setDefaultLookAndFeelDecorated(true);
492             } else {
493 
494                 // If we're on a Mac, use the screen menu bar.
495                 System.setProperty("apple.laf.useScreenMenuBar", "true");
496 
497                 // If we're on a Mac, use Aqua for some things.
498                 defineAquaSettings(uiDefaults);
499             }
500         }
501 
502         return uiDefaults;
503     }

2640     /**
2641      * Locate the style associated with the given region and component. This is
2642      * called from SeaGlassLookAndFeel in the SynthStyleFactory implementation.
2643      *
2644      * <p>Lookup occurs as follows:<br/>
2645      * Check the map of styles <code>styleMap</code>. If the map contains no
2646      * styles at all, then simply return the defaultStyle. If the map contains
2647      * styles, then iterate over all of the styles for the Region <code>r</code>
2648      * looking for the best match, based on prefix. If a match was made, then
2649      * return that SynthStyle. Otherwise, return the defaultStyle.</p>
2650      *
2651      * @param  c The component associated with this region. For example, if the
2652      *           Region is Region.Button then the component will be a JButton.
2653      *           If the Region is a subregion, such as ScrollBarThumb, then the
2654      *           associated component will be the component that subregion
2655      *           belongs to, such as JScrollBar. The JComponent may be named. It
2656      *           may not be null.
2657      * @param  r The region we are looking for a style for. May not be null.
2658      *
2659      * @return the style associated with the given region and component.
2660      */
2661     SynthStyle getSeaGlassStyle(JComponent c, Region r) {
2662         // validate method arguments
2663         if (c == null || r == null) {
2664             throw new IllegalArgumentException("Neither comp nor r may be null");
2665         }
2666 
2667         // if there are no lazy styles registered for the region r, then return
2668         // the default style
2669         List<LazyStyle> styles = styleMap.get(r);
2670 
2671         if (styles == null || styles.size() == 0) {
2672             return getDefaultStyle();
2673         }
2674 
2675         // Look for the best SynthStyle for this component/region pair.
2676         LazyStyle foundStyle = null;
2677 
2678         for (LazyStyle s : styles) {
2679 
2680             if (s.matches(c)) {
2681 
2682                 /*
2683                  * Replace the foundStyle if foundStyle is null, or if the new
2684                  * style "s" is more specific (ie, its path was longer), or if
2685                  * the foundStyle was "simple" and the new style was not (ie:
2686                  * the foundStyle was for something like Button and the new
2687                  * style was for something like "MyButton", hence, being more
2688                  * specific). In all cases, favor the most specific style found.
2689                  */
2690                 if (foundStyle == null || (foundStyle.parts.length < s.parts.length)
2691                         || (foundStyle.parts.length == s.parts.length && foundStyle.simple && !s.simple)) {
2692                     foundStyle = s;
2693                 }
2694             }
2695         }
2696 
2697         // return the style, if found, or the default style if not found
2698         return foundStyle == null ? getDefaultStyle() : foundStyle.getStyle(c);
2699     }
