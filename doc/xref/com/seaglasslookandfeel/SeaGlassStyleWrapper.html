45  /**
46   * A SynthStyle implementation used by SeaGlass. This just wraps a SynthStyle
47   * with a SeaGlass style in order to get at some SynthStyle methods that are
48   * package local.
49   *
50   * @see javax.swing.plaf.synth.SynthStyle

52  public final class SeaGlassStyleWrapper extends SeaGlassStyle {
53  
54      /** Shared SynthGraphics. */
55      private static final SynthGraphicsUtils SEAGLASS_GRAPHICS = new SeaGlassGraphicsUtils();
56  
57      /**
58       * The SynthPainter that will be returned from this SeaGlassStyle. The
59       * SynthPainter returned will be a SeaGlassSynthPainterImpl, which will in
60       * turn delegate back to this SeaGlassStyle for the proper Painter (not
61       * SynthPainter) to use for painting the foreground, background, or border.
62       */
63      private SynthPainter painter;
64  
65      private SynthStyle style;
66  
67      /**
68       * Create a new SeaGlassStyle. Only the prefix must be supplied. At the
69       * appropriate time, installDefaults will be called. At that point, all of
70       * the state information will be pulled from UIManager and stored locally
71       * within this style.
72       *
73       * @param style Something like Button or Slider.Thumb or
74       *              org.jdesktop.swingx.JXStatusBar or
75       *              ComboBox."ComboBox.arrowButton"
76       */
77      SeaGlassStyleWrapper(SynthStyle style) {
78          super(null, null);
79          this.style   = style;
80          this.painter = new SeaGlassSynthPainterImpl(this);
81      }

83      /**
84       * Returns the <code>SynthGraphicUtils</code> for the specified context.
85       *
86       * @param  context SynthContext identifying requester
87       *
88       * @return SynthGraphicsUtils
89       */
90      public SynthGraphicsUtils getGraphicsUtils(SynthContext context) {
91          return SEAGLASS_GRAPHICS;
92      }

94      /**
95       * Re-implements SynthStyle.installDefaults(SynthContext, SynthUI) because
96       * it's package local.
97       *
98       * @param context the context.
99       * @param ui      the UI delegate.
100      */
101     public void installDefaults(SeaGlassContext context, SynthUI ui) {
102         // Special case the Border as this will likely change when the LAF
103         // can have more control over this.
104         if (!context.isSubregion()) {
105             JComponent c      = context.getComponent();
106             Border     border = c.getBorder();
107 
108             if (border == null || border instanceof UIResource) {
109                 c.setBorder(new SeaGlassBorder(ui, getInsets(context, null)));
110             }
111         }
112 
113         installDefaults(context);
114     }

142     /**
143      * {@inheritDoc}
144      */
145     @Override
146     protected Font getFontForState(SynthContext ctx) {
147         Font f = (Font) get(ctx, "font");
148 
149         if (f == null)
150             f = UIManager.getFont("defaultFont");
151 
152         // Account for scale
153         // The key "JComponent.sizeVariant" is used to match Apple's LAF
154         String scaleKey = (String) ctx.getComponent().getClientProperty("JComponent.sizeVariant");
155 
156         if (scaleKey != null) {
157 
158             if (LARGE_KEY.equals(scaleKey)) {
159                 f = f.deriveFont(Math.round(f.getSize2D() * LARGE_SCALE));
160             } else if (SMALL_KEY.equals(scaleKey)) {
161                 f = f.deriveFont(Math.round(f.getSize2D() * SMALL_SCALE));
162             } else if (MINI_KEY.equals(scaleKey)) {
163                 f = f.deriveFont(Math.round(f.getSize2D() * MINI_SCALE));
164             }
165         }
166 
167         return f;
168 
169     }
