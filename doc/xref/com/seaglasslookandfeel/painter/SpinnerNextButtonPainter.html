34   * SpinnerNextButtonPainter implementation.

36  public final class SpinnerNextButtonPainter extends AbstractCommonColorsPainter {
37      public static enum Which {
38          BACKGROUND_DISABLED,
39          BACKGROUND_ENABLED,
40          BACKGROUND_FOCUSED,
41          BACKGROUND_PRESSED_FOCUSED,
42          BACKGROUND_PRESSED,
43          FOREGROUND_DISABLED,
44          FOREGROUND_ENABLED,
45          FOREGROUND_FOCUSED,
46          FOREGROUND_PRESSED_FOCUSED,
47          FOREGROUND_PRESSED,
48      }
49  
50      private Color             spinnerNextBorderBottomEnabled   = decodeColor("spinnerNextBorderBottomEnabled");
51      private Color             spinnerNextBorderBottomPressed   = decodeColor("spinnerNextBorderBottomPressed");
52      private Color             spinnerNextInteriorBottomEnabled = decodeColor("spinnerNextInteriorBottomEnabled");
53      private Color             spinnerNextInteriorBottomPressed = decodeColor("spinnerNextInteriorBottomPressed");
54  
55      private TwoColors         pressedButtonBorder              = getCommonBorderColors(CommonControlState.PRESSED);
56      private TwoColors         spinnerNextBorderEnabled         = new TwoColors(pressedButtonBorder.top, spinnerNextBorderBottomEnabled);
57      private TwoColors         spinnerNextBorderPressed         = new TwoColors(pressedButtonBorder.top, spinnerNextBorderBottomPressed);
58      private TwoColors         spinnerNextBorderDisabled        = disable(spinnerNextBorderEnabled);
59  
60      private FourColors        pressedButtonInterior            = getCommonInteriorColors(CommonControlState.PRESSED);
61      private FourColors        selectedButtonInterior           = getCommonInteriorColors(CommonControlState.SELECTED);
62      private TwoColors         spinnerNextInteriorEnabled       = new TwoColors(selectedButtonInterior.top, spinnerNextInteriorBottomEnabled);
63      private TwoColors         spinnerNextInteriorPressed       = new TwoColors(pressedButtonInterior.top, spinnerNextInteriorBottomPressed);
64      private TwoColors         spinnerNextInteriorDisabled      = desaturate(desaturate(spinnerNextInteriorEnabled));
65  
66      private PaintContext      ctx;
67      private CommonControlState type;
68      private boolean           focused;
69      private boolean           isForeground;
70  
71      public SpinnerNextButtonPainter(Which state) {
72          super();
73          this.ctx = new PaintContext(CacheMode.FIXED_SIZES);
74  
75          type = getButtonType(state);
76          focused = (state == Which.BACKGROUND_FOCUSED || state == Which.BACKGROUND_PRESSED_FOCUSED);
77          isForeground = isForeground(state);
78      }
79  
80      @Override
81      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
82          if (isForeground) {
83              paintArrow(g, width, height);
84          } else {
85              paintButton(g, c, width, height);
86          }
87      }
88  
89      @Override
90      protected PaintContext getPaintContext() {
91          return ctx;
92      }
93  
94      private CommonControlState getButtonType(Which state) {
95          switch (state) {
96          case BACKGROUND_DISABLED:
97          case FOREGROUND_DISABLED:
98              return CommonControlState.DISABLED;
99          case BACKGROUND_ENABLED:
100         case BACKGROUND_FOCUSED:
101         case FOREGROUND_ENABLED:
102         case FOREGROUND_FOCUSED:
103             return CommonControlState.ENABLED;
104         case BACKGROUND_PRESSED_FOCUSED:
105         case BACKGROUND_PRESSED:
106         case FOREGROUND_PRESSED_FOCUSED:
107         case FOREGROUND_PRESSED:
108             return CommonControlState.PRESSED;
109         }
110         return null;
111     }
112 
113     private boolean isForeground(Which state) {
114         switch (state) {
115         case FOREGROUND_DISABLED:
116         case FOREGROUND_ENABLED:
117         case FOREGROUND_FOCUSED:
118         case FOREGROUND_PRESSED_FOCUSED:
119         case FOREGROUND_PRESSED:
120             return true;
121         default:
122             return false;
123         }
124     }
125 
126     private void paintButton(Graphics2D g, JComponent c, int width, int height) {
127         boolean useToolBarColors = isInToolBar(c);
128         Shape s;
129 
130         if (focused) {
131             s = createButtonShape(0, 0, width, height, CornerSize.OUTER_FOCUS);
132             g.setPaint(getFocusPaint(s, FocusType.OUTER_FOCUS, useToolBarColors));
133             g.fill(s);
134 
135             s = createButtonShape(0, 1, width - 1, height - 1, CornerSize.INNER_FOCUS);
136             g.setPaint(getFocusPaint(s, FocusType.INNER_FOCUS, useToolBarColors));
137             g.fill(s);
138         }
139 
140         s = createButtonShape(0, 2, width - 2, height - 2, CornerSize.BORDER);
141         g.setPaint(getSpinnerNextBorderPaint(s, type));
142         g.fill(s);
143 
144         s = createButtonShape(1, 3, width - 4, height - 4, CornerSize.INTERIOR);
145         g.setPaint(getSpinnerNextInteriorPaint(s, type));
146         g.fill(s);
147     }
148 
149     private void paintArrow(Graphics2D g, int width, int height) {
150         Shape s = createArrowShape(width, height);
151         g.setPaint(getCommonArrowPaint(s, type));
152         g.fill(s);
153     }
154 
155     private Shape createButtonShape(int x, int y, int width, int height, CornerSize size) {
156         return shapeGenerator.createRoundRectangle(x, y, width, height, size, CornerStyle.SQUARE, CornerStyle.SQUARE, CornerStyle.SQUARE,
157             CornerStyle.ROUNDED);
158     }
159 
160     private Shape createArrowShape(int left, int height) {
161         int centerX = left / 2 -3;
162         int centerY = height / 2;
163         return shapeGenerator.createArrowUp(centerX, centerY, 4, 3);
164     }
165 
166     public Paint getSpinnerNextBorderPaint(Shape s, CommonControlState type) {
167         TwoColors colors = getSpinnerNextBorderColors(type);
168         return createVerticalGradient(s, colors);
169     }
170 
171     public Paint getSpinnerNextInteriorPaint(Shape s, CommonControlState type) {
172         TwoColors colors = getSpinnerNextInteriorColors(type);
173         return createVerticalGradient(s, colors);
174     }
175 
176     private TwoColors getSpinnerNextBorderColors(CommonControlState type) {
177         switch (type) {
178         case DISABLED:
179             return spinnerNextBorderDisabled;
180         case ENABLED:
181             return spinnerNextBorderEnabled;
182         case PRESSED:
183             return spinnerNextBorderPressed;
184         }
185         return null;
186     }
187 
188     private TwoColors getSpinnerNextInteriorColors(CommonControlState type) {
189         switch (type) {
190         case DISABLED:
191             return spinnerNextInteriorDisabled;
192         case ENABLED:
193             return spinnerNextInteriorEnabled;
194         case PRESSED:
195             return spinnerNextInteriorPressed;
196         }
197         return null;
198     }
