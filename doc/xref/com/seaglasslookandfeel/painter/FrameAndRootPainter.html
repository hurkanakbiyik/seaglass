98      /**
99       * {@inheritDoc}
100      */
101     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
102         if (state == Which.BACKGROUND_ENABLED_NOFRAME) {
103             return;
104         }
105 
106         Shape s = shapeGenerator.createRoundRectangle(0, 0, (width - 1), (height - 1), CornerSize.FRAME_BORDER);
107 
108         g.setPaint(getFrameBorderPaint(s));
109         g.draw(s);
110 
111         JMenuBar    mb     = null;
112         Component[] cArray = null;
113 
114         if (c instanceof JInternalFrame) {
115             JInternalFrame iframe = (JInternalFrame) c;
116 
117             mb     = iframe.getJMenuBar();
118             cArray = iframe.getContentPane().getComponents();
119         } else if (c instanceof JRootPane) {
120             JRootPane root = (JRootPane) c;
121 
122             mb     = root.getJMenuBar();
123             cArray = root.getContentPane().getComponents();
124         }
125 
126         int topToolBarHeight    = 0;
127         int bottomToolBarHeight = 0;
128 
129         if (cArray != null) {
130 
131             for (Component comp : cArray) {
132 
133                 if (comp instanceof JToolBar) {
134 
135                     if (toolBarNorthState.isInState((JComponent) comp)) {
136                         topToolBarHeight = comp.getHeight();
137                     } else if (toolBarSouthState.isInState((JComponent) comp)) {
138                         bottomToolBarHeight = comp.getHeight();
139                     }
140                 }
141             }
142         }
143 
144         int titleHeight = TITLE_BAR_HEIGHT;
145 
146         if (mb != null && c.getClientProperty("SeaGlass.JRootPane.MenuInTitle") == Boolean.TRUE) {
147             titleHeight += mb.getHeight();
148         }
149 
150         if (c.getClientProperty(SeaGlassRootPaneUI.UNIFIED_TOOLBAR_LOOK) == Boolean.TRUE) {
151             // Draw background gradient.
152             s = shapeGenerator.createRoundRectangle(1, 1, width - 2, height - 2, CornerSize.FRAME_INNER_HIGHLIGHT);
153             g.setPaint(getFrameInteriorPaint(s, titleHeight, topToolBarHeight, bottomToolBarHeight));
154             g.fill(s);
155         } else {
156             // Paint title bar.
157             s = shapeGenerator.createRoundRectangle(1, 1, width - 2, titleHeight, CornerSize.FRAME_INNER_HIGHLIGHT,
158                                                     CornerStyle.ROUNDED, CornerStyle.SQUARE, CornerStyle.SQUARE, CornerStyle.ROUNDED);
159             g.setPaint(getTitleBarInteriorPaint(s, titleHeight));
160             g.fill(s);
161             // Paint contents.
162             s = shapeGenerator.createRoundRectangle(1, titleHeight, width - 2, height - titleHeight - 2, CornerSize.FRAME_INNER_HIGHLIGHT,
163                                                     CornerStyle.SQUARE, CornerStyle.ROUNDED, CornerStyle.ROUNDED, CornerStyle.SQUARE);
164             g.setPaint(c.getBackground());
165             g.fill(s);
166             // Draw separator line.
167             g.setPaint(Color.BLACK);
168             g.drawLine(1, titleHeight, width - 2, titleHeight);
169         }
170 
171         s = shapeGenerator.createRoundRectangle(1, 1, width - 3, height - 3, CornerSize.FRAME_INTERIOR);
172         g.setPaint(getFrameInnerHighlightPaint(s));
173         g.draw(s);
174     }

268     /**
269      * Create the gradient to paint the frame interior.
270      *
271      * @param  s                   the interior shape.
272      * @param  titleHeight         the height of the title bar, or 0 if none.
273      * @param  topToolBarHeight    the height of the top toolbar, or 0 if none.
274      * @param  bottomToolBarHeight the height of the bottom toolbar, or 0 if
275      *                             none.
276      * @param  defColors           the color set to construct the gradient from.
277      *
278      * @return the gradient.
279      */
280     private Paint createFrameGradient(Shape s, int titleHeight, int topToolBarHeight, int bottomToolBarHeight, FourColors defColors) {
281         Rectangle2D bounds = s.getBounds2D();
282         float       x      = (float) bounds.getX();
283         float       y      = (float) bounds.getY();
284         float       w      = (float) bounds.getWidth();
285         float       h      = (float) bounds.getHeight();
286 
287         float midX        = x + w / 2.0f;
288         float titleBottom = titleHeight / h;
289 
290         if (titleBottom >= 1.0f) {
291             titleBottom = 1.0f - 0.00004f;
292         }
293 
294         float[] midPoints = null;
295         Color[] colors    = null;
296 
297         if (topToolBarHeight > 0 && bottomToolBarHeight > 0) {
298             float topToolBarBottom = (titleHeight + topToolBarHeight) / h;
299 
300             if (topToolBarBottom >= 1.0f) {
301                 topToolBarBottom = 1.0f - 0.00002f;
302             }
303 
304             float bottomToolBarTop = (h - 2 - bottomToolBarHeight) / h;
305 
306             if (bottomToolBarTop >= 1.0f) {
307                 bottomToolBarTop = 1.0f - 0.00002f;
308             }
309 
310             midPoints = new float[] { 0.0f, topToolBarBottom, bottomToolBarTop, 1.0f };
311             colors    = new Color[] { defColors.top, defColors.upperMid, defColors.lowerMid, defColors.bottom };
312         } else if (topToolBarHeight > 0) {
313             float toolBarBottom = (titleHeight + topToolBarHeight) / h;
314 
315             if (toolBarBottom >= 1.0f) {
316                 toolBarBottom = 1.0f - 0.00002f;
317             }
318 
319             midPoints = new float[] { 0.0f, toolBarBottom, 1.0f };
320             colors    = new Color[] { defColors.top, defColors.upperMid, defColors.lowerMid };
321         } else if (bottomToolBarHeight > 0) {
322             float bottomToolBarTop = (h - 2 - bottomToolBarHeight) / h;
323 
324             if (bottomToolBarTop >= 1.0f) {
325                 bottomToolBarTop = 1.0f - 0.00002f;
326             }
327 
328             midPoints = new float[] { 0.0f, titleBottom, bottomToolBarTop, 1.0f };
329             colors    = new Color[] { defColors.top, defColors.upperMid, defColors.lowerMid, defColors.bottom };
330         } else {
331             midPoints = new float[] { 0.0f, titleBottom, 1.0f };
332             colors    = new Color[] { defColors.top, defColors.upperMid, defColors.bottom };
333         }
334 
335         return createGradient(midX, y, midX, y + h, midPoints, colors);
336     }

338     /**
339      * Create the gradient to paint the frame interior.
340      *
341      * @param  s           the interior shape.
342      * @param  titleHeight the height of the title bar, or 0 if none.
343      * @param  defColors   the color set to construct the gradient from.
344      *
345      * @return the gradient.
346      */
347     private Paint createTitleBarGradient(Shape s, int titleHeight, FourColors defColors) {
348         Rectangle2D bounds = s.getBounds2D();
349         float       midX   = (float) bounds.getCenterX();
350         float       y      = (float) bounds.getY();
351         float       h      = (float) bounds.getHeight();
352 
353         return createGradient(midX, y, midX, y + h, new float[] { 0.0f, 1.0f },
354                               new Color[] { defColors.top, defColors.upperMid });
355     }
