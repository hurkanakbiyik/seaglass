37  public final class RadioButtonPainter extends AbstractCommonColorsPainter {
38      public static enum Which {
39          ICON_DISABLED,
40          ICON_ENABLED,
41          ICON_FOCUSED,
42          ICON_PRESSED,
43          ICON_PRESSED_FOCUSED,
44          ICON_SELECTED,
45          ICON_SELECTED_FOCUSED,
46          ICON_PRESSED_SELECTED,
47          ICON_PRESSED_SELECTED_FOCUSED,
48          ICON_DISABLED_SELECTED,
49      }
50  
51      private Color             buttonBulletBottomEnabled = decodeColor("buttonBulletBottomEnabled");
52  
53      private TwoColors         buttonBulletEnabled       = new TwoColors(deriveColor(buttonBulletBottomEnabled, 0f, 0f, 0.2f, 0),
54                                                              buttonBulletBottomEnabled);
55      private TwoColors         buttonbulletDisabled      = disable(buttonBulletEnabled);
56  
57      private Effect            dropShadow                = new SeaGlassDropShadowEffect();
58  
59      private PaintContext      ctx;
60      private boolean           focused;
61      private boolean           selected;
62      private CommonControlState type;
63  
64      public RadioButtonPainter(Which state) {
65          super();
66          ctx = new PaintContext(CacheMode.FIXED_SIZES);
67  
68          switch (state) {
69          case ICON_DISABLED:
70          case ICON_ENABLED:
71          case ICON_PRESSED:
72          case ICON_SELECTED:
73          case ICON_PRESSED_SELECTED:
74          case ICON_DISABLED_SELECTED:
75              focused = false;
76              break;
77          case ICON_FOCUSED:
78          case ICON_PRESSED_FOCUSED:
79          case ICON_SELECTED_FOCUSED:
80          case ICON_PRESSED_SELECTED_FOCUSED:
81              focused = true;
82              break;
83          }
84  
85          switch (state) {
86          case ICON_SELECTED:
87          case ICON_SELECTED_FOCUSED:
88          case ICON_PRESSED_SELECTED:
89          case ICON_PRESSED_SELECTED_FOCUSED:
90          case ICON_DISABLED_SELECTED:
91              selected = true;
92              break;
93          default:
94              selected = false;
95          }
96  
97          type = getButtonType(state);
98      }
99  
100     @Override
101     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
102         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
103 
104         Shape s;
105         if (focused) {
106             boolean useToolBarFocus = isInToolBar(c);
107             s = createBasicShape(width, width, height);
108             g.setPaint(getFocusPaint(s, FocusType.OUTER_FOCUS, useToolBarFocus));
109             g.fill(s);
110             s = createBasicShape(width - 2, width, height);
111             g.setPaint(getFocusPaint(s, FocusType.INNER_FOCUS, useToolBarFocus));
112             g.fill(s);
113         }
114 
115         s = createBasicShape(width - 4, width, height);
116         if (!focused) {
117             dropShadow.fill(g, s);
118         }
119         g.setPaint(getCommonBorderPaint(s, type));
120         g.fill(s);
121 
122         s = createBasicShape(width - 6, width, height);
123         g.setPaint(getCommonInteriorPaint(s, type));
124         g.fill(s);
125 
126         if (selected) {
127             s = createBasicShape(width / 4.5, width, height);
128             g.setPaint(getRadioButtonBulletPaint(s, type));
129             g.fill(s);
130         }
131     }
132 
133     private CommonControlState getButtonType(Which state) {
134         switch (state) {
135         case ICON_DISABLED:
136         case ICON_DISABLED_SELECTED:
137             return CommonControlState.DISABLED;
138         case ICON_ENABLED:
139         case ICON_FOCUSED:
140             return CommonControlState.ENABLED;
141         case ICON_PRESSED:
142         case ICON_PRESSED_FOCUSED:
143             return CommonControlState.PRESSED;
144         case ICON_SELECTED:
145         case ICON_SELECTED_FOCUSED:
146             return CommonControlState.SELECTED;
147         case ICON_PRESSED_SELECTED:
148         case ICON_PRESSED_SELECTED_FOCUSED:
149             return CommonControlState.PRESSED_SELECTED;
150         }
151         return null;
152     }
153 
154     @Override
155     protected PaintContext getPaintContext() {
156         return ctx;
157     }
158 
159     private Shape createBasicShape(double diameter, int width, int height) {
160         int pos = (int) ((width - diameter) / 2.0 + 0.5);
161         int iDiameter = (int) (diameter + 0.5);
162         return shapeGenerator.createRadioButton(pos, pos, iDiameter);
163     }
164 
165     public Paint getRadioButtonBulletPaint(Shape s, CommonControlState type) {
166         TwoColors colors = getRadioButtonBulletColors(type);
167         return createVerticalGradient(s, colors);
168     }
169 
170     private TwoColors getRadioButtonBulletColors(CommonControlState type) {
171         switch (type) {
172         case DISABLED:
173         case DISABLED_SELECTED:
174             return buttonbulletDisabled;
175         case ENABLED:
176         case PRESSED:
177         case SELECTED:
178         case PRESSED_SELECTED:
179             return buttonBulletEnabled;
180         }
181         return null;
182     }
