37      public static enum Which {
38          BACKGROUND_DISABLED,
39          BACKGROUND_ENABLED,
40          BACKGROUND_FOCUSED,
41          BACKGROUND_PRESSED_FOCUSED,
42          BACKGROUND_PRESSED,
43          FOREGROUND_DISABLED,
44          FOREGROUND_ENABLED,

51      private Color             spinnerPrevInteriorTopEnabled    = decodeColor("spinnerPrevInteriorTopEnabled");
52      private Color             spinnerPrevInteriorBottomEnabled = decodeColor("spinnerPrevInteriorBottomEnabled");
53      private Color             spinnerPrevInteriorPressedTop    = decodeColor("spinnerPrevInteriorPressedTop");
54      private Color             spinnerPrevInteriorPressedBottom = decodeColor("spinnerPrevInteriorPressedBottom");
55      private Color             spinnerPrevTopLineEnabled        = decodeColor("spinnerPrevTopLineEnabled");
56      private Color             spinnerPrevTopLinePressed        = decodeColor("spinnerPrevTopLinePressed");
57      private Color             scrollBarThumbBorderBasePressed  = decodeColor("scrollBarThumbBorderBasePressed");

59      private Color             scrollBarThumbBorderTopPressed   = deriveColor(scrollBarThumbBorderBasePressed, -0.002239f, 0.041885f, 0f, 0);
60      private TwoColors         spinnerPrevBorderEnabled         = new TwoColors(spinnerPrevBorderTopEnabled, scrollBarThumbBorderTopPressed);
61      private TwoColors         spinnerPrevBorderPressed         = new TwoColors(spinnerPrevBorderTopEnabled, scrollBarThumbBorderTopPressed);
62      private TwoColors         spinnerPrevBorderDisabled        = disable(spinnerPrevBorderEnabled);

64      private TwoColors         spinnerPrevInteriorEnabled       = new TwoColors(spinnerPrevInteriorTopEnabled,

66      private TwoColors         spinnerPrevInteriorPressed       = new TwoColors(spinnerPrevInteriorPressedTop,

72      private PaintContext      ctx;
73      private CommonControlState type;
74      private boolean           focused;
75      private boolean           isForeground;

77      public SpinnerPreviousButtonPainter(Which state) {
78          super();
79          this.ctx = new PaintContext(CacheMode.FIXED_SIZES);
80  
81          type = getButtonType(state);
82          focused = (state == Which.BACKGROUND_FOCUSED || state == Which.BACKGROUND_PRESSED_FOCUSED);
83          isForeground = isForeground(state);
84      }
85  
86      @Override
87      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
88          if (isForeground) {
89              paintArrow(g, width, height);
90          } else {
91              paintButton(g, c, width, height);
92          }
93      }
94  
95      @Override
96      protected PaintContext getPaintContext() {
97          return ctx;
98      }

100     private CommonControlState getButtonType(Which state) {
101         switch (state) {
102         case BACKGROUND_DISABLED:
103         case FOREGROUND_DISABLED:
104             return CommonControlState.DISABLED;
105         case BACKGROUND_ENABLED:
106         case BACKGROUND_FOCUSED:
107         case FOREGROUND_ENABLED:
108         case FOREGROUND_FOCUSED:
109             return CommonControlState.ENABLED;
110         case BACKGROUND_PRESSED_FOCUSED:
111         case BACKGROUND_PRESSED:
112         case FOREGROUND_PRESSED_FOCUSED:
113         case FOREGROUND_PRESSED:
114             return CommonControlState.PRESSED;
115         }
116         return null;
117     }
118 
119     private boolean isForeground(Which state) {
120         switch (state) {
121         case FOREGROUND_DISABLED:
122         case FOREGROUND_ENABLED:
123         case FOREGROUND_FOCUSED:
124         case FOREGROUND_PRESSED_FOCUSED:
125         case FOREGROUND_PRESSED:
126             return true;
127         default:
128             return false;
129         }
130     }

132     private void paintButton(Graphics2D g, JComponent c, int width, int height) {
133         boolean useToolBarColors = isInToolBar(c);
134         Shape s;
135 
136         if (focused) {
137             s = createButtonShape(0, 0, width, height, CornerSize.OUTER_FOCUS);
138             g.setPaint(getFocusPaint(s, FocusType.OUTER_FOCUS, useToolBarColors));
139             g.fill(s);
140 
141             s = createButtonShape(0, 0, width - 1, height - 1, CornerSize.INNER_FOCUS);
142             g.setPaint(getFocusPaint(s, FocusType.INNER_FOCUS, useToolBarColors));
143             g.fill(s);
144         }
145 
146         s = createButtonShape(0, 0, width - 2, height - 2, CornerSize.BORDER);
147         g.setPaint(getSpinnerPrevBorderPaint(s, type));
148         g.fill(s);
149 
150         s = createButtonShape(1, 1, width - 4, height - 4, CornerSize.INTERIOR);
151         g.setPaint(getSpinnerPrevInteriorPaint(s, type));
152         g.fill(s);
153 
154         s = shapeGenerator.createRectangle(1, 0, width - 4, 1);
155         g.setPaint(getSpinnerPrevTopLinePaint(s, type));
156         g.fill(s);
157     }
158 
159     private void paintArrow(Graphics2D g, int width, int height) {
160         Shape s = createArrowShape(width, height);
161         g.setPaint(getCommonArrowPaint(s, type));
162         g.fill(s);
163     }
164 
165     private Shape createButtonShape(int x, int y, int width, int height, CornerSize size) {
166         return shapeGenerator.createRoundRectangle(x, y, width, height, size, CornerStyle.SQUARE, CornerStyle.SQUARE, CornerStyle.ROUNDED,
167             CornerStyle.SQUARE);
168     }
169 
170     private Shape createArrowShape(int left, int height) {
171         int centerX = 8;
172         int centerY = height / 2;
173         return shapeGenerator.createArrowDown(centerX, centerY - 2, 4, 3);
174     }
