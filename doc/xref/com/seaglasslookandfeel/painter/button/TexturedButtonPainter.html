28  /**
29   * Paint a "textured" button. This is suitable for placing on darker grey
30   * backgrounds such as toolbars and bottom bars.
31   *
32   * @author Kathryn Huxtable
33   */
34  public class TexturedButtonPainter extends SegmentedButtonPainter {
35  
36      private Color texturedButtonBorderBaseEnabled = decodeColor("texturedButtonBorderBaseEnabled");
37  
38      private Color texturedButtonInteriorBaseEnabled         = decodeColor("texturedButtonInteriorBaseEnabled");
39      private Color texturedButtonInteriorBasePressed         = decodeColor("texturedButtonInteriorBasePressed");
40      private Color texturedButtonInteriorBaseSelected        = decodeColor("texturedButtonInteriorBaseSelected");
41      private Color texturedButtonInteriorBasePressedSelected = decodeColor("texturedButtonInteriorBasePressedSelected");
42  
43      private Color texturedButtonBorderTopEnabled    = texturedButtonBorderBaseEnabled;
44      private Color texturedButtonBorderBottomEnabled = texturedButtonBorderBaseEnabled;
45  
46      private TwoColors texturedButtonBorderEnabled          = new TwoColors(texturedButtonBorderTopEnabled,
47                                                                             texturedButtonBorderBottomEnabled);
48      private TwoColors texturedButtonBorderPressed          = super.getCommonBorderColors(CommonControlState.PRESSED);
49      private TwoColors texturedButtonBorderDefault          = texturedButtonBorderPressed;
50      private TwoColors texturedButtonBorderSelected         = texturedButtonBorderDefault;
51      private TwoColors texturedButtonBorderDefaultPressed   = texturedButtonBorderPressed;
52      private TwoColors texturedButtonBorderPressedSelected  = texturedButtonBorderDefaultPressed;
53      private TwoColors texturedButtonBorderDisabled         = disable(texturedButtonBorderEnabled);
54      private TwoColors texturedButtonBorderDisabledSelected = disable(texturedButtonBorderPressed);
55  
56      private Color texturedButtonInteriorTopEnabled            = deriveColor(texturedButtonInteriorBaseEnabled, 0f, 0f, 0.054902f, 0);
57      private Color texturedButtonInteriorBottomEnabled         = deriveColor(texturedButtonInteriorBaseEnabled, 0f, 0f, -0.062745f, 0);
58      private Color texturedButtonInteriorTopPressed            = deriveColor(texturedButtonInteriorBasePressed, -0.003064f, -0.044334f,
59                                                                              0.074510f, 0);
60      private Color texturedButtonInteriorBottomPressed         = deriveColor(texturedButtonInteriorBasePressed, 0.002723f, 0.057143f,
61                                                                              -0.082353f, 0);
62      private Color texturedButtonInteriorTopSelected           = deriveColor(texturedButtonInteriorBaseSelected, -0.007658f, -0.042881f,
63                                                                              0.078431f, 0);
64      private Color texturedButtonInteriorBottomSelected        = deriveColor(texturedButtonInteriorBaseSelected, 0.006872f, 0.053054f,
65                                                                              -0.082353f, 0);
66      private Color texturedButtonInteriorTopPressedSelected    = deriveColor(texturedButtonInteriorBasePressedSelected, -0.007749f,
67                                                                              -0.043169f, 0.070588f, 0);
68      private Color texturedButtonInteriorBottomPressedSelected = deriveColor(texturedButtonInteriorBasePressedSelected, 0.007011f,
69                                                                              0.056937f, -0.078431f, 0);
70  
71      private TwoColors texturedButtonInteriorEnabled          = new TwoColors(texturedButtonInteriorTopEnabled,
72                                                                               texturedButtonInteriorBottomEnabled);
73      private TwoColors texturedButtonInteriorPressed          = new TwoColors(texturedButtonInteriorTopPressed,
74                                                                               texturedButtonInteriorBottomPressed);
75      private TwoColors texturedButtonInteriorDefault          = new TwoColors(texturedButtonInteriorTopSelected,
76                                                                               texturedButtonInteriorBottomSelected);
77      private TwoColors texturedButtonInteriorSelected         = texturedButtonInteriorDefault;
78      private TwoColors texturedButtonInteriorDefaultPressed   = texturedButtonInteriorDefault;
79      private TwoColors texturedButtonInteriorPressedSelected  = new TwoColors(texturedButtonInteriorTopPressedSelected,
80                                                                               texturedButtonInteriorBottomPressedSelected);
81      private TwoColors texturedButtonInteriorDisabled         = desaturate(texturedButtonInteriorEnabled);
82      private TwoColors texturedButtonInteriorDisabledSelected = desaturate(texturedButtonInteriorDefault);
83  
84      /**
85       * Create a textured button painter.
86       *
87       * @param state the button state.
88       * @param ctx   the paint context.
89       */
90      public TexturedButtonPainter(Which state, PaintContext ctx) {
91          super(state, ctx);
92      }
93  
94      /**
95       * {@inheritDoc}
96       */
97      public Paint getCommonBorderPaint(Shape s, CommonControlState type) {
98          TwoColors colors = getTexturedButtonBorderColors(type);
99  
100         return createVerticalGradient(s, new TwoColors(colors.top, colors.bottom));
101     }
102 
103     /**
104      * {@inheritDoc}
105      */
106     public Paint getCommonInteriorPaint(Shape s, CommonControlState type) {
107         TwoColors colors = getTexturedButtonInteriorColors(type);
108 
109         return createVerticalGradient(s, new TwoColors(colors.top, colors.bottom));
110     }
111 
112     /**
113      * Get the colors for the border.
114      *
115      * @param  type the button type. Derived from state.
116      *
117      * @return a two-tuple of colors for top and bottom gradient.
118      */
119     public TwoColors getTexturedButtonBorderColors(CommonControlState type) {
120         switch (type) {
121 
122         case DISABLED:
123             return texturedButtonBorderDisabled;
124 
125         case DISABLED_SELECTED:
126             return texturedButtonBorderDisabledSelected;
127 
128         case ENABLED:
129             return texturedButtonBorderEnabled;
130 
131         case PRESSED:
132             return texturedButtonBorderPressed;
133 
134         case DEFAULT:
135             return texturedButtonBorderDefault;
136 
137         case SELECTED:
138             return texturedButtonBorderSelected;
139 
140         case DEFAULT_PRESSED:
141             return texturedButtonBorderDefaultPressed;
142 
143         case PRESSED_SELECTED:
144             return texturedButtonBorderPressedSelected;
145         }
146 
147         return null;
148     }
149 
150     /**
151      * Get the colors for the interior.
152      *
153      * @param  type the button type. Derived from the state.
154      *
155      * @return a two-tuple of colors for top and bottom gradient.
156      */
157     public TwoColors getTexturedButtonInteriorColors(CommonControlState type) {
158         switch (type) {
159 
160         case DISABLED:
161             return texturedButtonInteriorDisabled;
162 
163         case DISABLED_SELECTED:
164             return texturedButtonInteriorDisabledSelected;
165 
166         case ENABLED:
167             return texturedButtonInteriorEnabled;
168 
169         case PRESSED:
170             return texturedButtonInteriorPressed;
171 
172         case DEFAULT:
173             return texturedButtonInteriorDefault;
174 
175         case SELECTED:
176             return texturedButtonInteriorSelected;
177 
178         case DEFAULT_PRESSED:
179             return texturedButtonInteriorDefaultPressed;
180 
181         case PRESSED_SELECTED:
182             return texturedButtonInteriorPressedSelected;
183         }
184 
185         return null;
186     }
