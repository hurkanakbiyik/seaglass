117     /**
118      * {@inheritDoc}
119      */
120     public final void paint(Graphics2D g, JComponent c, int w, int h) {
121         // don't render if the width/height are too small
122         if (w <= 0 || h <= 0)
123             return;
124 
125         Object[] extendedCacheKeys = getExtendedCacheKeys(c);
126 
127         ctx = getPaintContext();
128         CacheMode cacheMode = ctx == null ? CacheMode.NO_CACHING : ctx.getCacheMode();
129 
130         if (cacheMode == CacheMode.NO_CACHING || !ImageCache.getInstance().isImageCachable(w, h) || g instanceof PrinterGraphics) {
131             paintDirectly(g, c, w, h, extendedCacheKeys);
132         } else {
133             paintWithCaching(g, c, w, h, extendedCacheKeys);
134         }
135     }

235     /**
236      * Decodes and returns a color, which is derived from a base color in UI
237      * defaults.
238      *
239      * @param  key     A key corresponding to the value in the UI Defaults table
240      *                 of UIManager where the base color is defined
241      * @param  hOffset The hue offset used for derivation.
242      * @param  sOffset The saturation offset used for derivation.
243      * @param  bOffset The brightness offset used for derivation.
244      * @param  aOffset The alpha offset used for derivation. Between 0...255
245      *
246      * @return The derived color, whos color value will change if the parent
247      *         uiDefault color changes.
248      */
249     protected final Color decodeColor(String key, float hOffset, float sOffset, float bOffset, int aOffset) {
250         if (UIManager.getLookAndFeel() instanceof SeaGlassLookAndFeel) {
251             SeaGlassLookAndFeel laf = (SeaGlassLookAndFeel) UIManager.getLookAndFeel();
252 
253             return laf.getDerivedColor(key, hOffset, sOffset, bOffset, aOffset, true);
254         } else {
255 
256             // can not give a right answer as painter should not be used outside
257             // of nimbus laf but do the best we can
258             return Color.getHSBColor(hOffset, sOffset, bOffset);
259         }
260     }

262     /**
263      * Derive and returns a color, which is based on an existing color.
264      *
265      * @param  src     The source color from which to derive the new color.
266      * @param  hOffset The hue offset used for derivation.
267      * @param  sOffset The saturation offset used for derivation.
268      * @param  bOffset The brightness offset used for derivation.
269      * @param  aOffset The alpha offset used for derivation. Between 0...255
270      *
271      * @return The derived color.
272      */
273     protected Color deriveColor(Color src, float hOffset, float sOffset, float bOffset, int aOffset) {
274         float[] tmp = Color.RGBtoHSB(src.getRed(), src.getGreen(), src.getBlue(), null);
275 
276         // apply offsets
277         tmp[0] = clamp(tmp[0] + hOffset);
278         tmp[1] = clamp(tmp[1] + sOffset);
279         tmp[2] = clamp(tmp[2] + bOffset);
280         int alpha = clamp(src.getAlpha() + aOffset);
281 
282         return new Color((Color.HSBtoRGB(tmp[0], tmp[1], tmp[2]) & 0xFFFFFF) | (alpha << 24), true);
283     }

285     /**
286      * Decodes and returns a color, which is derived from a offset between two

288      *
289      * @param  color1   The first color
290      * @param  color2   The second color
291      * @param  midPoint The offset between color 1 and color 2, a value of 0.0
292      *                  is color 1 and 1.0 is color 2;

294      * @return The derived color

296     protected final Color decodeColor(Color color1, Color color2, float midPoint) {

300     /**
301      * Derives the ARGB value for a color based on an offset between two other
302      * colors.
303      *
304      * @param  color1   The first color
305      * @param  color2   The second color
306      * @param  midPoint The offset between color 1 and color 2, a value of 0.0
307      *                  is color 1 and 1.0 is color 2;
308      *
309      * @return the ARGB value for a new color based on this derivation
310      */
311     public static int deriveARGB(Color color1, Color color2, float midPoint) {
312         int r = color1.getRed() + (int) ((color2.getRed() - color1.getRed()) * midPoint + 0.5f);
313         int g = color1.getGreen() + (int) ((color2.getGreen() - color1.getGreen()) * midPoint + 0.5f);
314         int b = color1.getBlue() + (int) ((color2.getBlue() - color1.getBlue()) * midPoint + 0.5f);
315         int a = color1.getAlpha() + (int) ((color2.getAlpha() - color1.getAlpha()) * midPoint + 0.5f);
316 
317         return ((a & 0xFF) << 24) | ((r & 0xFF) << 16) | ((g & 0xFF) << 8) | (b & 0xFF);
318     }

320     /**
321      * Given parameters for creating a LinearGradientPaint, this method will
322      * create and return a linear gradient paint. One primary purpose for this
323      * method is to avoid creating a LinearGradientPaint where the start and end
324      * points are equal. In such a case, the end y point is slightly increased
325      * to avoid the overlap.
326      *
327      * @param  x1
328      * @param  y1
329      * @param  x2
330      * @param  y2
331      * @param  midpoints
332      * @param  colors
333      *
334      * @return a valid LinearGradientPaint. This method never returns null.
335      */
336     protected final LinearGradientPaint createGradient(float x1, float y1, float x2, float y2, float[] midpoints, Color[] colors) {
337         if (x1 == x2 && y1 == y2) {
338             y2 += .00001f;
339         }
340 
341         return new LinearGradientPaint(x1, y1, x2, y2, midpoints, colors);
342     }

346      * create and return a radial gradient paint. One primary purpose for this
347      * method is to avoid creating a RadialGradientPaint where the radius is
348      * non-positive. In such a case, the radius is just slightly increased to
349      * avoid 0.

442     /**
443      * Get a color property from the given JComponent. First checks for a <code>
444      * getXXX()</code> method and if that fails checks for a client property
445      * with key <code>property</code>. If that still fails to return a Color
446      * then <code>defaultColor</code> is returned.
447      *
448      * @param  c                The component to get the color property from
449      * @param  property         The name of a bean style property or client
450      *                          property
451      * @param  defaultColor     The color to return if no color was obtained
452      *                          from the component.
453      * @param  saturationOffset The offset for the saturation.
454      * @param  brightnessOffset the offset for the brightness.
455      * @param  alphaOffset      the offset for alpha.
456      *
457      * @return The color that was obtained from the component or defaultColor
458      */
459     protected final Color getComponentColor(JComponent c, String property, Color defaultColor, float saturationOffset,
460             float brightnessOffset, int alphaOffset) {
461         Color color = null;
462 
463         if (c != null) {
464 
465             // handle some special cases for performance
466             if ("background".equals(property)) {
467                 color = c.getBackground();
468             } else if ("foreground".equals(property)) {
469                 color = c.getForeground();
470             } else if (c instanceof JList && "selectionForeground".equals(property)) {
471                 color = ((JList) c).getSelectionForeground();
472             } else if (c instanceof JList && "selectionBackground".equals(property)) {
473                 color = ((JList) c).getSelectionBackground();
474             } else if (c instanceof JTable && "selectionForeground".equals(property)) {
475                 color = ((JTable) c).getSelectionForeground();
476             } else if (c instanceof JTable && "selectionBackground".equals(property)) {
477                 color = ((JTable) c).getSelectionBackground();
478             } else {
479                 String s = "get" + Character.toUpperCase(property.charAt(0)) + property.substring(1);
480 
481                 try {
482                     Method method = c.getClass().getMethod(s);
483 
484                     color = (Color) method.invoke(c);
485                 } catch (Exception e) {
486                     // don't do anything, it just didn't work, that's all.
487                     // This could be a normal occurance if you use a property
488                     // name referring to a key in clientProperties instead of
489                     // a real property
490                 }
491 
492                 if (color == null) {
493                     Object value = c.getClientProperty(property);
494 
495                     if (value instanceof Color) {
496                         color = (Color) value;
497                     }
498                 }
499             }
500         }
501 
502         // we return the defaultColor if the color found is null, or if
503         // it is a UIResource. This is done because the color for the
504         // ENABLED state is set on the component, but you don't want to use
505         // that color for the over state. So we only respect the color
506         // specified for the property if it was set by the user, as opposed
507         // to set by us.
508         if (color == null || color instanceof UIResource) {
509             return defaultColor;
510         } else if (saturationOffset != 0 || brightnessOffset != 0 || alphaOffset != 0) {
511             float[] tmp = Color.RGBtoHSB(color.getRed(), color.getGreen(), color.getBlue(), null);
512 
513             tmp[1] = clamp(tmp[1] + saturationOffset);
514             tmp[2] = clamp(tmp[2] + brightnessOffset);
515             int alpha = clamp(color.getAlpha() + alphaOffset);
516 
517             return new Color((Color.HSBtoRGB(tmp[0], tmp[1], tmp[2]) & 0xFFFFFF) | (alpha << 24));
518         } else {
519             return color;
520         }
521     }

539      * Returns a new color with the saturation cut to one third the original,

541      *
542      * @param  color the original color.
543      *
544      * @return the new color.
545      */
546     protected Color desaturate(Color color) {
547         float[] tmp = Color.RGBtoHSB(color.getRed(), color.getGreen(), color.getBlue(), null);

549         tmp[1] /= 3.0f;
550         tmp[2] = clamp(1.0f - (1.0f - tmp[2]) / 3f);
551 
552         return new Color((Color.HSBtoRGB(tmp[0], tmp[1], tmp[2]) & 0xFFFFFF));
553     }

605     /**
606      * Paint the component, using a cached image if possible.
607      *
608      * @param g                 the Graphics2D context to paint with.
609      * @param c                 the component to paint.
610      * @param w                 the component width.
611      * @param h                 the component height.
612      * @param extendedCacheKeys extended cache keys.
613      */
614     private void paintWithCaching(Graphics2D g, JComponent c, int w, int h, Object[] extendedCacheKeys) {
615         VolatileImage img = getImage(g.getDeviceConfiguration(), c, w, h, extendedCacheKeys);
616 
617         if (img != null) {
618 
619             // render cached image
620             g.drawImage(img, 0, 0, null);
621         } else {
622 
623             // render directly
624             paintDirectly(g, c, w, h, extendedCacheKeys);
625         }
626     }

628     /**
629      * Convenience method which creates a temporary graphics object by creating
630      * a clone of the passed in one, configuring it, drawing with it, disposing
631      * it. These steps have to be taken to ensure that any hints set on the
632      * graphics are removed subsequent to painting.
633      *

647     /**
648      * Gets the rendered image for this painter at the requested size, either
649      * from cache or create a new one
650      *
651      * @param  config            the graphics configuration.
652      * @param  c                 the component to paint.
653      * @param  w                 the component width.
654      * @param  h                 the component height.
655      * @param  extendedCacheKeys extended cache keys.
656      *
657      * @return the new image.
658      */
659     private VolatileImage getImage(GraphicsConfiguration config, JComponent c, int w, int h, Object[] extendedCacheKeys) {
660         ImageCache imageCache = ImageCache.getInstance();
661 
662         // get the buffer for this component
663         VolatileImage buffer = (VolatileImage) imageCache.getImage(config, w, h, this, extendedCacheKeys);
664 
665         int renderCounter = 0; // to avoid any potential, though unlikely,
666 
667         // infinite loop
668         do {
669 
670             // validate the buffer so we can check for surface loss
671             int bufferStatus = VolatileImage.IMAGE_INCOMPATIBLE;
672 
673             if (buffer != null) {
674                 bufferStatus = buffer.validate(config);
675             }
676 
677             // If the buffer status is incompatible or restored, then we need to
678             // re-render to the volatile image
679             if (bufferStatus == VolatileImage.IMAGE_INCOMPATIBLE || bufferStatus == VolatileImage.IMAGE_RESTORED) {
680 
681                 // if the buffer is null (hasn't been created), or isn't the
682                 // right size, or has lost its contents,
683                 // then recreate the buffer
684                 if (buffer == null || buffer.getWidth() != w || buffer.getHeight() != h
685                         || bufferStatus == VolatileImage.IMAGE_INCOMPATIBLE) {
686 
687                     // clear any resources related to the old back buffer
688                     if (buffer != null) {
689                         buffer.flush();
690                         buffer = null;
691                     }
692 
693                     // recreate the buffer
694                     buffer = config.createCompatibleVolatileImage(w, h, Transparency.TRANSLUCENT);
695 
696                     // put in cache for future
697                     imageCache.setImage(buffer, config, w, h, this, extendedCacheKeys);
698                 }
699 
700                 // create the graphics context with which to paint to the buffer
701                 Graphics2D bg = buffer.createGraphics();
702 
703                 // clear the background before configuring the graphics
704                 bg.setComposite(AlphaComposite.Clear);
705                 bg.fillRect(0, 0, w, h);
706                 bg.setComposite(AlphaComposite.SrcOver);
707                 configureGraphics(bg);
708 
709                 // paint the painter into buffer
710                 paintDirectly(bg, c, w, h, extendedCacheKeys);
711 
712                 // close buffer graphics
713                 bg.dispose();
714             }
715         } while (buffer.contentsLost() && renderCounter++ < 3);
716 
717         // check if we failed
718         if (renderCounter == 3)
719             return null;
720 
721         // return image
722         return buffer;
723     }
