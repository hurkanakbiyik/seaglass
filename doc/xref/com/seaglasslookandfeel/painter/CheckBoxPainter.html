36  /**
37   * CheckBoxPainter implementation.
38   */
39  public final class CheckBoxPainter extends AbstractCommonColorsPainter {
40  
41      private static final double SIZE_MULTIPLIER = 2.0 / 3.0;
42      private static final double X_MULTIPLIER    = 5.0 / 18.0;
43      private static final double Y_MULTIPLIER    = 1.0 / 18.0;
44  
45      /**
46       * Control state.
47       */
48      public static enum Which {
49          ICON_DISABLED, ICON_ENABLED, ICON_FOCUSED, ICON_PRESSED, ICON_PRESSED_FOCUSED, ICON_SELECTED, ICON_SELECTED_FOCUSED,
50          ICON_PRESSED_SELECTED, ICON_PRESSED_SELECTED_FOCUSED, ICON_DISABLED_SELECTED,
51      }
52  
53      private Color buttonBulletBottomEnabled = decodeColor("buttonBulletBottomEnabled");
54  
55      private TwoColors buttonBulletEnabled  = new TwoColors(deriveColor(buttonBulletBottomEnabled, 0f, 0f, 0.2f, 0),
56                                                             buttonBulletBottomEnabled);
57      private TwoColors buttonbulletDisabled = disable(buttonBulletEnabled);
58  
59      private PaintContext ctx;
60      private boolean      focused;
61      private boolean      selected;
62  
63      private Effect dropShadow = new SeaGlassDropShadowEffect();
64  
65      private CommonControlState type;
66  
67      /**
68       * Creates a new CheckBoxPainter object.
69       *
70       * @param state the control state to paint.
71       */
72      public CheckBoxPainter(Which state) {
73          super();
74          this.ctx = new PaintContext(CacheMode.FIXED_SIZES);
75  
76          type = getControlType(state);
77  
78          focused  = false;
79          selected = false;
80  
81          if (state == Which.ICON_FOCUSED || state == Which.ICON_PRESSED_FOCUSED || state == Which.ICON_SELECTED_FOCUSED
82                  || state == Which.ICON_PRESSED_SELECTED_FOCUSED) {
83              focused = true;
84          }
85  
86          if (state == Which.ICON_SELECTED || state == Which.ICON_PRESSED_SELECTED || state == Which.ICON_DISABLED_SELECTED
87                  || state == Which.ICON_SELECTED_FOCUSED || state == Which.ICON_PRESSED_SELECTED_FOCUSED) {
88              selected = true;
89          }
90      }

92      /**
93       * {@inheritDoc}
94       */
95      @Override
96      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
97          g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
98  
99          int size = width < height ? width : height;
100         int x    = (width - size) / 2;
101         int y    = (height - size) / 2;
102 
103         Shape s;
104 
105         if (focused) {
106             boolean useToolBarFocus = isInToolBar(c);
107 
108             s = shapeGenerator.createRoundRectangle(x, y, size, size, CornerSize.CHECKBOX_OUTER_FOCUS);
109             g.setPaint(getFocusPaint(s, FocusType.OUTER_FOCUS, useToolBarFocus));
110             g.fill(s);
111             s = shapeGenerator.createRoundRectangle(x + 1, y + 1, size - 2, size - 2, CornerSize.CHECKBOX_INNER_FOCUS);
112             g.setPaint(getFocusPaint(s, FocusType.INNER_FOCUS, useToolBarFocus));
113             g.fill(s);
114         }
115 
116         s = shapeGenerator.createRoundRectangle(x + 2, y + 2, size - 4, size - 4, CornerSize.CHECKBOX_BORDER);
117 
118         if (!focused) {
119             dropShadow.fill(g, s);
120         }
121 
122         g.setPaint(getCommonBorderPaint(s, type));
123         g.fill(s);
124 
125         s = shapeGenerator.createRoundRectangle(x + 3, y + 3, size - 6, size - 6, CornerSize.CHECKBOX_INTERIOR);
126         g.setPaint(getCommonInteriorPaint(s, type));
127         g.fill(s);
128 
129         if (selected) {
130             s = createCheckMark(x, y, size);
131             g.setPaint(getCheckBoxBulletPaint(s, type));
132             g.fill(s);
133         }
134     }

143     /**
144      * Get the CommonControlState for the state.
145      *
146      * @param  state the control state.
147      *
148      * @return the common control color state.
149      */
150     private CommonControlState getControlType(Which state) {
151         switch (state) {
152 
153         case ICON_DISABLED:
154             return CommonControlState.DISABLED;
155 
156         case ICON_ENABLED:
157         case ICON_FOCUSED:
158             return CommonControlState.ENABLED;
159 
160         case ICON_PRESSED:
161         case ICON_PRESSED_FOCUSED:
162             return CommonControlState.PRESSED;
163 
164         case ICON_SELECTED:
165         case ICON_SELECTED_FOCUSED:
166             return CommonControlState.SELECTED;
167 
168         case ICON_PRESSED_SELECTED:
169         case ICON_PRESSED_SELECTED_FOCUSED:
170             return CommonControlState.PRESSED_SELECTED;
171 
172         case ICON_DISABLED_SELECTED:
173             return CommonControlState.DISABLED_SELECTED;
174         }
175 
176         return null;
177     }

179     /**
180      * Create the check mark shape.
181      *
182      * @param  x    the x coordinate of the upper-left corner of the check mark.
183      * @param  y    the y coordinate of the upper-left corner of the check mark.
184      * @param  size the check mark size in pixels.
185      *
186      * @return the check mark shape.
187      */
188     private Shape createCheckMark(int x, int y, int size) {
189         int markSize = (int) (size * SIZE_MULTIPLIER + 0.5);
190         int markX    = x + (int) (size * X_MULTIPLIER + 0.5);
191         int markY    = y + (int) (size * Y_MULTIPLIER + 0.5);
192 
193         return shapeGenerator.createCheckMark(markX, markY, markSize, markSize);
194     }
195 
196     /**
197      * Create the paint for the check mark.
198      *
199      * @param  s    the check mark shape. Used for bounds.
200      * @param  type the color type to create.
201      *
202      * @return the paint to use for the check mark.
203      */
204     public Paint getCheckBoxBulletPaint(Shape s, CommonControlState type) {
205         return createCheckMarkGradient(s, getCheckBoxBulletColors(type));
206     }
207 
208     /**
209      * Get the colors to use for the check mark.
210      *
211      * @param  type the state.
212      *
213      * @return the color set.
214      */
215     private TwoColors getCheckBoxBulletColors(CommonControlState type) {
216         switch (type) {
217 
218         case DISABLED:
219         case DISABLED_SELECTED:
220             return buttonbulletDisabled;
221 
222         case ENABLED:
223         case PRESSED:
224         case SELECTED:
225         case PRESSED_SELECTED:
226             return buttonBulletEnabled;
227         }
228 
229         return null;
230     }
231 
232     /**
233      * Creates the gradient to use for the check mark.
234      *
235      * @param  s      the shape of the check mark. Used for bounds.
236      * @param  colors the color set.
237      *
238      * @return the paint for the check mark.
239      */
240     private Paint createCheckMarkGradient(Shape s, TwoColors colors) {
241         Rectangle2D bounds = s.getBounds2D();
242         float       x      = (float) bounds.getX();
243         float       y      = (float) bounds.getY();
244         float       w      = (float) bounds.getWidth();
245         float       h      = (float) bounds.getHeight();
246 
247         return createGradient(x + w, y, (0.3f * w) + x, h + y, new float[] { 0f, 1f }, new Color[] { colors.top, colors.bottom });
248     }
