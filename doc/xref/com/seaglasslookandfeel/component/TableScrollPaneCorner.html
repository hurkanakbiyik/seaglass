20  package com.seaglasslookandfeel.component;

22  import java.awt.Graphics;
23  import java.awt.Graphics2D;
24  
25  import javax.swing.JComponent;
26  import javax.swing.UIManager;
27  import javax.swing.plaf.UIResource;

29  import com.seaglasslookandfeel.painter.SeaGlassPainter;
30  
31  /**
32   * TableScrollPaneCorner - A simple component that paints itself using the table
33   * header background painter. It is used to fill the top right corner of
34   * scrollpane.
35   *
36   * <p>Based on Nimbus's TableScrollPaneCorner by Jasper Potts</p>
37   *
38   */
39  public class TableScrollPaneCorner extends JComponent implements UIResource {
40      private static final long serialVersionUID = -904714225885809898L;
41  
42      /**
43       * Paint the component using the Nimbus Table Header Background Painter
44       *
45       * @param g the Graphics context to paint with.
46       */
47      @Override
48      protected void paintComponent(Graphics g) {
49          SeaGlassPainter painter = (SeaGlassPainter) UIManager.get("TableHeader:\"TableHeader.renderer\"[Enabled].backgroundPainter");
50  
51          if (painter != null) {
52  
53              Graphics2D g2d = (Graphics2D) g;
54  
55              painter.paint(g2d, this, getWidth() + 1, getHeight());
56          }
57      }
58  }
