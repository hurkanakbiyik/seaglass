259      * Iconify/Restore the window.
260      *
261      * @param iconify {@code true} if we are to iconify the window,
262      *                {@code false} if we are to restore the window.
263      */
264     private void setParentIcon(boolean iconify) {
265         if (rootParent instanceof JFrame) {
266             JFrame frame = (JFrame) rootParent;
267             int    state = frame.getExtendedState();
268 
269             ((JFrame) rootParent).setExtendedState(iconify ? state | Frame.ICONIFIED : state & ~Frame.ICONIFIED);
270         }

273     /**
274      * Maximize/Restore the window.
275      *
276      * @param maximize iconify {@code true} if we are to maximize the window,
277      *                 {@code false} if we are to restore the window.
278      */
279     private void setParentMaximum(boolean maximize) {
280         if (rootParent instanceof JFrame) {
281             JFrame frame = (JFrame) rootParent;
282             int    state = frame.getExtendedState();
283 
284             if (maximize) {
285                 GraphicsConfiguration gc = frame.getGraphicsConfiguration();
286                 Insets                i  = Toolkit.getDefaultToolkit().getScreenInsets(gc);
287                 Rectangle             r  = gc.getBounds();
288 
289                 r.x      += i.left;
290                 r.y      += i.top;
291                 r.width  -= i.left + i.right;
292                 r.height -= i.top + i.bottom;
293                 frame.setMaximizedBounds(r);
294             }
295 
296             frame.setExtendedState(maximize ? state | Frame.MAXIMIZED_BOTH : state & ~Frame.MAXIMIZED_BOTH);
297         }
298     }

424     /**
425      * Resets the menuButton icon to match that of the frame.
426      */
427     private void updateMenuIcon() {
428         List<Image> iconList = null;
429         if (rootParent instanceof JFrame) {
430             iconList = ((JFrame) rootParent).getIconImages();
431         } else if (rootParent instanceof JDialog) {
432             iconList = ((JDialog) rootParent).getIconImages();
433         }
434         
435         Image frameIcon = iconList != null && iconList.size() > 0 ? iconList.get(0) : null;
436 
437         SeaGlassContext context   = getContext(this);
438 
439         if (frameIcon != null) {
440             Dimension maxSize   = (Dimension) context.getStyle().get(context, "InternalFrameTitlePane.maxFrameIconSize");
441             int       maxWidth  = 16;
442             int       maxHeight = 16;
443 
444             if (maxSize != null) {
445                 maxWidth  = maxSize.width;
446                 maxHeight = maxSize.height;
447             }
448             menuButton.setIcon(new ImageIcon(frameIcon.getScaledInstance(maxWidth, maxHeight, Image.SCALE_SMOOTH)));
449         }
450 
451         context.dispose();
452     }

624     /**
625      * Paint the title pane.
626      *
627      * @param context the SynthContext to use to get the component.
628      * @param g       the Graphics context to paint with.
629      */
630     private void paint(SeaGlassContext context, Graphics g) {
631         String title = getTitle();
632 
633         if (title != null) {
634             SynthStyle style = context.getStyle();
635 
636             Color color = style.getColor(context, ColorType.TEXT_FOREGROUND);
637 
638             // TODO style.getColor returns improper color for state? Why?
639             if ((context.getComponentState() & 512) != 0) {
640                 Object obj = style.get(context, "[WindowFocused].textForeground");
641 
642                 if (obj != null && obj instanceof Color) {
643                     color = (Color) obj;
644                 }
645 
646                 // FIXME The state *still* doesn't get the color right!!!
647                 color = Color.BLACK;
648             }
649 
650             g.setColor(color);
651             g.setFont(style.getFont(context));
652 
653             // Center text vertically.
654             FontMetrics fm         = SwingUtilities2.getFontMetrics(rootPane, g);
655             int         baseline   = (getHeight() + fm.getAscent() - fm.getLeading() - fm.getDescent()) / 2;
656             JButton     lastButton = null;
657 
658             if (isParentIconifiable()) {
659                 lastButton = iconButton;
660             } else if (isParentMaximizable()) {
661                 lastButton = maxButton;
662             } else if (isParentClosable()) {
663                 lastButton = closeButton;
664             }
665 
666             int     maxX;
667             int     minX;
668             boolean ltr = isParentLeftToRight();
669 
670             if (ltr) {
671 
672                 if (lastButton != null) {
673                     maxX = lastButton.getX() - titleSpacing;
674                 } else {
675                     maxX = getParentWidth() - getParentInsets().right - titleSpacing;
676                 }
677 
678                 minX = menuButton.getX() + menuButton.getWidth() + titleSpacing;
679             } else {
680 
681                 if (lastButton != null) {
682                     minX = lastButton.getX() + lastButton.getWidth() + titleSpacing;
683                 } else {
684                     minX = getParentInsets().left + titleSpacing;
685                 }
686 
687                 maxX = getParentWidth() - getParentInsets().right - menuButton.getX() - titleSpacing;
688             }
689 
690             String clippedTitle = getTitle(title, fm, maxX - minX);
691 
692             if (clippedTitle == title) {
693                 int width = style.getGraphicsUtils(context).computeStringWidth(context, g.getFont(), fm, title);
694 
695                 minX = Math.max(minX, (getWidth() - width) / 2);
696                 minX = Math.min(maxX - width, minX);
697             }
698 
699             // style.getGraphicsUtils(context).paintText(context, g,
700             // clippedTitle, minX, baseline - fm.getAscent(), -1);
701             ((SeaGlassGraphicsUtils) style.getGraphicsUtils(context)).drawEmphasizedText(g, color, DEFAULT_EMPHASIS_COLOR, clippedTitle,
702                                                                                          minX, baseline); // - fm.getAscent());
703         }
704     }

782     /**
783      * @see java.beans.PropertyChangeListener#propertyChange(java.beans.PropertyChangeEvent)
784      */
785     public void propertyChange(PropertyChangeEvent evt) {
786         if (evt.getSource() == this) {
787 
788             if (SeaGlassLookAndFeel.shouldUpdateStyle(evt)) {
789                 updateStyle(this);
790             }
791         } else {
792             // Changes for the internal frame
793             if (evt.getPropertyName() == "iconImage") {
794                 updateMenuIcon();
795             }
796         }
797 
798         // Basic (from Handler inner class)
799         String prop = (String) evt.getPropertyName();
800 
801         if (closeButton != null && WINDOW_DOCUMENT_MODIFIED.equals(prop)) {
802             closeButton.revalidate();
803             closeButton.repaint();
804 
805             return;
806         }
807 
808         if (prop == JInternalFrame.IS_SELECTED_PROPERTY) {
809             repaint();
810 
811             return;
812         }
813 
814         // Frame.state isn't currently bound.
815         if ("resizable".equals(prop) || "state".equals(prop)) {
816             Frame frame = (JFrame) rootParent;
817 
818             if (frame != null) {
819                 setState(frame.getExtendedState(), true);
820             }
821 
822             if ("resizable".equals(prop)) {
823                 getRootPane().repaint();
824             }
825         } else if ("title".equals(prop)) {
826             repaint();
827         } else if ("componentOrientation" == prop) {
828             revalidate();
829             repaint();
830         } else if ("iconImage" == prop) {
831             revalidate();
832             repaint();
833         }
834 
835         if (prop == JInternalFrame.IS_ICON_PROPERTY || prop == JInternalFrame.IS_MAXIMUM_PROPERTY) {
836             setButtonTooltips();
837             enableActions();
838 
839             return;
840         }
841 
842         if ("closable" == prop) {
843 
844             if ((Boolean) evt.getNewValue() == Boolean.TRUE) {
845                 add(closeButton);
846             } else {
847                 remove(closeButton);
848             }
849         } else if ("maximizable" == prop) {
850 
851             if ((Boolean) evt.getNewValue() == Boolean.TRUE) {
852                 add(maxButton);
853             } else {
854                 remove(maxButton);
855             }
856         } else if ("iconable" == prop) {
857 
858             if ((Boolean) evt.getNewValue() == Boolean.TRUE) {
859                 add(iconButton);
860             } else {
861                 remove(iconButton);
862             }
863         }
864 
865         enableActions();
866 
867         revalidate();
868         repaint();
869     }

873      * and the state has not changed, this will update anyway.

875      * @param state            the state of the window.
876      * @param updateRegardless {@code true} if we are to update regardless of

879     private void setState(int state, boolean updateRegardless) {
880         Window w = (Window) rootParent;
881 
882         if (w != null && rootPane.getWindowDecorationStyle() == JRootPane.FRAME) {
883

894                         && frame.isShowing()) {
895                     rootPane.setBorder(null);

910                     } else {
911

913                         maximizeAction.setEnabled(true);
914                         restoreAction.setEnabled(false);
915                     }
916                     // if (toggleButton.getParent() == null ||
917                     // iconifyButton.getParent() == null) {
918                     // add(toggleButton);
919                     // add(iconifyButton);
920                     // revalidate();
921                     // repaint();
922                     // }
923                     // toggleButton.setText(null);
924                 } else {
925                     maximizeAction.setEnabled(false);
926                     restoreAction.setEnabled(false);
927                     // if (toggleButton.getParent() != null) {
928                     // remove(toggleButton);
929                     // revalidate();
930                     // repaint();
931                     // }
932                 }
933             } else {
934 
935                 // Not contained in a Frame
936                 maximizeAction.setEnabled(false);
937                 restoreAction.setEnabled(false);
938                 iconifyAction.setEnabled(false);
939 
940                 // remove(toggleButton);
941                 // remove(iconifyButton);
942                 revalidate();
943                 repaint();

1025             SynthGraphicsUtils graphicsUtils = context.getStyle().getGraphicsUtils(context);

1030

1033                 int subtitle_w = graphicsUtils.computeStringWidth(context, fm.getFont(), fm, frameTitle.substring(0, 3) + "...");
1034

1036             } else {

1038             }

1040             height = Math.max(fm.getHeight(), height);
1041 
1042             width += titleSpacing + titleSpacing;
1043 
1044             Insets insets = getInsets();
1045 
1046             height += insets.top + insets.bottom;
1047             width  += insets.left + insets.right;
