40  /**
41   * SeaGlassBorder is a border that delegates to a Painter. The Insets are
42   * determined at construction time.
43   *
44   * <p>Based on SynthBorder by Scott Violet.</p>
45   *
46   * @see javax.swing.plaf.synth.SynthBorder
47   */
48  public class SeaGlassBorder extends AbstractBorder implements UIResource {
49      private static final long serialVersionUID = 2565840147056359672L;
50      private SeaglassUI        ui;
51      private Insets            insets;
52  
53      /**
54       * Creates a new SeaGlassBorder object.
55       *
56       * @param ui     the UI delegate for the control.
57       * @param insets the border insets.
58       */
59      public SeaGlassBorder(SeaglassUI ui, Insets insets) {
60          this.ui     = ui;
61          this.insets = insets;
62      }
63  
64      /**
65       * Creates a new SeaGlassBorder object with no insets.
66       *
67       * @param ui the UI degelate for the control.
68       */
69      public SeaGlassBorder(SeaglassUI ui) {
70          this(ui, null);
71      }
72  
73      /**
74       * @see javax.swing.border.AbstractBorder#paintBorder(java.awt.Component,java.awt.Graphics,
75       *      int, int, int, int)
76       */
77      public void paintBorder(Component c, Graphics g, int x, int y, int width, int height) {
78          JComponent      jc      = (JComponent) c;
79          SeaGlassContext context = (SeaGlassContext) ui.getContext(jc);
80          SeaGlassStyle   style   = (SeaGlassStyle) context.getStyle();
81  
82          if (style == null) {
83              assert false : "SeaGlassBorder is being used outside after the UI has been uninstalled";
84  
85              return;
86          }
87  
88          ui.paintBorder(context, g, x, y, width, height);
89          context.dispose();
90      }
91  
92      /**
93       * This default implementation returns a new <code>Insets</code> instance
94       * where the <code>top</code>, <code>left</code>, <code>bottom</code>, and
95       * <code>right</code> fields are set to <code>0</code>.
96       *
97       * @param  c the component for which this border insets value applies
98       *
99       * @return the new <code>Insets</code> object initialized to 0
100      */
101     public Insets getBorderInsets(Component c) {
102         return getBorderInsets(c, null);
103     }
104 
105     /**
106      * Reinitializes the insets parameter with this Border's current Insets.
107      *
108      * @param  c      the component for which this border insets value applies
109      * @param  insets the object to be reinitialized
110      *
111      * @return the <code>insets</code> object
112      */
113     public Insets getBorderInsets(Component c, Insets insets) {
114         if (this.insets != null) {
115 
116             if (insets == null) {
117                 insets = new Insets(this.insets.top, this.insets.left, this.insets.bottom, this.insets.right);
118             } else {
119                 insets.top    = this.insets.top;
120                 insets.bottom = this.insets.bottom;
121                 insets.left   = this.insets.left;
122                 insets.right  = this.insets.right;
123             }
124         } else if (insets == null) {
125             insets = new Insets(0, 0, 0, 0);
126         } else {
127             insets.top = insets.bottom = insets.left = insets.right = 0;
128         }
129 
130         if (c instanceof JComponent) {
131             Insets margin = null;
132             Region region = SeaGlassLookAndFeel.getRegion((JComponent) c);
133 
134             if ((region == Region.ARROW_BUTTON || region == Region.BUTTON || region == Region.CHECK_BOX
135                         || region == Region.CHECK_BOX_MENU_ITEM || region == Region.MENU || region == Region.MENU_ITEM
136                         || region == Region.RADIO_BUTTON || region == Region.RADIO_BUTTON_MENU_ITEM || region == Region.TOGGLE_BUTTON)
137                     && (c instanceof AbstractButton)) {
138                 margin = ((AbstractButton) c).getMargin();
139             } else if ((region == Region.EDITOR_PANE || region == Region.FORMATTED_TEXT_FIELD || region == Region.PASSWORD_FIELD
140                         || region == Region.TEXT_AREA || region == Region.TEXT_FIELD || region == Region.TEXT_PANE)
141                     && (c instanceof JTextComponent)) {
142                 margin = ((JTextComponent) c).getMargin();
143             } else if (region == Region.TOOL_BAR && (c instanceof JToolBar)) {
144                 margin = ((JToolBar) c).getMargin();
145             } else if (region == Region.MENU_BAR && (c instanceof JMenuBar)) {
146                 margin = ((JMenuBar) c).getMargin();
147             }
148 
149             if (margin != null) {
150                 insets.top    += margin.top;
151                 insets.bottom += margin.bottom;
152                 insets.left   += margin.left;
153                 insets.right  += margin.right;
154             }
155         }
156 
157         return insets;
158     }

160     /**
161      * This default implementation returns false.
162      *
163      * @return false
164      */
165     public boolean isBorderOpaque() {
166         return false;
167     }
