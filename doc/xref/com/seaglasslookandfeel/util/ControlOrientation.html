29  /**

33   * bar or tabbed pane tab area. A scroll bar, regardless of its orientation, has
34   * a length and thickness, as does the tab area of a tabbed pane. These values
35   * correspond to different dimensions (x or y) depending on the orientation.

196     /**
197      * Converts a Swing orientation (either {@link SwingConstants#HORIZONTAL} or
198      * {@link SwingConstants#VERTICAL} to a {@code ControlOrientation}.
199      *
200      * @param  swingScrollBarOrientation the Swing scroll bar orientation,
201      *                                   either
202      *                                   {@link SwingConstants#HORIZONTAL} or
203      *                                   {@link SwingConstants#VERTICAL}
204      *
205      * @return the {@code ControlOrientation} to the corresponding Swing scroll
206      *         bar orientation.
207      */
208     public static ControlOrientation getOrientation(int swingScrollBarOrientation) {
209         if (swingScrollBarOrientation != SwingConstants.HORIZONTAL && swingScrollBarOrientation != SwingConstants.VERTICAL) {
210             throw new IllegalArgumentException("The given value is not a valid orientation for ControlOrientation.");
211         }
212 
213         return swingScrollBarOrientation == SwingConstants.HORIZONTAL ? HORIZONTAL : VERTICAL;
214     }
