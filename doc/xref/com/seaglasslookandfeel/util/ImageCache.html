35  /**
36   * ImageCache - A fixed pixel count sized cache of Images keyed by arbitrary set
37   * of arguments. All images are held with SoftReferences so they will be dropped
38   * by the GC if heap memory gets tight. When our size hits max pixel count least
39   * recently requested images are removed first.
40   *
41   * <p>Based on Nimbus's ImageCache by Jasper Potts. This was package local.</p>
42   *
43   * @see com.sun.java.swing.plaf.nimbus.ImageCache
44   */
45  public class ImageCache {
46  
47      // Singleton Instance
48      private static final ImageCache instance = new ImageCache();
49  
50      // Ordered Map keyed by args hash, ordered by most recent accessed entry.
51      private final LinkedHashMap<Integer, PixelCountSoftReference> map = new LinkedHashMap<Integer, PixelCountSoftReference>(16, 0.75f,
52                                                                                                                              true);
53  
54      // Maximum number of pixels to cache, this is used if maxCount
55      private final int maxPixelCount;
56  
57      // Maximum cached image size in pxiels
58      private final int maxSingleImagePixelSize;
59  
60      // The current number of pixels stored in the cache
61      private int currentPixelCount = 0;
62  
63      // Lock for concurrent access to map
64      private ReadWriteLock lock = new ReentrantReadWriteLock();
65  
66      // Reference queue for tracking lost softreferences to images in the cache
67      private ReferenceQueue<Image> referenceQueue = new ReferenceQueue<Image>();
68  
69      /**
70       * Creates a new ImageCache object.
71       */
72      public ImageCache() {
73          this.maxPixelCount           = (8 * 1024 * 1024) / 4; // 8Mb of pixels
74          this.maxSingleImagePixelSize = 300 * 300;
75      }
76  
77      /**
78       * Creates a new ImageCache object.
79       *
80       * @param maxPixelCount           the maximum pixel count.
81       * @param maxSingleImagePixelSize the maximum single image pixel size.
82       */
83      public ImageCache(int maxPixelCount, int maxSingleImagePixelSize) {
84          this.maxPixelCount           = maxPixelCount;
85          this.maxSingleImagePixelSize = maxSingleImagePixelSize;
86      }

88      /**
89       * Get static singleton instance.
90       *
91       * @return the singleton image cache.
92       */
93      public static ImageCache getInstance() {
94          return instance;
95      }

97      /**
98       * Clear the cache.
99       */
100     public void flush() {
101         lock.readLock().lock();
102 
103         try {
104             map.clear();
105         } finally {
106             lock.readLock().unlock();
107         }
108     }

110     /**
111      * Check if the image size is to big to be stored in the cache.
112      *
113      * @param  w The image width
114      * @param  h The image height
115      *
116      * @return True if the image size is less than max
117      */
118     public boolean isImageCachable(int w, int h) {
119         return (w * h) < maxSingleImagePixelSize;
120     }
121 
122     /**
123      * Get the cached image for given keys
124      *
125      * @param  config The graphics configuration, needed if cached image is a
126      *                Volatile Image. Used as part of cache key
127      * @param  w      The image width, used as part of cache key
128      * @param  h      The image height, used as part of cache key
129      * @param  args   Other arguments to use as part of the cache key
130      *
131      * @return Returns the cached Image, or null there is no cached image for
132      *         key
133      */
134     public Image getImage(GraphicsConfiguration config, int w, int h, Object... args) {
135         lock.readLock().lock();
136 
137         try {
138             PixelCountSoftReference ref = map.get(hash(config, w, h, args));
139 
140             // check reference has not been lost and the key truly matches, in
141             // case of false positive hash match
142             if (ref != null && ref.equals(config, w, h, args)) {
143                 return ref.get();
144             } else {
145                 return null;
146             }
147         } finally {
148             lock.readLock().unlock();
149         }
150     }
151 
152     /**
153      * Sets the cached image for the specified constraints.
154      *
155      * @param  image  The image to store in cache
156      * @param  config The graphics configuration, needed if cached image is a
157      *                Volatile Image. Used as part of cache key
158      * @param  w      The image width, used as part of cache key
159      * @param  h      The image height, used as part of cache key
160      * @param  args   Other arguments to use as part of the cache key
161      *
162      * @return true if the image could be cached or false if the image is too
163      *         big
164      */
165     public boolean setImage(Image image, GraphicsConfiguration config, int w, int h, Object... args) {
166         if (!isImageCachable(w, h))
167             return false;
168 
169         int hash = hash(config, w, h, args);
170 
171         lock.writeLock().lock();
172 
173         try {
174             PixelCountSoftReference ref = map.get(hash);
175 
176             // check if currently in map
177             if (ref != null && ref.get() == image) {
178                 return true;
179             }
180 
181             // clear out old
182             if (ref != null) {
183                 currentPixelCount -= ref.pixelCount;
184                 map.remove(hash);
185             }
186 
187             // add new image to pixel count
188             int newPixelCount = image.getWidth(null) * image.getHeight(null);
189 
190             currentPixelCount += newPixelCount;
191 
192             // clean out lost references if not enough space
193             if (currentPixelCount > maxPixelCount) {
194 
195                 while ((ref = (PixelCountSoftReference) referenceQueue.poll()) != null) {
196 
197                     // reference lost
198                     map.remove(ref.hash);
199                     currentPixelCount -= ref.pixelCount;
200                 }
201             }
202 
203             // remove old items till there is enough free space
204             if (currentPixelCount > maxPixelCount) {
205                 Iterator<Map.Entry<Integer, PixelCountSoftReference>> mapIter = map.entrySet().iterator();
206 
207                 while ((currentPixelCount > maxPixelCount) && mapIter.hasNext()) {
208                     Map.Entry<Integer, PixelCountSoftReference> entry = mapIter.next();
209 
210                     mapIter.remove();
211                     Image img = entry.getValue().get();
212 
213                     if (img != null)
214                         img.flush();
215 
216                     currentPixelCount -= entry.getValue().pixelCount;
217                 }
218             }
219 
220             // finaly put new in map
221             map.put(hash, new PixelCountSoftReference(image, referenceQueue, newPixelCount, hash, config, w, h, args));
222 
223             return true;
224         } finally {
225             lock.writeLock().unlock();
226         }
227     }

229     /**
230      * Create a unique hash from all the input.
231      *
232      * @param  config the graphics configuration.
233      * @param  w      the width.
234      * @param  h      the height.
235      * @param  args   the arguments for the image.
236      *
237      * @return the hash for the object instance.
238      */
239     private int hash(GraphicsConfiguration config, int w, int h, Object... args) {
240         int hash;
241 
242         hash = (config != null ? config.hashCode() : 0);
243         hash = 31 * hash + w;
244         hash = 31 * hash + h;
245         hash = 31 * hash + Arrays.deepHashCode(args);
246 
247         return hash;
248     }

250     /**
251      * Extended SoftReference that stores the pixel count even after the image
252      * is lost.
253      */
254     private static class PixelCountSoftReference extends SoftReference<Image> {
255         private final int pixelCount;
256         private final int hash;
257 
258         // key parts
259         private final GraphicsConfiguration config;
260         private final int                   w;
261         private final int                   h;
262         private final Object[]              args;
263 
264         /**
265          * Creates a new PixelCountSoftReference object.
266          *
267          * @param referent   the image.
268          * @param q          the reference queue.
269          * @param pixelCount the pixel count.
270          * @param hash       the hash.
271          * @param config     the graphics configuration.
272          * @param w          the width.
273          * @param h          the height.
274          * @param args       the arguments for the image.
275          */
276         public PixelCountSoftReference(Image referent, ReferenceQueue<? super Image> q, int pixelCount, int hash,
277                 GraphicsConfiguration config, int w, int h, Object[] args) {
278             super(referent, q);
279             this.pixelCount = pixelCount;
280             this.hash       = hash;
281             this.config     = config;
282             this.w          = w;
283             this.h          = h;
284             this.args       = args;
285         }
286 
287         /**
288          * Tests for equality.
289          *
290          * @param  config the graphics configuration.
291          * @param  w      the width.
292          * @param  h      the height.
293          * @param  args   the arguments for the image.
294          *
295          * @return {@code true} if the arguments describe the same object as
296          *         this one, {@code false} otherwise.
297          */
298         public boolean equals(GraphicsConfiguration config, int w, int h, Object[] args) {
299             return config == this.config && w == this.w && h == this.h && Arrays.equals(args, this.args);
300         }
301     }
