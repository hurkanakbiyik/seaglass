80      public void updateStyle(AbstractButton b) {
81          SeaGlassContext context = getContext(b, SynthConstants.ENABLED);
82          SynthStyle oldStyle = style;
83          style = SeaGlassLookAndFeel.updateStyle(context, this);
84          if (style != oldStyle) {
85              if (b.getMargin() == null || (b.getMargin() instanceof UIResource)) {
86                  Insets margin = (Insets) style.get(context, getPropertyPrefix() + "margin");
87  
88                  if (margin == null) {
89                      // Some places assume margins are non-null.
90                      margin = SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS;
91                  }
92                  b.setMargin(margin);
93              }
94  
95              Object value = style.get(context, getPropertyPrefix() + "iconTextGap");
96              if (value != null) {
97                  LookAndFeel.installProperty(b, "iconTextGap", value);
98              }
99  
100             value = style.get(context, getPropertyPrefix() + "contentAreaFilled");
101             LookAndFeel.installProperty(b, "contentAreaFilled", value != null ? value : Boolean.TRUE);
102 
103             value = b.getClientProperty(APPLE_PREFIX + "buttonType");
104             if (value != null) {
105                 if ("segmented".equals(value)) {
106                     b.setMargin(SeaGlassLookAndFeel.EMPTY_UIRESOURCE_INSETS);
107                 }
108             }
109 
110             if (oldStyle != null) {
111                 uninstallKeyboardActions(b);
112                 installKeyboardActions(b);
113             }
114         }
115         context.dispose();
116     }

144     /**
145      * Returns the current state of the passed in <code>AbstractButton</code>.
146      */
147     private int getComponentState(JComponent c) {
148         int state = ENABLED;
149 
150         if (!c.isEnabled()) {
151             state = DISABLED;
152         }
153         if (SeaGlassLookAndFeel.selectedUI == this) {
154             return SeaGlassLookAndFeel.selectedUIState | SynthConstants.ENABLED;
155         }
156         AbstractButton button = (AbstractButton) c;
157         ButtonModel model = button.getModel();
158 
159         if (model.isPressed()) {
160             if (model.isArmed()) {
161                 state = PRESSED;
162             } else {
163                 state = MOUSE_OVER;
164             }
165         }
166         if (model.isRollover()) {
167             state |= MOUSE_OVER;
168         }
169         if (model.isSelected()) {
170             state |= SELECTED;
171         }
172         if (c.isFocusOwner() && button.isFocusPainted()) {
173             state |= FOCUSED;
174         }
175         if ((c instanceof JButton) && ((JButton) c).isDefaultButton()) {
176             state |= DEFAULT;
177         }
178         return state;
179     }

181     public int getBaseline(JComponent c, int width, int height) {
182         if (c == null) {
183             throw new NullPointerException("Component must be non-null");
184         }
185         if (width < 0 || height < 0) {
186             throw new IllegalArgumentException("Width and height must be >= 0");
187         }
188         AbstractButton b = (AbstractButton) c;
189         String text = b.getText();
190         if (text == null || "".equals(text)) {
191             return -1;
192         }
193         Insets i = b.getInsets();
194         Rectangle viewRect = new Rectangle();
195         Rectangle textRect = new Rectangle();
196         Rectangle iconRect = new Rectangle();
197         viewRect.x = i.left;
198         viewRect.y = i.top;
199         viewRect.width = width - (i.right + viewRect.x);
200         viewRect.height = height - (i.bottom + viewRect.y);
201 
202         // layout the text and icon
203         SeaGlassContext context = getContext(b);
204         FontMetrics fm = context.getComponent().getFontMetrics(context.getStyle().getFont(context));
205         context.getStyle().getGraphicsUtils(context).layoutText(context, fm, b.getText(), b.getIcon(), b.getHorizontalAlignment(),
206             b.getVerticalAlignment(), b.getHorizontalTextPosition(), b.getVerticalTextPosition(), viewRect, iconRect, textRect,
207             b.getIconTextGap());
208         View view = (View) b.getClientProperty(BasicHTML.propertyKey);
209         int baseline;
210         if (view != null) {
211             baseline = BasicHTML.getHTMLBaseline(view, textRect.width, textRect.height);
212             if (baseline >= 0) {
213                 baseline += textRect.y;
214             }
215         } else {
216             baseline = textRect.y + fm.getAscent();
217         }
218         context.dispose();
219         return baseline;
220     }

241     protected void paint(SeaGlassContext context, Graphics g) {
242         AbstractButton b = (AbstractButton) context.getComponent();
243 
244         g.setColor(context.getStyle().getColor(context, ColorType.TEXT_FOREGROUND));
245         g.setFont(style.getFont(context));
246         context.getStyle().getGraphicsUtils(context).paintText(context, g, b.getText(), getIcon(b), b.getHorizontalAlignment(),
247             b.getVerticalAlignment(), b.getHorizontalTextPosition(), b.getVerticalTextPosition(), b.getIconTextGap(),
248             b.getDisplayedMnemonicIndex(), getTextShiftOffset(context));

276     /**
277      * Returns the Icon to use in painting the button.
278      */
279     protected Icon getIcon(AbstractButton b) {
280         Icon icon = b.getIcon();
281         ButtonModel model = b.getModel();
282 
283         if (!model.isEnabled()) {
284             icon = getSynthDisabledIcon(b, icon);
285         } else if (model.isPressed() && model.isArmed()) {
286             icon = getPressedIcon(b, getSelectedIcon(b, icon));
287         } else if (b.isRolloverEnabled() && model.isRollover()) {
288             icon = getRolloverIcon(b, getSelectedIcon(b, icon));
289         } else if (model.isSelected()) {
290             icon = getSelectedIcon(b, icon);
291         } else {
292             icon = getEnabledIcon(b, icon);
293         }
294         if (icon == null) {
295             return getDefaultIcon(b);
296         }
297         return icon;
298     }

300     /**
301      * This method will return the icon that should be used for a button. We
302      * only want to use the synth icon defined by the style if the specific icon
303      * has not been defined for the button state and the backup icon is a
304      * UIResource (we set it, not the developer).
305      * 
306      * @param b
307      *            button
308      * @param specificIcon
309      *            icon returned from the button for the specific state
310      * @param defaultIcon
311      *            fallback icon
312      * @param state
313      *            the synth state of the button
314      */
315     private Icon getIcon(AbstractButton b, Icon specificIcon, Icon defaultIcon, int state) {
316         Icon icon = specificIcon;
317         if (icon == null) {
318             if (defaultIcon instanceof UIResource) {
319                 icon = getSynthIcon(b, state);
320                 if (icon == null) {
321                     icon = defaultIcon;
322                 }
323             } else {
324                 icon = defaultIcon;
325             }
326         }
327         return icon;
328     }

375         AbstractButton button = (AbstractButton) state.getComponent();
376         ButtonModel model = button.getModel();

378         if (model.isArmed() && model.isPressed() && button.getPressedIcon() == null) {

432     /**
433      * Returns the Icon used in calculating the pref/min/max size.
434      */
435     protected Icon getSizingIcon(AbstractButton b) {
436         // NOTE: this is slightly different than BasicButtonUI, where it
437         // would just use getIcon, but this should be ok.
438         Icon icon = (b.isEnabled() || b.getDisabledIcon() == null) ? b.getIcon() : b.getDisabledIcon();
439         if (icon == null) {
440             icon = getDefaultIcon(b);
441         }
442         return icon;
