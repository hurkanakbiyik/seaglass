150     private void updateStyle(JTable c) {
151         SeaGlassContext context = getContext(c, ENABLED);
152         SynthStyle oldStyle = style;
153         style = SeaGlassLookAndFeel.updateStyle(context, this);
154 
155         selectionActiveBottomBorderColor = UIManager.getColor("seaGlassTableSelectionActiveBottom");
156         selectionInactiveBottomBorderColor = UIManager.getColor("seaGlassTableSelectionInactiveBottom");
157         transparentColor = UIManager.getColor("seaGlassTransparent");
158 
159         if (style != oldStyle) {
160             table.remove(rendererPane);
161             rendererPane = createCustomCellRendererPane();
162             table.add(rendererPane);
163 
164             context.setComponentState(ENABLED | SELECTED);
165 
166             Color sbg = table.getSelectionBackground();
167             if (sbg == null || sbg instanceof UIResource) {
168                 table.setSelectionBackground(style.getColor(context, ColorType.TEXT_BACKGROUND));
169             }
170 
171             Color sfg = table.getSelectionForeground();
172             if (sfg == null || sfg instanceof UIResource) {
173                 table.setSelectionForeground(style.getColor(context, ColorType.TEXT_FOREGROUND));
174             }
175 
176             context.setComponentState(ENABLED);
177 
178             Color gridColor = table.getGridColor();
179             if (gridColor == null || gridColor instanceof UIResource) {
180                 gridColor = (Color) style.get(context, "Table.gridColor");
181                 if (gridColor == null) {
182                     gridColor = style.getColor(context, ColorType.FOREGROUND);
183                 }
184                 table.setGridColor(gridColor);
185             }
186 
187             useTableColors = style.getBoolean(context, "Table.rendererUseTableColors", true);
188             useUIBorder = style.getBoolean(context, "Table.rendererUseUIBorder", true);
189 
190             Object rowHeight = style.get(context, "Table.rowHeight");
191             if (rowHeight != null) {
192                 LookAndFeel.installProperty(table, "rowHeight", rowHeight);
193             }
194             boolean showGrid = style.getBoolean(context, "Table.showGrid", true);
195             if (!showGrid) {
196                 table.setShowGrid(false);
197             }
198             Dimension d = table.getIntercellSpacing();
199             // if (d == null || d instanceof UIResource) {
200             if (d != null) {
201                 d = (Dimension) style.get(context, "Table.intercellSpacing");
202             }
203             alternateColor = (Color) style.get(context, "Table.alternateRowColor");
204             if (d != null) {
205                 table.setIntercellSpacing(d);
206             }
207 
208             table.setOpaque(false);
209 
210             if (alternateColor != null) {
211                 table.setShowHorizontalLines(false);
212             }
213 
214             // Make header column extend the width of the viewport (if there is
215             // a viewport).
216             table.getTableHeader().setBorder(createTableHeaderEmptyColumnPainter(table));
217             setViewPortListeners(table);
218 
219             if (oldStyle != null) {
220                 uninstallKeyboardActions();
221                 installKeyboardActions();
222             }
223         }
224         context.dispose();
225     }

298     protected void paint(SeaGlassContext context, Graphics g) {
299         Rectangle clip = g.getClipBounds();
300 
301         Rectangle bounds = table.getBounds();
302         // Account for the fact that the graphics has already been translated
303         // into the table's bounds.
304         bounds.x = bounds.y = 0;
305 
306         // This check prevents us from painting the entire table when the clip
307         // doesn't intersect our bounds at all.
308         if (table.getRowCount() <= 0 || table.getColumnCount() <= 0 || !bounds.intersects(clip)) {
309             paintDropLines(context, g);
310             return;
311         }
312 
313         boolean ltr = table.getComponentOrientation().isLeftToRight();
314 
315         Point upperLeft = clip.getLocation();
316         if (!ltr) {
317             upperLeft.x++;
318         }
319 
320         Point lowerRight = new Point(clip.x + clip.width - (ltr ? 1 : 0), clip.y + clip.height);
321 
322         int rMin = table.rowAtPoint(upperLeft);
323         int rMax = table.rowAtPoint(lowerRight);
324         // This should never happen (as long as our bounds intersect the clip,
325         // which is why we bail above if that is the case).
326         if (rMin == -1) {
327             rMin = 0;
328         }
329         // If the table does not have enough rows to fill the view we'll get -1.
330         // (We could also get -1 if our bounds don't intersect the clip,
331         // which is why we bail above if that is the case).
332         // Replace this with the index of the last row.
333         if (rMax == -1) {
334             rMax = table.getRowCount() - 1;
335         }
336 
337         int cMin = table.columnAtPoint(ltr ? upperLeft : lowerRight);
338         int cMax = table.columnAtPoint(ltr ? lowerRight : upperLeft);
339         // This should never happen.
340         if (cMin == -1) {
341             cMin = 0;
342         }
343         // If the table does not have enough columns to fill the view we'll get
344         // -1.
345         // Replace this with the index of the last column.
346         if (cMax == -1) {
347             cMax = table.getColumnCount() - 1;
348         }
349 
350         // Paint the grid.
351         if (!(table.getParent() instanceof JViewport)
352                 || (table.getParent() != null && !(table.getParent().getParent() instanceof JScrollPane))) {
353             // FIXME We need to not repaint the entire table any time something
354             // changes.
355             // paintGrid(context, g, rMin, rMax, cMin, cMax);
356             paintStripesAndGrid(context, g, table, table.getWidth(), table.getHeight(), 0);
357         }
358 
359         // Paint the cells.
360         paintCells(context, g, rMin, rMax, cMin, cMax);
361 
362         paintDropLines(context, g);
363     }

369     public void paintStripesAndGrid(SeaGlassContext context, Graphics g, JComponent c, int width, int height, int top) {
370         int rh = table.getRowHeight();
371         int n = table.getRowCount();
372         int row = Math.abs(top / rh);
373 
374         // Paint the background, including stripes if requested.
375         if (alternateColor != null) {
376             // Fill the viewport with background color.
377             g.setColor(table.getBackground());
378             g.fillRect(0, 0, width, height);
379 
380             // Now check if we need to paint some stripes
381             g.setColor(alternateColor);
382 
383             // Paint table rows to fill the viewport.
384             for (int y = top + row * rh, ymax = height; y < ymax; y += rh) {
385                 if (row % 2 == 0) {
386                     g.fillRect(0, y, width, rh);
387                 }
388                 row++;
389             }
390         } else {
391             // Fill the viewport with the background color of the table
392             g.setColor(table.getBackground());
393             g.fillRect(0, 0, c.getWidth(), c.getHeight());
394         }
395 
396         SynthGraphicsUtils synthG = context.getStyle().getGraphicsUtils(context);
397 
398         // Paint the horizontal grid lines
399         if (table.getShowHorizontalLines()) {
400             g.setColor(table.getGridColor());
401             row = Math.abs(top / rh);
402             int y = top + row * rh + rh - 1;
403             while (y < height) {
404                 synthG.drawLine(context, "Table.grid", g, 0, y, width, y);
405                 y += rh;
406             }
407         }
408 
409         // Paint the vertical grid lines
410         if (table.getShowVerticalLines()) {
411             g.setColor(table.getGridColor());
412             TableColumnModel cm = table.getColumnModel();
413             n = cm.getColumnCount();
414             int y = top + row * rh;
415             ;
416             int x = -1;
417             for (int i = 0; i < n; i++) {
418                 TableColumn col = cm.getColumn(i);
419                 x += col.getWidth();
420                 synthG.drawLine(context, "Table.grid", g, x, y, x, height);
421             }
422         }
423     }

425     private void paintDropLines(SeaGlassContext context, Graphics g) {
426         JTable.DropLocation loc = table.getDropLocation();
427         if (loc == null) {
428             return;
429         }
430 
431         Color color = (Color) style.get(context, "Table.dropLineColor");
432         Color shortColor = (Color) style.get(context, "Table.dropLineShortColor");
433         if (color == null && shortColor == null) {
434             return;
435         }
436 
437         Rectangle rect;
438 
439         rect = getHDropLineRect(loc);
440         if (rect != null) {
441             int x = rect.x;
442             int w = rect.width;
443             if (color != null) {
444                 extendRect(rect, true);
445                 g.setColor(color);
446                 g.fillRect(rect.x, rect.y, rect.width, rect.height);
447             }
448             if (!loc.isInsertColumn() && shortColor != null) {
449                 g.setColor(shortColor);
450                 g.fillRect(x, rect.y, w, rect.height);
451             }
452         }
453 
454         rect = getVDropLineRect(loc);
455         if (rect != null) {
456             int y = rect.y;
457             int h = rect.height;
458             if (color != null) {
459                 extendRect(rect, false);
460                 g.setColor(color);
461                 g.fillRect(rect.x, rect.y, rect.width, rect.height);
462             }
463             if (!loc.isInsertRow() && shortColor != null) {
464                 g.setColor(shortColor);
465                 g.fillRect(rect.x, y, rect.width, h);
466             }
467         }
468     }

470     private Rectangle getHDropLineRect(JTable.DropLocation loc) {
471         if (!loc.isInsertRow()) {
472             return null;
473         }
474 
475         int row = loc.getRow();
476         int col = loc.getColumn();
477         if (col >= table.getColumnCount()) {
478             col--;
479         }
480 
481         Rectangle rect = table.getCellRect(row, col, true);
482 
483         if (row >= table.getRowCount()) {
484             row--;
485             Rectangle prevRect = table.getCellRect(row, col, true);
486             rect.y = prevRect.y + prevRect.height;
487         }
488 
489         if (rect.y == 0) {
490             rect.y = -1;
491         } else {
492             rect.y -= 2;
493         }
494 
495         rect.height = 3;
496 
497         return rect;
498     }

500     private Rectangle getVDropLineRect(JTable.DropLocation loc) {
501         if (!loc.isInsertColumn()) {
502             return null;
503         }
504 
505         boolean ltr = table.getComponentOrientation().isLeftToRight();
506         int col = loc.getColumn();
507         Rectangle rect = table.getCellRect(loc.getRow(), col, true);
508 
509         if (col >= table.getColumnCount()) {
510             col--;
511             rect = table.getCellRect(loc.getRow(), col, true);
512             if (ltr) {
513                 rect.x = rect.x + rect.width;
514             }
515         } else if (!ltr) {
516             rect.x = rect.x + rect.width;
517         }
518 
519         if (rect.x == 0) {
520             rect.x = -1;
521         } else {
522             rect.x -= 2;
523         }
524 
525         rect.width = 3;
526 
527         return rect;
528     }

530     private Rectangle extendRect(Rectangle rect, boolean horizontal) {
531         if (rect == null) {
532             return rect;
533         }
534 
535         if (horizontal) {
536             rect.x = 0;
537             rect.width = table.getWidth();
538         } else {
539             rect.y = 0;
540 
541             if (table.getRowCount() != 0) {
542                 Rectangle lastRect = table.getCellRect(table.getRowCount() - 1, 0, true);
543                 rect.height = lastRect.y + lastRect.height;
544             } else {
545                 rect.height = table.getHeight();
546             }
547         }
548 
549         return rect;
550     }

552     /*
553      * Paints the grid lines within <I>aRect</I>, using the grid color set with
554      * <I>setGridColor</I>. Paints vertical lines if
555      * <code>getShowVerticalLines()</code> returns true and paints horizontal
556      * lines if <code>getShowHorizontalLines()</code> returns true.
557      * 
558      * TODO See if we want to remove this method.
559      */
560     @SuppressWarnings("unused")
561     private void paintGrid(SeaGlassContext context, Graphics g, int rMin, int rMax, int cMin, int cMax) {
562         g.setColor(table.getGridColor());
563 
564         Rectangle minCell = table.getCellRect(rMin, cMin, true);
565         Rectangle maxCell = table.getCellRect(rMax, cMax, true);
566         Rectangle damagedArea = minCell.union(maxCell);
567         SynthGraphicsUtils synthG = context.getStyle().getGraphicsUtils(context);
568 
569         if (table.getShowHorizontalLines()) {
570             int tableWidth = damagedArea.x + damagedArea.width;
571             int y = damagedArea.y;
572             for (int row = rMin; row <= rMax; row++) {
573                 y += table.getRowHeight(row);
574                 synthG.drawLine(context, "Table.grid", g, damagedArea.x, y - 1, tableWidth - 1, y - 1);
575             }
576         }
577         if (table.getShowVerticalLines()) {
578             TableColumnModel cm = table.getColumnModel();
579             int tableHeight = damagedArea.y + damagedArea.height;
580             int x;
581             if (table.getComponentOrientation().isLeftToRight()) {
582                 x = damagedArea.x;
583                 for (int column = cMin; column <= cMax; column++) {
584                     int w = cm.getColumn(column).getWidth();
585                     x += w;
586                     synthG.drawLine(context, "Table.grid", g, x - 1, 0, x - 1, tableHeight - 1);
587                 }
588             } else {
589                 x = damagedArea.x;
590                 for (int column = cMax; column >= cMin; column--) {
591                     int w = cm.getColumn(column).getWidth();
592                     x += w;
593                     synthG.drawLine(context, "Table.grid", g, x - 1, 0, x - 1, tableHeight - 1);
594                 }
595             }
596         }
597     }

599     private int viewIndexForColumn(TableColumn aColumn) {
600         TableColumnModel cm = table.getColumnModel();
601         for (int column = 0; column < cm.getColumnCount(); column++) {
602             if (cm.getColumn(column) == aColumn) {
603                 return column;
604             }
605         }
606         return -1;
607     }

609     private void paintCells(SeaGlassContext context, Graphics g, int rMin, int rMax, int cMin, int cMax) {
610         JTableHeader header = table.getTableHeader();
611         TableColumn draggedColumn = (header == null) ? null : header.getDraggedColumn();
612 
613         TableColumnModel cm = table.getColumnModel();
614         int columnMargin = cm.getColumnMargin();
615 
616         Rectangle cellRect;
617         TableColumn aColumn;
618         int columnWidth;
619         if (table.getComponentOrientation().isLeftToRight()) {
620             for (int row = rMin; row <= rMax; row++) {
621                 cellRect = table.getCellRect(row, cMin, false);
622                 for (int column = cMin; column <= cMax; column++) {
623                     aColumn = cm.getColumn(column);
624                     columnWidth = aColumn.getWidth();
625                     cellRect.width = columnWidth - columnMargin;
626                     if (aColumn != draggedColumn) {
627                         paintCell(context, g, cellRect, row, column);
628                     }
629                     cellRect.x += columnWidth;
630                 }
631             }
632         } else {
633             for (int row = rMin; row <= rMax; row++) {
634                 cellRect = table.getCellRect(row, cMin, false);
635                 aColumn = cm.getColumn(cMin);
636                 if (aColumn != draggedColumn) {
637                     columnWidth = aColumn.getWidth();
638                     cellRect.width = columnWidth - columnMargin;
639                     paintCell(context, g, cellRect, row, cMin);
640                 }
641                 for (int column = cMin + 1; column <= cMax; column++) {
642                     aColumn = cm.getColumn(column);
643                     columnWidth = aColumn.getWidth();
644                     cellRect.width = columnWidth - columnMargin;
645                     cellRect.x -= columnWidth;
646                     if (aColumn != draggedColumn) {
647                         paintCell(context, g, cellRect, row, column);
648                     }
649                 }
650             }
651         }
652 
653         // Paint the dragged column if we are dragging.
654         if (draggedColumn != null) {
655             paintDraggedArea(context, g, rMin, rMax, draggedColumn, header.getDraggedDistance());
656         }
657 
658         // Remove any renderers that may be left in the rendererPane.
659         rendererPane.removeAll();
660     }

662     private void paintDraggedArea(SeaGlassContext context, Graphics g, int rMin, int rMax, TableColumn draggedColumn, int distance) {
663         int draggedColumnIndex = viewIndexForColumn(draggedColumn);
664 
665         Rectangle minCell = table.getCellRect(rMin, draggedColumnIndex, true);
666         Rectangle maxCell = table.getCellRect(rMax, draggedColumnIndex, true);
667 
668         Rectangle vacatedColumnRect = minCell.union(maxCell);
669 
670         // Paint a gray well in place of the moving column.
671         g.setColor(table.getParent().getBackground());
672         g.fillRect(vacatedColumnRect.x, vacatedColumnRect.y, vacatedColumnRect.width, vacatedColumnRect.height);
673 
674         // Move to the where the cell has been dragged.
675         vacatedColumnRect.x += distance;
676 
677         // Fill the background.
678         g.setColor(context.getStyle().getColor(context, ColorType.BACKGROUND));
679         g.fillRect(vacatedColumnRect.x, vacatedColumnRect.y, vacatedColumnRect.width, vacatedColumnRect.height);
680 
681         SynthGraphicsUtils synthG = context.getStyle().getGraphicsUtils(context);
682 
683         // Paint the vertical grid lines if necessary.
684         if (table.getShowVerticalLines()) {
685             g.setColor(table.getGridColor());
686             int x1 = vacatedColumnRect.x;
687             int y1 = vacatedColumnRect.y;
688             int x2 = x1 + vacatedColumnRect.width - 1;
689             int y2 = y1 + vacatedColumnRect.height - 1;
690             // Left
691             synthG.drawLine(context, "Table.grid", g, x1 - 1, y1, x1 - 1, y2);
692             // Right
693             synthG.drawLine(context, "Table.grid", g, x2, y1, x2, y2);
694         }
695 
696         for (int row = rMin; row <= rMax; row++) {
697             // Render the cell value
698             Rectangle r = table.getCellRect(row, draggedColumnIndex, false);
699             r.x += distance;
700             paintCell(context, g, r, row, draggedColumnIndex);
701 
702             // Paint the (lower) horizontal grid line if necessary.
703             if (table.getShowHorizontalLines()) {
704                 g.setColor(table.getGridColor());
705                 Rectangle rcr = table.getCellRect(row, draggedColumnIndex, true);
706                 rcr.x += distance;
707                 int x1 = rcr.x;
708                 int y1 = rcr.y;
709                 int x2 = x1 + rcr.width - 1;
710                 int y2 = y1 + rcr.height - 1;
711                 synthG.drawLine(context, "Table.grid", g, x1, y2, x2, y2);
712             }
713         }
714     }
