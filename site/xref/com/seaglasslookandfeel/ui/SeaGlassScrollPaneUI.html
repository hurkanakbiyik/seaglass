105     protected void installDefaults(JScrollPane scrollpane) {
106         LookAndFeel.installBorder(scrollpane, "ScrollPane.border");
107         LookAndFeel.installColorsAndFont(scrollpane, "ScrollPane.background", "ScrollPane.foreground", "ScrollPane.font");
108 
109         Border vpBorder = scrollpane.getViewportBorder();
110         if ((vpBorder == null) || (vpBorder instanceof UIResource)) {
111             vpBorder = UIManager.getBorder("ScrollPane.viewportBorder");
112             scrollpane.setViewportBorder(vpBorder);
113         }
114 
115         Object obj = UIManager.get("ScrollPane.cornerPainter");
116         if (obj != null && obj instanceof Painter) {
117             cornerPainter = (Painter) obj;
118         }
119 
120         LookAndFeel.installProperty(scrollpane, "opaque", Boolean.TRUE);
121         updateStyle(scrollpane);
122     }

135     private void updateStyle(JScrollPane c) {
136         SeaGlassContext context = getContext(c, ENABLED);
137         SynthStyle oldStyle = style;
138 
139         style = SeaGlassLookAndFeel.updateStyle(context, this);
140         if (style != oldStyle) {
141             Border vpBorder = scrollpane.getViewportBorder();
142             if ((vpBorder == null) || (vpBorder instanceof UIResource)) {
143                 scrollpane.setViewportBorder(new ViewportBorder(context));
144             }
145             if (oldStyle != null) {
146                 uninstallKeyboardActions(c);
147                 installKeyboardActions(c);
148             }
149         }
150         context.dispose();
151     }

153     protected void installListeners(JScrollPane c) {
154         vsbChangeListener = createVSBChangeListener();
155         vsbPropertyChangeListener = createVSBPropertyChangeListener();
156         hsbChangeListener = createHSBChangeListener();
157         hsbPropertyChangeListener = createHSBPropertyChangeListener();
158         viewportChangeListener = createViewportChangeListener();
159         spPropertyChangeListener = createPropertyChangeListener();
160 
161         JViewport viewport = scrollpane.getViewport();
162         JScrollBar vsb = scrollpane.getVerticalScrollBar();
163         JScrollBar hsb = scrollpane.getHorizontalScrollBar();
164 
165         if (viewport != null) {
166             viewport.addChangeListener(viewportChangeListener);
167         }
168         if (vsb != null) {
169             vsb.getModel().addChangeListener(vsbChangeListener);
170             vsb.addPropertyChangeListener(vsbPropertyChangeListener);
171         }
172         if (hsb != null) {
173             hsb.getModel().addChangeListener(hsbChangeListener);
174             hsb.addPropertyChangeListener(hsbPropertyChangeListener);
175         }
176 
177         scrollpane.addPropertyChangeListener(spPropertyChangeListener);
178 
179         mouseScrollListener = createMouseWheelListener();
180         scrollpane.addMouseWheelListener(mouseScrollListener);
181 
182         // From SynthScrollPaneUI.
183         c.addPropertyChangeListener(this);
184         if (UIManager.getBoolean("ScrollPane.useChildTextComponentFocus")) {
185             viewportViewFocusHandler = new ViewportViewFocusHandler();
186             c.getViewport().addContainerListener(viewportViewFocusHandler);
187             Component view = c.getViewport().getView();
188             if (view instanceof JTextComponent) {
189                 view.addFocusListener(viewportViewFocusHandler);
190             }
191         }

194     protected void uninstallListeners(JComponent c) {
195         JViewport viewport = scrollpane.getViewport();
196         JScrollBar vsb = scrollpane.getVerticalScrollBar();
197         JScrollBar hsb = scrollpane.getHorizontalScrollBar();
198 
199         if (viewport != null) {
200             viewport.removeChangeListener(viewportChangeListener);
201         }
202         if (vsb != null) {
203             vsb.getModel().removeChangeListener(vsbChangeListener);
204             vsb.removePropertyChangeListener(vsbPropertyChangeListener);
205         }
206         if (hsb != null) {
207             hsb.getModel().removeChangeListener(hsbChangeListener);
208             hsb.removePropertyChangeListener(hsbPropertyChangeListener);
209         }
210 
211         scrollpane.removePropertyChangeListener(spPropertyChangeListener);
212 
213         if (mouseScrollListener != null) {
214             scrollpane.removeMouseWheelListener(mouseScrollListener);
215         }
216 
217         vsbChangeListener = null;
218         hsbChangeListener = null;
219         viewportChangeListener = null;
220         spPropertyChangeListener = null;
221         mouseScrollListener = null;
222         handler = null;
223 
224         // From SynthScrollPaneUI.
225         c.removePropertyChangeListener(this);
226         if (viewportViewFocusHandler != null) {
227             viewport.removeContainerListener(viewportViewFocusHandler);
228             if (viewport.getView() != null) {
229                 viewport.getView().removeFocusListener(viewportViewFocusHandler);
230             }
231             viewportViewFocusHandler = null;
232         }

268      * @param g
269      * @param c
270      */
271     private void paintScrollPaneCorner(Graphics g, JComponent c) {
272         if (!scrollpane.getHorizontalScrollBar().isVisible() || !scrollpane.getVerticalScrollBar().isVisible()) {
273             return;
274         }
275 
276         int vBarWidth = scrollpane.getVerticalScrollBar().getWidth();
277         int hBarHeight = scrollpane.getHorizontalScrollBar().getHeight();
278 
279         Insets insets = c.getInsets();
280 
281         Graphics2D g2 = (Graphics2D) g.create();
282         if (scrollpane.getComponentOrientation().isLeftToRight()) {
283             g2.translate(c.getWidth() - insets.right - vBarWidth, c.getHeight() - insets.bottom - hBarHeight);
284             g2.setClip(0, 0, vBarWidth, hBarHeight);
285         } else {
286             g2.translate(15 + insets.right, c.getHeight() - insets.bottom - hBarHeight);
287             g2.scale(-1, 1);
288             g2.setClip(0, 0, vBarWidth, hBarHeight);
289         }
290         cornerPainter.paint(g2, c, 15, 15);
291     }

319     protected void syncScrollPaneWithViewport() {
320         JViewport viewport = scrollpane.getViewport();
321         JScrollBar vsb = scrollpane.getVerticalScrollBar();
322         JScrollBar hsb = scrollpane.getHorizontalScrollBar();
323         JViewport rowHead = scrollpane.getRowHeader();
324         JViewport colHead = scrollpane.getColumnHeader();
325         boolean ltr = scrollpane.getComponentOrientation().isLeftToRight();
326 
327         if (viewport != null) {
328             Dimension extentSize = viewport.getExtentSize();
329             Dimension viewSize = viewport.getViewSize();
330             Point viewPosition = viewport.getViewPosition();
331 
332             if (vsb != null) {
333                 int extent = extentSize.height;
334                 int max = viewSize.height;
335                 int value = Math.max(0, Math.min(viewPosition.y, max - extent));
336                 vsb.setValues(value, extent, 0, max);
337             }
338 
339             if (hsb != null) {
340                 int extent = extentSize.width;
341                 int max = viewSize.width;
342                 int value;
343 
344                 if (ltr) {
345                     value = Math.max(0, Math.min(viewPosition.x, max - extent));
346                 } else {
347                     int currentValue = hsb.getValue();
348 
349                     /*
350                      * Use a particular formula to calculate "value" until
351                      * effective x coordinate is calculated.
352                      */
353                     if (setValueCalled && ((max - currentValue) == viewPosition.x)) {
354                         value = Math.max(0, Math.min(max - extent, currentValue));
355                         /*
356                          * After "extent" is set, turn setValueCalled flag off.
357                          */
358                         if (extent != 0) {
359                             setValueCalled = false;
360                         }
361                     } else {
362                         if (extent > max) {
363                             viewPosition.x = max - extent;
364                             viewport.setViewPosition(viewPosition);
365                             value = 0;
366                         } else {
367                             /*
368                              * The following line can't handle a small value of
369                              * viewPosition.x like Integer.MIN_VALUE correctly
370                              * because (max - extent - viewPositoiin.x) causes
371                              * an overflow. As a result, value becomes zero.
372                              * (e.g. setViewPosition(Integer.MAX_VALUE, ...) in
373                              * a user program causes a overflow. Its expected
374                              * value is (max - extent).) However, this seems a
375                              * trivial bug and adding a fix makes this
376                              * often-called method slow, so I'll leave it until
377                              * someone claims.
378                              */
379                             value = Math.max(0, Math.min(max - extent, max - extent - viewPosition.x));
380                         }
381                     }
382                 }
383                 hsb.setValues(value, extent, 0, max);
384             }
385 
386             if (rowHead != null) {
387                 Point p = rowHead.getViewPosition();
388                 p.y = viewport.getViewPosition().y;
389                 p.x = 0;
390                 rowHead.setViewPosition(p);
391             }
392 
393             if (colHead != null) {
394                 Point p = colHead.getViewPosition();
395                 if (ltr) {
396                     p.x = viewport.getViewPosition().x;
397                 } else {
398                     p.x = Math.max(0, viewport.getViewPosition().x);
399                 }
400                 p.y = 0;
401                 colHead.setViewPosition(p);
402             }
403         }
404     }

594     class Handler implements ChangeListener, PropertyChangeListener, MouseWheelListener {
595         //
596         // MouseWheelListener
597         //
598         public void mouseWheelMoved(MouseWheelEvent e) {
599             if (scrollpane.isWheelScrollingEnabled() && e.getWheelRotation() != 0) {
600                 boolean isHorizontal = (e.getModifiersEx() & MouseWheelEvent.SHIFT_DOWN_MASK) != 0;
601                 JScrollBar toScroll = scrollpane.getVerticalScrollBar();
602                 int direction = e.getWheelRotation() < 0 ? -1 : 1;
603                 int orientation = SwingConstants.VERTICAL;
604 
605                 // find which scrollbar to scroll, or return if none
606                 if (toScroll == null || !toScroll.isVisible() || isHorizontal) {
607                     toScroll = scrollpane.getHorizontalScrollBar();
608                     if (toScroll == null || !toScroll.isVisible()) {
609                         return;
610                     }
611                     orientation = SwingConstants.HORIZONTAL;
612                 }
613 
614                 if (e.getScrollType() == MouseWheelEvent.WHEEL_UNIT_SCROLL) {
615                     JViewport vp = scrollpane.getViewport();
616                     if (vp == null) {
617                         return;
618                     }
619                     Component comp = vp.getView();
620                     int units = Math.abs(e.getUnitsToScroll());
621 
622                     // When the scrolling speed is set to maximum, it's possible
623                     // for a single wheel click to scroll by more units than
624                     // will fit in the visible area. This makes it
625                     // hard/impossible to get to certain parts of the scrolling
626                     // Component with the wheel. To make for more accurate
627                     // low-speed scrolling, we limit scrolling to the block
628                     // increment if the wheel was only rotated one click.
629                     boolean limitScroll = Math.abs(e.getWheelRotation()) == 1;
630 
631                     // Check if we should use the visibleRect trick
632                     Object fastWheelScroll = toScroll.getClientProperty("JScrollBar.fastWheelScrolling");
633                     if (Boolean.TRUE == fastWheelScroll && comp instanceof Scrollable) {
634                         // 5078454: Under maximum acceleration, we may scroll
635                         // by many 100s of units in ~1 second.
636                         //
637                         // BasicScrollBarUI.scrollByUnits() can bog down the EDT
638                         // with repaints in this situation. However, the
639                         // Scrollable interface allows us to pass in an
640                         // arbitrary visibleRect. This allows us to accurately
641                         // calculate the total scroll amount, and then update
642                         // the GUI once. This technique provides much faster
643                         // accelerated wheel scrolling.
644                         Scrollable scrollComp = (Scrollable) comp;
645                         Rectangle viewRect = vp.getViewRect();
646                         int startingX = viewRect.x;
647                         boolean leftToRight = comp.getComponentOrientation().isLeftToRight();
648                         int scrollMin = toScroll.getMinimum();
649                         int scrollMax = toScroll.getMaximum() - toScroll.getModel().getExtent();
650 
651                         if (limitScroll) {
652                             int blockIncr = scrollComp.getScrollableBlockIncrement(viewRect, orientation, direction);
653                             if (direction < 0) {
654                                 scrollMin = Math.max(scrollMin, toScroll.getValue() - blockIncr);
655                             } else {
656                                 scrollMax = Math.min(scrollMax, toScroll.getValue() + blockIncr);
657                             }
658                         }
659 
660                         for (int i = 0; i < units; i++) {
661                             int unitIncr = scrollComp.getScrollableUnitIncrement(viewRect, orientation, direction);
662                             // Modify the visible rect for the next unit, and
663                             // check to see if we're at the end already.
664                             if (orientation == SwingConstants.VERTICAL) {
665                                 if (direction < 0) {
666                                     viewRect.y -= unitIncr;
667                                     if (viewRect.y <= scrollMin) {
668                                         viewRect.y = scrollMin;
669                                         break;
670                                     }
671                                 } else { // (direction > 0
672                                     viewRect.y += unitIncr;
673                                     if (viewRect.y >= scrollMax) {
674                                         viewRect.y = scrollMax;
675                                         break;
676                                     }
677                                 }
678                             } else {
679                                 // Scroll left
680                                 if ((leftToRight && direction < 0) || (!leftToRight && direction > 0)) {
681                                     viewRect.x -= unitIncr;
682                                     if (leftToRight) {
683                                         if (viewRect.x < scrollMin) {
684                                             viewRect.x = scrollMin;
685                                             break;
686                                         }
687                                     }
688                                 }
689                                 // Scroll right
690                                 else if ((leftToRight && direction > 0) || (!leftToRight && direction < 0)) {
691                                     viewRect.x += unitIncr;
692                                     if (leftToRight) {
693                                         if (viewRect.x > scrollMax) {
694                                             viewRect.x = scrollMax;
695                                             break;
696                                         }
697                                     }
698                                 } else {
699                                     assert false : "Non-sensical ComponentOrientation / scroll direction";
700                                 }
701                             }
702                         }
703                         // Set the final view position on the ScrollBar
704                         if (orientation == SwingConstants.VERTICAL) {
705                             toScroll.setValue(viewRect.y);
706                         } else {
707                             if (leftToRight) {
708                                 toScroll.setValue(viewRect.x);
709                             } else {
710                                 // rightToLeft scrollbars are oriented with
711                                 // minValue on the right and maxValue on the
712                                 // left.
713                                 int newPos = toScroll.getValue() - (viewRect.x - startingX);
714                                 if (newPos < scrollMin) {
715                                     newPos = scrollMin;
716                                 } else if (newPos > scrollMax) {
717                                     newPos = scrollMax;
718                                 }
719                                 toScroll.setValue(newPos);
720                             }
721                         }
722                     } else {
723                         // Viewport's view is not a Scrollable, or fast wheel
724                         // scrolling is not enabled.
725                         scrollByUnits(toScroll, direction, units, limitScroll);
726                     }
727                 } else if (e.getScrollType() == MouseWheelEvent.WHEEL_BLOCK_SCROLL) {
728                     scrollByBlock(toScroll, direction);
729                 }
730             }
731         }

733         /*
734          * Method for scrolling by a block increment. Added for mouse wheel
735          * scrolling support, RFE 4202656.
736          */
737         void scrollByBlock(JScrollBar scrollbar, int direction) {
738             // This method is called from BasicScrollPaneUI to implement wheel
739             // scrolling, and also from scrollByBlock().
740             int oldValue = scrollbar.getValue();
741             int blockIncrement = scrollbar.getBlockIncrement(direction);
742             int delta = blockIncrement * ((direction > 0) ? +1 : -1);
743             int newValue = oldValue + delta;
744 
745             // Check for overflow.
746             if (delta > 0 && newValue < oldValue) {
747                 newValue = scrollbar.getMaximum();
748             } else if (delta < 0 && newValue > oldValue) {
749                 newValue = scrollbar.getMinimum();
750             }
751 
752             scrollbar.setValue(newValue);
753         }
754 
755         /*
756          * Method for scrolling by a unit increment. Added for mouse wheel
757          * scrolling support, RFE 4202656.
758          * 
759          * If limitByBlock is set to true, the scrollbar will scroll at least 1
760          * unit increment, but will not scroll farther than the block increment.
761          * See BasicScrollPaneUI.Handler.mouseWheelMoved().
762          */
763         void scrollByUnits(JScrollBar scrollbar, int direction, int units, boolean limitToBlock) {
764             // This method is called from BasicScrollPaneUI to implement wheel
765             // scrolling, as well as from scrollByUnit().
766             int delta;
767             int limit = -1;
768 
769             if (limitToBlock) {
770                 if (direction < 0) {
771                     limit = scrollbar.getValue() - scrollbar.getBlockIncrement(direction);
772                 } else {
773                     limit = scrollbar.getValue() + scrollbar.getBlockIncrement(direction);
774                 }
775             }
776 
777             for (int i = 0; i < units; i++) {
778                 if (direction > 0) {
779                     delta = scrollbar.getUnitIncrement(direction);
780                 } else {
781                     delta = -scrollbar.getUnitIncrement(direction);
782                 }
783 
784                 int oldValue = scrollbar.getValue();
785                 int newValue = oldValue + delta;
786 
787                 // Check for overflow.
788                 if (delta > 0 && newValue < oldValue) {
789                     newValue = scrollbar.getMaximum();
790                 } else if (delta < 0 && newValue > oldValue) {
791                     newValue = scrollbar.getMinimum();
792                 }
793                 if (oldValue == newValue) {
794                     break;
795                 }
796 
797                 if (limitToBlock && i > 0) {
798                     assert limit != -1;
799                     if ((direction < 0 && newValue < limit) || (direction > 0 && newValue > limit)) {
800                         break;
801                     }
802                 }
803                 scrollbar.setValue(newValue);
804             }
805         }

837         private void hsbStateChanged(JViewport viewport, ChangeEvent e) {
838             BoundedRangeModel model = (BoundedRangeModel) (e.getSource());
839             Point p = viewport.getViewPosition();
840             int value = model.getValue();
841             if (scrollpane.getComponentOrientation().isLeftToRight()) {
842                 p.x = value;
843             } else {
844                 int max = viewport.getViewSize().width;
845                 int extent = viewport.getExtentSize().width;
846                 int oldX = p.x;
847 
848                 /*
849                  * Set new X coordinate based on "value".
850                  */
851                 p.x = max - extent - value;
852 
853                 /*
854                  * If setValue() was called before "extent" was fixed, turn
855                  * setValueCalled flag on.
856                  */
857                 if ((extent == 0) && (value != 0) && (oldX == max)) {
858                     setValueCalled = true;
859                 } else {
860                     /*
861                      * When a pane without a horizontal scroll bar was reduced
862                      * and the bar appeared, the viewport should show the right
863                      * side of the view.
864                      */
865                     if ((extent != 0) && (oldX < 0) && (p.x == 0)) {
866                         p.x += value;
867                     }
868                 }
869             }
870             viewport.setViewPosition(p);
