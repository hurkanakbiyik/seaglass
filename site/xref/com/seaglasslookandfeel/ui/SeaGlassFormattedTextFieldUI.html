22  import java.awt.Graphics;

24  import javax.swing.JComponent;
25  import javax.swing.plaf.ComponentUI;
26  import javax.swing.plaf.synth.SynthContext;
27  
28  import com.seaglasslookandfeel.SeaGlassContext;
29  
30  /**
31   * Sea Glass FormattedTextField UI delegate.
32   * 
33   * Based on SynthFormattedTextFieldUI, but needs to extend SeaGlassTextFieldUI
34   * instead.
35   */
36  public class SeaGlassFormattedTextFieldUI extends SeaGlassTextFieldUI {
37      /**
38       * Creates a UI for a JFormattedTextField.
39       * 
40       * @param c
41       *            the formatted text field
42       * @return the UI
43       */
44      public static ComponentUI createUI(JComponent c) {
45          return new SeaGlassFormattedTextFieldUI();
46      }
47  
48      /**
49       * Fetches the name used as a key to lookup properties through the
50       * UIManager. This is used as a prefix to all the standard text properties.
51       * 
52       * @return the name "FormattedTextField"
53       */
54      protected String getPropertyPrefix() {
55          return "FormattedTextField";
56      }
57  
58      void paintBackground(SeaGlassContext context, Graphics g, JComponent c) {
59          context.getPainter().paintFormattedTextFieldBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
60      }
61  
62      public void paintBorder(SynthContext context, Graphics g, int x, int y, int w, int h) {
63          ((SeaGlassContext) context).getPainter().paintFormattedTextFieldBorder(context, g, x, y, w, h);
64      }
65  }
