49      private static final int LEFT_SHIFT = 5;

51      private Color BORDER_COLOR = UIManager.getColor("seaGlassPopupBorder");

55      /**
56       * Creates a new SeaGlassComboPopup object.
57       *
58       * @param combo the combo box.
59       */
60      public SeaGlassComboPopup(JComboBox combo) {
61          super(combo);

64      /**
65       * Is the combo box popup a "drop down" popup, or does it overlay the box?
66       *
67       * @return {@code true} if the popup should drop down from the combo box,
68       *         {@code false} if it should overlay the combo box.
69       */
70      private boolean isDropDown() {
71          return comboBox.isEditable() || hasScrollBars();
72      }
73  
74      /**
75       * Does the combo box have scroll bars?
76       *
77       * @return {@code true} if the combo box popup has scroll bars,
78       *         {@code false} otherwise.
79       */
80      private boolean hasScrollBars() {
81          return comboBox.getModel().getSize() > getMaximumRowCount();
82      }
83  
84      /**
85       * Is the combo box editable?
86       *
87       * @return {@code true} if the combo box popup is editable, {@code false}
88       *         otherwise.
89       */
90      private boolean isEditable() {
91          return comboBox.isEditable();
92      }

94      /**
95       * Configures the list which is used to hold the combo box items in the
96       * popup. This method is called when the UI class is created.
97       */
98      @Override
99      protected void configureList() {
100         list.setFont(comboBox.getFont());
101 
102         list.setCellRenderer(comboBox.getRenderer());
103         list.setFocusable(false);
104         list.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
105         int selectedIndex = comboBox.getSelectedIndex();
106 
107         if (selectedIndex == -1) {
108             list.clearSelection();
109         } else {
110             list.setSelectedIndex(selectedIndex);
111             list.ensureIndexIsVisible(selectedIndex);
112         }
113 
114         installListListeners();
115     }

117     /**
118      * Configures the popup portion of the combo box. This method is called when
119      * the UI class is created.
120      */
121     protected void configurePopup() {
122         setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
123         setBorderPainted(true);
124         setBorder(LIST_BORDER);
125         setOpaque(false);
126         add(scroller);
127         setDoubleBuffered(true);

131     /**
132      * Calculate the placement and size of the popup portion of the combo box
133      * based on the combo box location and the enclosing screen bounds. If no
134      * transformations are required, then the returned rectangle will have the
135      * same values as the parameters.
136      *
137      * @param  px starting x location
138      * @param  py starting y location
139      * @param  pw starting width
140      * @param  ph starting height
141      *
142      * @return a rectangle which represents the placement and size of the popup
143      */
144     protected Rectangle computePopupBounds(int px, int py, int pw, int ph) {
145         Toolkit   toolkit      = Toolkit.getDefaultToolkit();
146         Rectangle screenBounds;
147         int       listWidth    = getList().getPreferredSize().width;
148         Insets    margin       = comboBox.getInsets();
149 
150         if (hasScrollBars()) {
151             px += margin.left;
152             pw = Math.max(pw - margin.left - margin.right, listWidth + 16);
153         } else {
154             px += margin.left;
155             pw = Math.max(pw - LEFT_SHIFT - margin.left, listWidth);
156         }
157 
158         // Calculate the desktop dimensions relative to the combo box.
159         GraphicsConfiguration gc = comboBox.getGraphicsConfiguration();
160         Point                 p  = new Point();
161 
162         SwingUtilities.convertPointFromScreen(p, comboBox);
163 
164         if (gc == null) {
165             screenBounds = new Rectangle(p, toolkit.getScreenSize());
166         } else {
167 
168             // Get the screen insets.
169             Insets screenInsets = Toolkit.getDefaultToolkit().getScreenInsets(gc);
170 
171             screenBounds        =  new Rectangle(gc.getBounds());
172             screenBounds.width  -= (screenInsets.left + screenInsets.right);
173             screenBounds.height -= (screenInsets.top + screenInsets.bottom);
174             screenBounds.x      += screenInsets.left;
175             screenBounds.y      += screenInsets.top;
176         }
177 
178         if (isDropDown()) {
179 
180             if (isEditable()) {
181                 py -= margin.bottom + 2;
182             } else {
183                 py -= margin.bottom;
184             }
185         } else {
186             int yOffset       = -margin.top;
187             int selectedIndex = comboBox.getSelectedIndex();
188 
189             if (selectedIndex <= 0) {
190                 py = -yOffset;
191             } else {
192                 py = -yOffset - list.getCellBounds(0, selectedIndex - 1).height;
193 
194             }
195         }
196 
197         // Compute the rectangle for the popup menu
198         Rectangle rect = new Rectangle(px, Math.max(py, p.y + screenBounds.y), Math.min(screenBounds.width, pw),
199                                        Math.min(screenBounds.height - 40, ph));
200 
201         // Add the preferred scroll bar width, if the popup does not fit
202         // on the available rectangle.
203         if (rect.height < ph) {
204             rect.width += 16;
205         }
206 
207         return rect;
208     }

210     /**
211      * Sets the list selection index to the selectedIndex. This method is used
212      * to synchronize the list selection with the combo box selection.
213      *
214      * @param selectedIndex the index to set the list
215      */
216     private void setListSelection(int selectedIndex) {
217         if (selectedIndex == -1) {
218             list.clearSelection();
219         } else {
220             list.setSelectedIndex(selectedIndex);
221             list.ensureIndexIsVisible(selectedIndex);
222         }
223     }

225     /**
226      * Implementation of ComboPopup.show().
227      */
228     public void show() {
229         setListSelection(comboBox.getSelectedIndex());
230 
231         Point location = getPopupLocation();
232 
233         show(comboBox, location.x, location.y);
234 
235         // For some reason, this is needed to clear the old selection being
236         // displayed. Calling list.clearSelection() doesn't cut it.
237         list.repaint();
238     }

240     /**
241      * Get the maximum row count to display. If none is set, return 100.
242      *
243      * @return the maximum row count to display.
244      */
245     private int getMaximumRowCount() {
246         return isEditable() ? comboBox.getMaximumRowCount() : 100;
247     }
248 
249     /**
250      * Calculates the upper left location of the Popup.
251      *
252      * @return the Point representing the upper-left coordinate of the Popup.
253      */
254     private Point getPopupLocation() {
255         Dimension popupSize = comboBox.getSize();
256         Insets    insets    = getInsets();
257 
258         // reduce the width of the scrollpane by the insets so that the popup
259         // is the same width as the combo box.
260         popupSize.setSize(popupSize.width - (insets.right + insets.left), getPopupHeightForRowCount(getMaximumRowCount()));
261         Rectangle popupBounds   = computePopupBounds(0, comboBox.getBounds().height, popupSize.width, popupSize.height);
262         Dimension scrollSize    = popupBounds.getSize();
263         Point     popupLocation = popupBounds.getLocation();
264 
265         scroller.setMaximumSize(scrollSize);
266         scroller.setPreferredSize(scrollSize);
267         scroller.setMinimumSize(scrollSize);
268 
269         list.revalidate();
270 
271         return popupLocation;
272     }
