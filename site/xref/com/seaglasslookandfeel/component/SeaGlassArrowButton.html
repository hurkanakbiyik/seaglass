177         /**
178          * @see com.seaglasslookandfeel.ui.SeaGlassButtonUI#getPreferredSize(javax.swing.JComponent)
179          */
180         public Dimension getPreferredSize(JComponent c) {
181             SeaGlassContext context = getContext(c);
182             Dimension       dim     = null;
183 
184             if (context.getComponent().getName() == "ScrollBar.button") {
185 
186                 // ScrollBar arrow buttons can be non-square when
187                 // the ScrollBar.squareButtons property is set to FALSE
188                 // and the ScrollBar.buttonSize property is non-null
189                 dim = (Dimension) context.getStyle().get(context, "ScrollBar.buttonSize");
190             }
191 
192             if (dim == null) {
193 
194                 // For all other cases (including Spinner, ComboBox), we will
195                 // fall back on the single ArrowButton.size value to create
196                 // a square return value
197                 int size = context.getStyle().getInt(context, "ArrowButton.size", 16);
198 
199                 dim = new Dimension(size, size);
200             }
201 
202             // handle scaling for sizeVarients for special case components. The
203             // key "JComponent.sizeVariant" scales for large/small/mini
204             // components are based on Apples LAF
205             JComponent parent = (JComponent) context.getComponent().getParent();
206 
207             if (parent != null && !(parent instanceof JComboBox)) {
208                 String scaleKey = (String) parent.getClientProperty("JComponent.sizeVariant");
209 
210                 if (scaleKey != null) {
211 
212                     if ("large".equals(scaleKey)) {
213                         dim = new Dimension((int) (dim.width * 1.15), (int) (dim.height * 1.15));
214                     } else if ("small".equals(scaleKey)) {
215                         dim = new Dimension((int) (dim.width * 0.857), (int) (dim.height * 0.857));
216                     } else if ("mini".equals(scaleKey)) {
217                         dim = new Dimension((int) (dim.width * 0.714), (int) (dim.height * 0.714));
218                     }
219                 }
220             }
221 
222             context.dispose();
223 
224             return dim;
225         }
