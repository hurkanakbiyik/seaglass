28  /**
29   * <p>Represents a built in, or custom, state in Sea Glass.</p>
30   *
31   * <p>Synth provides several built in states, which are:</p>
32   *
33   * <ul>
34   *   <li>Enabled</li>
35   *   <li>Mouse Over</li>
36   *   <li>Pressed</li>
37   *   <li>Disabled</li>
38   *   <li>Focused</li>
39   *   <li>Selected</li>
40   *   <li>Default</li>
41   * </ul>
42   *
43   * <p>However, there are many more states that could be described in a
44   * LookAndFeel, and it would be nice to style components differently based on
45   * these different states. For example, a progress bar could be "indeterminate".
46   * It would be very convenient to allow this to be defined as a "state".</p>
47   *
48   * <p>This class, State, is intended to be used for such situations. Simply
49   * implement the abstract #isInState method. It returns true if the given
50   * JComponent is "in this state", false otherwise. This method will be called
51   * <em>many</em> times in <em>performance sensitive loops</em>. It must execute
52   * very quickly.</p>
53   *
54   * <p>For example, the following might be an implementation of a custom
55   * "Indeterminate" state for JProgressBars:</p>
56   *
57   * <pre>
58   * &lt;code&gt;
59   *     public final class IndeterminateState extends State&lt;JProgressBar&gt; {
60   *         public IndeterminateState() {
61   *             super(&quot;Indeterminate&quot;);
62   *         }
63   *
64   *         &#064;Override
65   *         protected boolean isInState(JProgressBar c) {
66   *             return c.isIndeterminate();
67   *         }
68   *     }
69   * &lt;/code&gt;
70   * </pre>
71   *
72   * <p>Based on Nimbus's State class, which has too much package local stuff.</p>
73   *
74   * @see com.sun.java.swing.plaf.nimbus.State
75   */
76  public abstract class State<T extends JComponent> {
77      static final Map<String, StandardState> standardStates = new HashMap<String, StandardState>(7);
78

80      public static final State               Enabled   = new StandardState(SynthConstants.ENABLED);

85      /** Pressed state. */

88      /** Disabled state. */
89      public static final State               Disabled = new StandardState(SynthConstants.DISABLED);

94      /** Selected state. */
95      public static final State               Selected = new StandardState(SynthConstants.SELECTED);
96

98      public static final State               Default = new StandardState(SynthConstants.DEFAULT);

102     /**
103      * <p>Create a new custom State. Specify the name for the state. The name
104      * should be unique within the states set for any one particular component.
105      * The name of the state should coincide with the name used in UIDefaults.
106      * </p>
107      *
108      * <p>For example, the following would be correct:</p>
109      *
110      * <pre>
111      * &lt;code&gt;
112      *     defaults.put(&quot;Button.States&quot;, &quot;Enabled, Foo, Disabled&quot;);
113      *     defaults.put(&quot;Button.Foo&quot;, new FooState(&quot;Foo&quot;));
114      * &lt;/code&gt;
115      * </pre>
116      *
117      * @param name a simple user friendly name for the state, such as
118      *             "Indeterminate" or "EmbeddedPanel" or "Blurred". It is
119      *             customary to use camel case, with the first letter
120      *             capitalized.
121      */
122     protected State(String name) {
123         this.name = name;
124     }

133     /**
134      * <p>This is the main entry point, called by NimbusStyle.</p>
135      *
136      * <p>There are both custom states and standard states. Standard states
137      * correlate to the states defined in SynthConstants. When a UI delegate
138      * constructs a SynthContext, it specifies the state that the component is
139      * in according to the states defined in SynthConstants. Our NimbusStyle
140      * will then take this state, and query each State instance in the style
141      * asking whether isInState(c, s).</p>
142      *
143      * <p>Now, only the standard states care about the "s" param. So we have
144      * this odd arrangement:</p>
145      *
146      * <ul>
147      *   <li>NimbusStyle calls State.isInState(c, s)</li>
148      *   <li>State.isInState(c, s) simply delegates to State.isInState(c)</li>
149      *   <li><em>EXCEPT</em>, StandardState overrides State.isInState(c, s) and
150      *     returns directly from that method after checking its state, and does
151      *     not call isInState(c) (since it is not needed for standard
152      *     states).</li>
153      * </ul>
154      *
155      * @param  c DOCUMENT ME!
156      * @param  s DOCUMENT ME!
157      *
158      * @return DOCUMENT ME!
159      */
160     public boolean isInState(T c, int s) {
161         return isInState(c);
162     }
163 
164     /**
165      * <p>Gets whether the specified JComponent is in the custom state
166      * represented by this class. <em>This is an extremely performance sensitive
167      * loop.</em> Please take proper precautions to ensure that it executes
168      * quickly.</p>
169      *
170      * <p>Nimbus uses this method to help determine what state a JComponent is
171      * in. For example, a custom State could exist for JProgressBar such that it
172      * would return <code>true</code> when the progress bar is indeterminate.
173      * Such an implementation of this method would simply be:</p>
174      *
175      * <pre>
176      * &lt;code&gt; return c.isIndeterminate();&lt;/code&gt;
177      * </pre>
178      *
179      * @param  c the JComponent to test. This will never be null.
180      *
181      * @return true if <code>c</code> is in the custom state represented by this
182      *         <code>State</code> instance
183      */
184     public abstract boolean isInState(T c);

260         /**
261          * DOCUMENT ME!
262          *
263          * @param  state DOCUMENT ME!
264          *
265          * @return DOCUMENT ME!
266          */
267         private static String toString(int state) {
268             StringBuffer buffer = new StringBuffer();
269 
270             if ((state & SynthConstants.DEFAULT) == SynthConstants.DEFAULT) {
271                 buffer.append("Default");
272             }
273 
274             if ((state & SynthConstants.DISABLED) == SynthConstants.DISABLED) {
275 
276                 if (buffer.length() > 0)
277                     buffer.append("+");
278 
279                 buffer.append("Disabled");
280             }
281 
282             if ((state & SynthConstants.ENABLED) == SynthConstants.ENABLED) {
283 
284                 if (buffer.length() > 0)
285                     buffer.append("+");
286 
287                 buffer.append("Enabled");
288             }
289 
290             if ((state & SynthConstants.FOCUSED) == SynthConstants.FOCUSED) {
291 
292                 if (buffer.length() > 0)
293                     buffer.append("+");
294 
295                 buffer.append("Focused");
296             }
297 
298             if ((state & SynthConstants.MOUSE_OVER) == SynthConstants.MOUSE_OVER) {
299 
300                 if (buffer.length() > 0)
301                     buffer.append("+");
302 
303                 buffer.append("MouseOver");
304             }
305 
306             if ((state & SynthConstants.PRESSED) == SynthConstants.PRESSED) {
307 
308                 if (buffer.length() > 0)
309                     buffer.append("+");
310 
311                 buffer.append("Pressed");
312             }
313 
314             if ((state & SynthConstants.SELECTED) == SynthConstants.SELECTED) {
315 
316                 if (buffer.length() > 0)
317                     buffer.append("+");
318 
319                 buffer.append("Selected");
320             }
321 
322             return buffer.toString();
323         }
