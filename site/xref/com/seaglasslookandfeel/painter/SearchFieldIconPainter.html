20  package com.seaglasslookandfeel.painter;
21  
22  import java.awt.BasicStroke;
23  import java.awt.Color;
24  import java.awt.Graphics2D;
25  import java.awt.RenderingHints;
26  import java.awt.Shape;
27  
28  import javax.swing.JComponent;
29  
30  /**
31   * Search field icon implementation.
32   */
33  public final class SearchFieldIconPainter extends AbstractRegionPainter {
34      public static enum Which {
35          FIND_ICON_DISABLED, FIND_ICON_ENABLED, FIND_ICON_ENABLED_POPUP,
36  
37          CANCEL_ICON_DISABLED, CANCEL_ICON_ENABLED, CANCEL_ICON_PRESSED,
38      }
39  
40      private Color        searchIconColor        = decodeColor("seaGlassSearchIcon");
41      private Color        cancelIconColor        = decodeColor("seaGlassCancelIcon");
42      private Color        cancelIconPressedColor = decodeColor("seaGlassCancelIconPressed");
43  
44      private Which        state;
45      private PaintContext ctx;
46  
47      public SearchFieldIconPainter(Which state) {
48          super();
49          this.state = state;
50          this.ctx = new PaintContext(AbstractRegionPainter.PaintContext.CacheMode.FIXED_SIZES);
51      }
52  
53      /**
54       * {@inheritDoc}
55       */
56      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
57          switch (state) {
58          case FIND_ICON_ENABLED:
59              paintSearchGlass(g, 0, 0, width, height, false);
60              break;
61          case FIND_ICON_ENABLED_POPUP:
62              paintSearchGlass(g, 0, 0, width, height, true);
63              break;
64          case CANCEL_ICON_ENABLED:
65              paintCancelIcon(g, 0, 0, width, height, cancelIconColor);
66              break;
67          case CANCEL_ICON_PRESSED:
68              paintCancelIcon(g, 0, 0, width, height, cancelIconPressedColor);
69              break;
70          }
71      }
72  
73      /**
74       * {@inheritDoc}
75       */
76      protected PaintContext getPaintContext() {
77          return ctx;
78      }
79  
80      private void paintSearchGlass(Graphics2D g, int x, int y, int width, int height, boolean hasPopup) {
81          g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
82  
83          final int glassX = 2;
84          final int glassY = 2;
85          final int glassRadius = 8;
86  
87          g.setStroke(new BasicStroke(2));
88          g.setColor(searchIconColor);
89          g.drawOval(glassX, glassY, glassRadius, glassRadius);
90  
91          final int handleOffset = 7;
92          final int handleX = glassX + handleOffset;
93          final int handleY = glassY + handleOffset;
94          final int handleLength = 3;
95  
96          g.drawLine(handleX, handleY, handleX + handleLength, handleY + handleLength);
97  
98          if (hasPopup) {
99              final int popupX = glassX + glassRadius + 3;
100             final int popupY = glassY + 3;
101 
102             Shape s = shapeGenerator.createArrowDown(popupX, popupY, 7, 4);
103             g.setColor(searchIconColor);
104             g.fill(s);
105         }
106     }
107 
108     private void paintCancelIcon(Graphics2D g, int x, int y, int width, int height, Color color) {
109         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
110 
111         Shape s = shapeGenerator.createCancelIcon(2, 2, width - 4, height - 4);
112         g.setColor(color);
113         g.fill(s);
114     }
115 }
