186     /**
187      * {@inheritDoc}
188      */
189     @Override
190     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
191         switch (state) {
192 
193         case FOREGROUND_DISABLED:
194         case FOREGROUND_ENABLED:
195         case FOREGROUND_PRESSED:
196             paintBackgroundApart(g, width, height);
197             paintDecreaseButtonApart(g, width, height);
198             break;
199 
200         case FOREGROUND_DISABLED_TOGETHER:
201         case FOREGROUND_ENABLED_TOGETHER:
202         case FOREGROUND_PRESSED_TOGETHER:
203             paintBackgroundTogetherDecrease(g, width, height);
204             paintDecreaseButtonTogether(g, width, height);
205             break;
206 
207         case FOREGROUND_INCREASE_DISABLED:
208         case FOREGROUND_INCREASE_ENABLED:
209         case FOREGROUND_INCREASE_PRESSED:
210             paintBackgroundApart(g, width, height);
211             paintIncreaseButtonApart(g, width, height);
212             break;
213 
214         case FOREGROUND_INCREASE_DISABLED_TOGETHER:
215         case FOREGROUND_INCREASE_ENABLED_TOGETHER:
216         case FOREGROUND_INCREASE_PRESSED_TOGETHER:
217             paintBackgroundTogetherIncrease(g, width, height);
218             paintIncreaseButtonTogether(g, width, height);
219             break;
220 
221         case FOREGROUND_CAP:
222             paintBackgroundCap(g, width, height);
223             break;
224         }
225     }

291     /**
292      * DOCUMENT ME!
293      *
294      * @param g               DOCUMENT ME!
295      * @param s               DOCUMENT ME!
296      * @param isIncrease      DOCUMENT ME!
297      * @param buttonsTogether DOCUMENT ME!
298      */
299     private void fillScrollBarButtonInteriorColors(Graphics2D g, Shape s, boolean isIncrease, boolean buttonsTogether) {
300         g.setPaint(getScrollBarButtonBackgroundPaint(s, isIncrease, buttonsTogether));
301         g.fill(s);
302 
303         int width = s.getBounds().width;
304 
305         g.setPaint(getScrollBarButtonLinePaint());
306         g.drawLine(0, 0, width - 1, 0);
307 
308         if (state != Which.FOREGROUND_CAP && buttonsTogether) {
309             int height = s.getBounds().height;
310 
311             g.setPaint(getScrollBarButtonDividerPaint(isIncrease));
312             g.drawLine(width - 1, 1, width - 1, height - 1);
313         }
314     }

418     /**
419      * DOCUMENT ME!
420      *
421      * @param  buttonsTogether DOCUMENT ME!
422      * @param  isIncrease      DOCUMENT ME!
423      *
424      * @return DOCUMENT ME!
425      */
426     private TwoColors getScrollBarButtonBackgroundColors(boolean buttonsTogether, boolean isIncrease) {
427         if (state == Which.FOREGROUND_CAP) {
428             return scrollBarCapColors;
429         } else if (isPressed) {
430             return isIncrease ? scrollBarButtonIncreasePressed : scrollBarButtonDecreasePressed;
431         } else {
432 
433             if (buttonsTogether) {
434                 return isIncrease ? scrollBarButtonIncreaseTogether : scrollBarButtonDecreaseTogether;
435             } else {
436                 return isIncrease ? scrollBarButtonIncreaseApart : scrollBarButtonDecreaseApart;
437             }
438         }
439     }
