81      private static String deriveColor(Color color, float hOffset, float sOffset, float bOffset) {
82          Color dest = deriveColor(color, hOffset, sOffset, bOffset, 0);
83          return String.format("%06x", (dest.getRGB() & 0xFFFFFF));
84      }

86      private static Color deriveColor(Color src, float hOffset, float sOffset, float bOffset, int aOffset) {
87          float[] tmp = Color.RGBtoHSB(src.getRed(), src.getGreen(), src.getBlue(), null);
88  
89          // apply offsets
90          tmp[0] = clamp(tmp[0] + hOffset);
91          tmp[1] = clamp(tmp[1] + sOffset);
92          tmp[2] = clamp(tmp[2] + bOffset);
93          int alpha = clamp(src.getAlpha() + aOffset);
94  
95          return new Color((Color.HSBtoRGB(tmp[0], tmp[1], tmp[2]) & 0xFFFFFF) | (alpha << 24), true);
96      }
