135     public static void main(String[] args) {
136         SwingUtilities.invokeLater(new Runnable() {
137             public void run() {
138                 LookAndFeel laf = new SeaGlassLookAndFeel();
139                 try {
140                     UIManager.setLookAndFeel(laf);
141                 } catch (UnsupportedLookAndFeelException e1) {
142                     e1.printStackTrace();
143                 }
144                 // get defaults
145                 UIDefaults defaults = UIManager.getLookAndFeelDefaults();
146                 // split of components
147                 Map<String, Map<String, Object>> componentDefaults = new HashMap<String, Map<String, Object>>();
148                 Map<String, Object> others = new HashMap<String, Object>();
149                 Object[] defaultKeys = defaults.keySet().toArray();
150                 for (Object keyObj : defaultKeys) {
151                     String key = keyObj.toString();
152                     if (key.contains("MouseOver")) {
153                         continue;
154                     }
155                     boolean matchesComponent = false;
156                     componentloop: for (String componentName : NIMBUS_COMPONENTS) {
157                         if (key.startsWith(componentName + ".") || key.startsWith(componentName + ":")
158                                 || key.startsWith(componentName + "[")) {
159                             Map<String, Object> keys = componentDefaults.get(componentName);
160                             if (keys == null) {
161                                 keys = new HashMap<String, Object>();
162                                 componentDefaults.put(componentName, keys);
163                             }
164                             keys.put(key, defaults.get(key));
165                             matchesComponent = true;
166                             break componentloop;
167                         }
168                     }
169                     if (!matchesComponent) others.put(key, defaults.get(key));
170                 }
171                 // split out primary, secondary colors
172                 Map<String, Object> primaryColors = new HashMap<String, Object>();
173                 Map<String, Object> secondaryColors = new HashMap<String, Object>();
174                 for (Map.Entry<String, Object> entry : others.entrySet()) {
175                     if (NIMBUS_PRIMARY_COLORS.contains(entry.getKey())) {
176                         primaryColors.put(entry.getKey(), (Color) entry.getValue());
177                     }
178                     if (NIMBUS_SECONDARY_COLORS.contains(entry.getKey())) {
179                         secondaryColors.put(entry.getKey(), (Color) entry.getValue());
180                     }
181                 }
182                 for (String key : NIMBUS_PRIMARY_COLORS) {
183                     others.remove(key);
184                 }
185                 for (String key : NIMBUS_SECONDARY_COLORS) {
186                     others.remove(key);
187                 }
188                 // split out UIs
189                 Map<String, Object> uiClasses = new HashMap<String, Object>();
190                 for (Map.Entry<String, Object> entry : others.entrySet()) {
191                     if (entry.getKey().endsWith("UI")) {
192                         uiClasses.put(entry.getKey(), entry.getValue());
193                     }
194                 }
195                 for (String key : uiClasses.keySet()) {
196                     others.remove(key);
197                 }
198                 // write html file
199                 try {
200                     File dir = new File("target/browser");
201                     dir.mkdir();
202                     IMAGES_DIR = new File(dir, "images");
203                     IMAGES_DIR.mkdir();
204                     System.out.println("Outputing to " + dir.getAbsolutePath());
205                     File htmlFile = new File(dir, "seaglass.html");
206                     PrintWriter html = new PrintWriter(htmlFile);
207 
208                     html.println("<html>");
209                     html.println("<head>");
210                     html.println("<link href=\"style.css\" rel=\"stylesheet\" type=\"text/css\" />");
211                     html.println("</head>");
212                     html.println("<body>");
213                     html.println("<h1>Primary Colors</h1>");
214                     printTable(html, primaryColors);
215 
216                     html.println("<h1>Secondary Colors</h1>");
217                     printTable(html, secondaryColors);
218 
219                     html.println("<h1>Components</h1>");
220                     for (Map.Entry<String, Map<String, Object>> entry : componentDefaults.entrySet()) {
221                         html.println("<h2>" + entry.getKey() + "</h2>");
222                         printTable(html, entry.getValue());
223                     }
224 
225                     html.println("<h1>Others</h1>");
226                     printTable(html, others);
227 
228                     html.println("<h1>UI Classes</h1>");
229                     printTable(html, uiClasses);
230 
231                     html.println("</body>");
232                     html.println("</html>");
233 
234                     html.flush();
235                     html.close();
236                 } catch (Exception e) {
237                     e.printStackTrace();
238                 }
239             }
240         });
241     }

243     static void printTable(PrintWriter html, Map<String, Object> map) {
244         List<String> keys = new ArrayList<String>(map.keySet());
245         Collections.sort(keys);
246         html.println("<table>");
247         html.println("<tr><th>Key</th><th>Value</th><th>Preview</th></tr>");
248         for (String key : keys) {
249             printRow(html, key, map.get(key));
250         }
251         html.println("</table>");
252     }

254     static void printRow(PrintWriter html, String key, Object value) {
255         html.println("<tr>");
256         // key
257         html.println("<td width=\"250\"><code>" + key + "</code></td>");
258         if (value instanceof Color) {
259             printColor(html, (Color) value);
260         } else if (value instanceof Font) {
261             printFont(html, (Font) value);
262         } else if (value instanceof Dimension) {
263             printDimension(html, (Dimension) value);
264         } else if (value instanceof Insets) {
265             printInsets(html, (Insets) value);
266         } else if (value instanceof Border) {
267             printBorder(html, (Border) value);
268         } else if (value instanceof Painter) {
269             printPainter(html, (Painter<?>) value);
270         } else if (value instanceof InputMap) {
271             // ignore, not intresting
272             html.println("<td>&nbsp;</td>");
273             html.println("<td>&nbsp;</td>");
274         } else if (value instanceof Icon) {
275             printIcon(html, (Icon) value);
276         } else {
277             html.println("<td>" + (value == null ? "no value" : value.toString()) + "</td>");
278             html.println("<td>&nbsp;</td>");
279         }
280         html.println("</tr>");
281     }

289     static void printPainter(PrintWriter html, Painter<?> painter) {
290         html.println("<td>Painter</td>");
291         int w = 25, h = 25;
292         try {
293             BufferedImage img = new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB);
294             Graphics2D g2 = img.createGraphics();
295             Composite old = g2.getComposite();
296             g2.setComposite(AlphaComposite.Clear);
297             g2.fillRect(0, 0, w, h);
298             g2.setComposite(old);
299             painter.paint(g2, null, w, h);
300             g2.dispose();
301             html.println("<td>" + saveImage(img));
302             html.println("</td>");
303         } catch (Exception e) {
304             // e.printStackTrace();
305             html.println("<td>NPE&nbsp;</td>");
306         }
307     }

309     static void printFont(PrintWriter html, Font font) {
310         String style = "";
311         if (font.isBold() && font.isItalic()) {
312             style = "Bold & Italic";
313         } else if (font.isBold()) {
314             style = "Bold";
315         } else if (font.isItalic()) {
316             style = "Italic";
317         }
318         html.println("<td>Font: " + font.getFamily() + " " + font.getSize() + " " + style + "</td>");
319         int w = 300, h = 30;
320         BufferedImage img = new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB);
321         Graphics2D g2 = img.createGraphics();
322         Composite old = g2.getComposite();
323         g2.setComposite(AlphaComposite.Clear);
324         g2.fillRect(0, 0, w, h);
325         g2.setComposite(old);
326         g2.setColor(Color.BLACK);
327         g2.setFont(font);
328         g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
329         g2.drawString("The quick brown fox jumps over the lazy dog.", 5, 20);
330         g2.dispose();
331         html.println("<td>" + saveImage(img) + "</td>");
332     }

334     static void printInsets(PrintWriter html, Insets insets) {
335         html.println("<td>Insets (" + insets.top + "," + insets.left + "," + insets.bottom + "," + insets.right + ")</pre></td>");

342         g2.fillRect(0, 0, w, h);
343         g2.setComposite(old);
344         g2.setColor(Color.BLACK);
345         g2.drawRect(insets.left, insets.top, 49, 19);
346         g2.setColor(Color.RED);
347         g2.drawRect(0, 0, w - 1, h - 1);
348         g2.dispose();
349         html.println("<td>" + saveImage(img) + "</td>");

376     static void printDimension(PrintWriter html, Dimension dim) {
377         html.println("<td>Dimension (" + dim.width + "," + dim.height + ")</pre></td>");
378         int w = dim.width;
379         int h = dim.height;
380         if (w == 0 || h == 0) {
381             html.println("<td>&nbsp;</td>");
382         } else {
383             BufferedImage img = new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB);
384             Graphics2D g2 = img.createGraphics();
385             Composite old = g2.getComposite();
386             g2.setComposite(AlphaComposite.Clear);
387             g2.fillRect(0, 0, w, h);
388             g2.setComposite(old);
389             g2.setColor(Color.RED);
390             g2.drawRect(0, 0, w - 1, h - 1);
391             g2.dispose();
392             html.println("<td>" + saveImage(img) + "</td>");
393         }

396     static void printIcon(PrintWriter html, Icon icon) {
397         html.println("<td>Icon " + icon.getIconWidth() + " x " + icon.getIconHeight() + "</pre></td>");
398         BufferedImage img = new BufferedImage(icon.getIconWidth(), icon.getIconHeight(), BufferedImage.TYPE_INT_ARGB);
399         Graphics2D g2 = img.createGraphics();
400         Composite old = g2.getComposite();
401         g2.setComposite(AlphaComposite.Clear);
402         g2.fillRect(0, 0, icon.getIconWidth(), icon.getIconHeight());
403         g2.setComposite(old);
404         JButton foo = new JButton();
405         icon.paintIcon(foo, g2, 0, 0);
406         g2.dispose();
407         html.println("<td>" + saveImage(img) + "</td>");
408     }

410     static String saveImage(BufferedImage img) {
411         File imgFile = new File(IMAGES_DIR, "img_" + (IMAGE_COUNT++) + ".png");
412         try {
413             ImageIO.write(img, "png", imgFile);
414         } catch (IOException e) {
415             e.printStackTrace();
416         }
417         return "<img src=\"images/" + imgFile.getName() + "\">";
418     }

420     static String getWebColor(Color color) {
421         String result = "";
422         String num;
423         num = Integer.toHexString(color.getRed());
424         if (num.length() == 1) num = "0" + num;
425         result += num;
426         num = Integer.toHexString(color.getGreen());
427         if (num.length() == 1) num = "0" + num;
428         result += num;
429         num = Integer.toHexString(color.getBlue());
430         if (num.length() == 1) num = "0" + num;
431         result += num;
432         return result;
433     }
