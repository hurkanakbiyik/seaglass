73      private static String deriveColor(Color color1, Color color2) {
74          int dest = deriveARGB(color1, color2, 0.5f);
75          return String.format("%06x", dest & 0xFFFFFF);
76      }

78      private static int deriveARGB(Color color1, Color color2, float midPoint) {
79          int r = color1.getRed() + (int) ((color2.getRed() - color1.getRed()) * midPoint + 0.5f);
80          int g = color1.getGreen() + (int) ((color2.getGreen() - color1.getGreen()) * midPoint + 0.5f);
81          int b = color1.getBlue() + (int) ((color2.getBlue() - color1.getBlue()) * midPoint + 0.5f);
82          int a = color1.getAlpha() + (int) ((color2.getAlpha() - color1.getAlpha()) * midPoint + 0.5f);
83          return ((a & 0xFF) << 24) | ((r & 0xFF) << 16) | ((g & 0xFF) << 8) | (b & 0xFF);
84      }
