37  /**
38   * Sea Glass PasswordField UI delegate.
39   *
40   * <p>Based on SynthPasswordTextFieldUI, but needs to extend SeaGlassTextFieldUI
41   * instead.</p>
42   */
43  public class SeaGlassPasswordFieldUI extends SeaGlassTextFieldUI {
44  
45      /**
46       * Creates a UI for a JPasswordField.
47       *
48       * @param  c the JPasswordField
49       *
50       * @return the UI
51       */
52      public static ComponentUI createUI(JComponent c) {
53          return new SeaGlassPasswordFieldUI();
54      }
55  
56      /**
57       * Fetches the name used as a key to look up properties through the
58       * UIManager. This is used as a prefix to all the standard text properties.
59       *
60       * @return the name ("PasswordField")
61       */
62      protected String getPropertyPrefix() {
63          return "PasswordField";
64      }
65  
66      /**
67       * Creates a view (PasswordView) for an element.
68       *
69       * @param  elem the element
70       *
71       * @return the view
72       */
73      public View create(Element elem) {
74          return new PasswordView(elem);
75      }
76  
77      /**
78       * @see com.seaglasslookandfeel.ui.SeaGlassTextFieldUI#paintBackground(com.seaglasslookandfeel.SeaGlassContext,
79       *      java.awt.Graphics, javax.swing.JComponent)
80       */
81      void paintBackground(SeaGlassContext context, Graphics g, JComponent c) {
82          context.getPainter().paintPasswordFieldBackground(context, g, 0, 0, c.getWidth(), c.getHeight());
83      }
84  
85      /**
86       * @see com.seaglasslookandfeel.ui.SeaGlassTextFieldUI#paintBorder(javax.swing.plaf.synth.SynthContext,
87       *      java.awt.Graphics, int, int, int, int)
88       */
89      public void paintBorder(SynthContext context, Graphics g, int x, int y, int w, int h) {
90          ((SeaGlassContext) context).getPainter().paintPasswordFieldBorder(context, g, x, y, w, h);
91      }
92  
93      /**
94       * @see javax.swing.plaf.basic.BasicTextUI#installKeyboardActions()
95       */
96      protected void installKeyboardActions() {
97          super.installKeyboardActions();
98          ActionMap map = SwingUtilities.getUIActionMap(getComponent());
99  
100         if (map != null && map.get(DefaultEditorKit.selectWordAction) != null) {
101             Action a = map.get(DefaultEditorKit.selectLineAction);
102 
103             if (a != null) {
104                 map.put(DefaultEditorKit.selectWordAction, a);
105             }
106         }
107     }
