90          int state = SeaGlassLookAndFeel.getComponentState(c);

97      public int getBaseline(JComponent c, int width, int height) {
98          if (c == null) {
99              throw new NullPointerException("Component must be non-null");
100         }
101         if (width < 0 || height < 0) {
102             throw new IllegalArgumentException("Width and height must be >= 0");
103         }
104         JLabel label = (JLabel) c;
105         String text = label.getText();
106         if (text == null || "".equals(text)) {
107             return -1;
108         }
109         Insets i = label.getInsets();
110         Rectangle viewRect = new Rectangle();
111         Rectangle textRect = new Rectangle();
112         Rectangle iconRect = new Rectangle();
113         viewRect.x = i.left;
114         viewRect.y = i.top;
115         viewRect.width = width - (i.right + viewRect.x);
116         viewRect.height = height - (i.bottom + viewRect.y);
117 
118         // layout the text and icon
119         SeaGlassContext context = getContext(label);
120         FontMetrics fm = context.getComponent().getFontMetrics(context.getStyle().getFont(context));
121         context.getStyle().getGraphicsUtils(context).layoutText(context, fm, label.getText(), label.getIcon(),
122             label.getHorizontalAlignment(), label.getVerticalAlignment(), label.getHorizontalTextPosition(),
123             label.getVerticalTextPosition(), viewRect, iconRect, textRect, label.getIconTextGap());
124         View view = (View) label.getClientProperty(BasicHTML.propertyKey);
125         int baseline;
126         if (view != null) {
127             baseline = BasicHTML.getHTMLBaseline(view, textRect.width, textRect.height);
128             if (baseline >= 0) {
129                 baseline += textRect.y;
130             }
131         } else {
132             baseline = textRect.y + fm.getAscent();
133         }
134         context.dispose();
135         return baseline;
136     }

144         SeaGlassContext context = getContext(c);
145 
146         SeaGlassLookAndFeel.update(context, g);
147         context.getPainter().paintLabelBackground(context, g, 0, 0, c.getWidth(), c.getHeight());

159     protected void paint(SeaGlassContext context, Graphics g) {
160         JLabel label = (JLabel) context.getComponent();
161         Icon icon = (label.isEnabled()) ? label.getIcon() : label.getDisabledIcon();
162 
163         g.setColor(context.getStyle().getColor(context, ColorType.TEXT_FOREGROUND));
164         g.setFont(style.getFont(context));
165         context.getStyle().getGraphicsUtils(context).paintText(context, g, label.getText(), icon, label.getHorizontalAlignment(),
166             label.getVerticalAlignment(), label.getHorizontalTextPosition(), label.getVerticalTextPosition(),
167             label.getIconTextGap(), label.getDisplayedMnemonicIndex(), 0);
168     }
