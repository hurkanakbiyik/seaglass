78      @Override
79      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
80          switch (state) {
81  
82          case BACKGROUND_DISABLED:
83          case BACKGROUND_ENABLED:
84          case BACKGROUND_ENABLED_WINDOWNOTFOCUSED:
85              paintCloseEnabled(g, c, width, height);
86              break;
87  
88          case BACKGROUND_MOUSEOVER:
89              paintCloseHover(g, c, width, height);
90              break;
91  
92          case BACKGROUND_PRESSED:
93          case BACKGROUND_PRESSED_WINDOWNOTFOCUSED:
94              paintClosePressed(g, c, width, height);
95              break;
96          }
97      }

143     /**
144      * Paint the background of the button using the specified colors.
145      *
146      * @param g      the Graphics2D context to paint with.
147      * @param c      the component.
148      * @param width  the width of the component.
149      * @param height the height of the component.
150      * @param colors the color set to use to paint the button.
151      */
152     private void paintClose(Graphics2D g, JComponent c, int width, int height, ButtonColors colors) {
153         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
154         Shape s = decodeInterior(width, height);
155 
156         g.setPaint(decodeCloseGradient(s, colors.interiorTop, colors.interiorBottom));
157         g.fill(s);
158         s = decodeEdge(width, height);
159         g.setColor(colors.edge);
160         g.draw(s);
161         s = decodeShadow(width, height);
162         g.setColor(colors.shadow);
163         g.draw(s);
164         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
165         g.setColor(colors.top);
166         g.drawLine(0, 0, width - 2, 0);
167         g.setColor(colors.left);
168         g.drawLine(0, 1, 0, height - 2);
169         s = decodeMarkInterior(width, height);
170         g.setColor(colors.markInterior);
171         g.fill(s);
172         s = decodeMarkBorder(width, height);
173         g.setColor(colors.markBorder);
174         g.draw(s);
175     }

177     /**
178      * Create the gradient for the close button.
179      *
180      * @param  s      the shape to fill.
181      * @param  top    the top color.
182      * @param  bottom the bottom color.
183      *
184      * @return the gradient.
185      */
186     private Paint decodeCloseGradient(Shape s, Color top, Color bottom) {
187         Rectangle r      = s.getBounds();
188         int       width  = r.width;
189         int       height = r.height;
190 
191         return createGradient(r.x + width / 2, r.y, r.x + width / 2, r.y + height - 1, new float[] { 0f, 1f }, new Color[] { top, bottom });
192     }

194     /**
195      * Create the interior of the button.
196      *
197      * @param  width  the width.
198      * @param  height the height.
199      *
200      * @return the shape.
201      */
202     private Shape decodeInterior(int width, int height) {
203         path.reset();
204         path.moveTo(1, 1);
205         path.lineTo(width - 2, 1);
206         path.lineTo(width - 2, height - 3);
207         path.lineTo(width - 3, height - 2);
208         path.lineTo(1, height - 2);
209         path.closePath();
210 
211         return path;
212     }

329     /**
330      * Extend Path2D to have a pointAt method to draw a single point.
331      */
332     private static class MyPath2D extends Path2D.Double {
333         private static final long serialVersionUID = -3577694636085901945L;
334 
335         /**
336          * Draw a single point at the (x,y) coordinate specified.
337          *
338          * @param x the x coordinate to draw the point at.
339          * @param y the y coordinate to draw the point at.
340          */
341         public void pointAt(int x, int y) {
342             moveTo(x, y);
343             lineTo(x, y);
344         }
345     }
