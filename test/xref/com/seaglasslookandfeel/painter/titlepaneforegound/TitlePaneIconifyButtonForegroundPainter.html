20  package com.seaglasslookandfeel.painter.titlepaneforegound;

22  import java.awt.Color;
23  import java.awt.Graphics2D;
24  import java.awt.RenderingHints;
25  
26  import javax.swing.JComponent;
27  
28  /**
29   * Painter for iconify foreground shape.
30   *
31   * @author Kathryn Huxtable
32   */
33  public class TitlePaneIconifyButtonForegroundPainter extends TitlePaneButtonForegroundPainter {
34  
35      /**
36       * {@inheritDoc}
37       */
38      protected void paint(Graphics2D g, JComponent c, int width, int height, Color border, Color corner, Color interior) {
39          int top  = (height - 2) / 2;
40          int left = (width - 3) / 2 - 4;
41  
42          g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
43  
44          g.setColor(interior);
45          g.fillRect(left + 1, top + 1, 10, 3);
46  
47          g.setColor(border);
48          g.drawLine(left + 1, top + 0, left + 9, top + 0);
49          g.drawLine(left + 1, top + 4, left + 9, top + 4);
50          g.drawLine(left + 0, top + 1, left + 0, top + 3);
51          g.drawLine(left + 10, top + 1, left + 10, top + 3);
52      }
53  }
